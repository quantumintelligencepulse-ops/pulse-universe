def human_time(iso_str: str) -> str:
    try:
        dt = datetime.fromisoformat(iso_str.replace("Z", "+00:00"))
        return dt.strftime("%Y-%m-%d %H:%M")
    except Exception:
        return iso_str


def parse_dt(iso_str: str) -> datetime:
    try:
        return datetime.fromisoformat(iso_str.replace("Z", "+00:00"))
    except Exception:
        return datetime.utcnow()


def slugify(name: str, max_len: int = 80) -> str:
    name = unicodedata.normalize("NFKD", name)
    name = "".join(c for c in name if not unicodedata.combining(c))
    slug_chars = []
    for c in name:
        if c.isalnum():
            slug_chars.append(c.lower())
        else:
            slug_chars.append("-")
    slug = "".join(slug_chars)
    slug = re.sub(r"-{2,}", "-", slug).strip("-")
    if len(slug) > max_len:
        slug = slug[:max_len].rstrip("-")
    if not slug:
        slug = "item"
    return slug


def clean_text(text: str) -> str:
    text = text.replace("\r", " ").replace("\n", " ")
    text = re.sub(r"\s+", " ", text)
    return text.strip()


def sanitize_source(text: str) -> str:
    replacements = [
        ("Banking With Billy", "Equity Investing Alert"),
        ("BANKING WITH BILLY", "Equity Investing Alert"),
        ("Banking with Billy", "Equity Investing Alert"),
        ("Banking With billy", "Equity Investing Alert"),
    ]
    for old, new in replacements:
        text = text.replace(old, new)
    text = text.replace("•  ", " ").replace("• ", " ")
    return text.strip()


def extract_tickers(text: str):
    tickers = []
    for word in text.split():
        w = re.sub(r"[^A-Z]", "", word)
        if 2 <= len(w) <= 5 and w.isupper():
            tickers.append(w)
    return sorted(set(tickers))


def extract_numbers(text: str):
    return re.findall(r"\b\d+(\.\d+)?%?\b", text)


def hash_int(s: str) -> int:
    return int(hashlib.sha256(s.encode("utf-8")).hexdigest(), 16)

def is_video_url(url: str) -> bool:
    if not url:
        return False
    lower = url.lower()
    if "video.twimg.com" in lower:
        return True
    if any(lower.split("?", 1)[0].endswith(ext) for ext in VIDEO_EXTENSIONS):
        return True
    return False

def pick_pool_key(tags):
    tags_lower = [t.lower() for t in tags]
    if any(t in tags_lower for t in ["crypto", "rwa", "defi"]):
        return "crypto"
    if any(t in tags_lower for t in ["markets", "fundamentals", "general"]):
        return "markets"
    if any(t in tags_lower for t in ["geopolitics", "politics"]):
        return "geopolitics"
    if any(t in tags_lower for t in ["ai", "technology", "tech"]):
        return "ai"
    if any(t in tags_lower for t in ["energy"]):
        return "energy"
    if any(t in tags_lower for t in ["commodities"]):
        return "commodities"
    if any(t in tags_lower for t in ["macro", "central-banks"]):
        return "macro"
    return "default"

def pick_image_from_pool(tags, raw: str, used_images_set: set) -> str:
    key = pick_pool_key(tags)
    pool = IMAGE_POOLS.get(key, IMAGE_POOLS["default"])
    for candidate in pool:
        if candidate not in used_images_set:
            used_images_set.add(candidate)
            return candidate
    h = hash_int(raw)
    idx = h % len(pool)
    url = pool[idx]
    used_images_set.add(url)
    return url

def duckduckgo_image_search(query: str, timeout_sec: float = 5.0) -> str | None:
    try:
        if not query.strip():
            return None
        ensure_http_deps()
        import requests
        from bs4 import BeautifulSoup

        params = {"q": query.strip(), "iax": "images", "ia": "images"}
        resp = requests.get("https://duckduckgo.com/", params=params, timeout=timeout_sec)
        if resp.status_code != 200:
            return None
        html = resp.text
        matches = re.findall(r'"image":"(https:[^"]+)"', html)
        if matches:
            return matches[0]
        soup = BeautifulSoup(html, "html.parser")
        for img in soup.find_all("img"):
            src = img.get("src") or ""
            if src.startswith("http"):
                return src
        return None
    except Exception:
        return None

def infer_region(raw: str) -> str:
    t = raw.lower()
    if any(k in t for k in ["eu", "european", "germany", "france", "hungary", "italy", "uk"]):
        return "Europe"
    if any(k in t for k in ["us ", "u.s.", "united states", "america"]):
        return "United States"
    if any(k in t for k in ["china", "beijing", "shanghai"]):
        return "China"
    if any(k in t for k in ["japan", "tokyo", "boj"]):
        return "Japan"
    if any(k in t for k in ["middle east", "iran", "israel", "gaza", "saudi"]):
        return "Middle East"
    return "Global"

def build_headline(raw: str, tickers, tags) -> str:
    base = clean_text(raw)
    if len(base) > 110:
        base = base[:107].rstrip() + "..."
    if tickers:
        return f"Equity Alert: {', '.join(tickers)} – {base}"
    if "geopolitics" in tags:
        return f"Geopolitics Briefing: {base}"
    if "crypto" in tags:
        return f"Crypto Wire: {base}"
    if "macro" in tags or "central-banks" in tags:
        return f"Macro Watch: {base}"
    if "deals" in tags:
        return f"Deal Watch: {base}"
    if "video" in tags:
        return f"Video Briefing: {base}"
    return f"Market Wire: {base}"

        deck = deck[:257].rstrip() + "..."
    return deck


def summarize_context_for_article(current_raw: str, current_tickers, current_tags, context_articles):
    if not context_articles:
        return ""

    same_ticker = []
    same_tags = []
    for a in context_articles[-200:]:
        if set(current_tickers) & set(a.get("tickers", [])):
            same_ticker.append(a)
        elif set(current_tags) & set(a.get("tags", [])):
            same_tags.append(a)

    lines = []

    if same_ticker:
        same_ticker_sorted = sorted(same_ticker, key=lambda x: parse_dt(x["created_at"]), reverse=True)[:6]
        headlines = [clean_text(x["headline"]) for x in same_ticker_sorted]
        lines.append(
            f"Recent Equity Network coverage around {', '.join(current_tickers)} includes: "
            + "; ".join(headlines) + "."
        )

    if same_tags and not same_ticker:
        same_tags_sorted = sorted(same_tags, key=lambda x: parse_dt(x["created_at"]), reverse=True)[:6]
        headlines = [clean_text(x["headline"]) for x in same_tags_sorted]
        tag_label = ", ".join(sorted(set(current_tags)))
        lines.append(
            f"This headline sits inside a broader {tag_label} storyline on the Equity Network wire. "
            "Recent related pieces: "
            + "; ".join(headlines) + "."
        )

    if not lines:
        return ""

    return " ".join(lines)

    breaking_flag = is_breaking(raw)
    created_iso = msg.created_at.replace(microsecond=0).isoformat() + "Z"
    article_id = str(msg.id)

    base_slug = slugify(headline, max_len=60)
    article_slug = f"{article_id}-{base_slug}"
    if len(article_slug) > 100:
        article_slug = f"{article_id}-{hash_int(headline) % 10_000_000}"

    author_name = "Equity Network AI Desk"

    section_name = CHANNEL_LABELS.get(msg.channel.id, "Equity Wire")

    article = {
        "id": article_id,
        "slug": article_slug,
        "headline": headline,
        "deck": deck,
        "body": body,
        "raw": raw,
        "author_name": author_name,
        "section_name": section_name,
        "created_at": created_iso,
        "tickers": tickers,
        "numbers": numbers,
        "tags": tags,
        "hero_image": hero_image,
        "breaking": breaking_flag,
        "video_url": video_url,
    }

    update_knowledge_from_article(article)
    return article

def build_nav_html(all_articles):
    tag_counts = defaultdict(int)
    for a in all_articles:
        for t in a["tags"]:
            tag_counts[t] += 1

    top_tags = sorted(tag_counts.items(), key=lambda x: x[0])
    top_tags = sorted(top_tags, key=lambda x: x[1], reverse=True)
    top_tags = [t for t, _ in top_tags if t != "general"][:12]

    static_sections = [
        ("Home", f"{BASE_URL}/index.html"),
        ("Markets", f"{BASE_URL}/topic/{slugify('markets')}.html"),
        ("Macro", f"{BASE_URL}/topic/{slugify('macro')}.html"),
        ("Geopolitics", f"{BASE_URL}/topic/{slugify('geopolitics')}.html"),
        ("Crypto", f"{BASE_URL}/topic/{slugify('crypto')}.html"),
        ("AI", f"{BASE_URL}/topic/{slugify('ai')}.html"),
        ("Deals", f"{BASE_URL}/topic/{slugify('deals')}.html"),
        ("Video", f"{BASE_URL}/topic/{slugify('video')}.html"),
    ]

    nav_items = []
    for label, href in static_sections:
        nav_items.append(f'<a class="nav-link" href="{href}">{label}</a>')

    if top_tags:
        nav_items.append('<div class="nav-section-label">Active topics</div>')
        for t in top_tags:
            nav_items.append(
                f'<a class="nav-link nav-tag" href="{BASE_URL}/topic/{slugify(t)}.html">{t}</a>'

    nav_items.append('<div class="nav-section-label">Wire access</div>')

# ---------- STATIC EXPORT ----------

def compute_related_articles(all_articles):
    related_map = {}
    for a in all_articles:
        candidates = []
        seen_ids = set()
        for b in all_articles:
            if a["id"] == b["id"]:
                continue
            if b["id"] in seen_ids:
                continue
            score = 0
            if set(a["tickers"]) & set(b["tickers"]):
                score += 3
            if set(a["tags"]) & set(b["tags"]):
                score += 2
            if a["section_name"] == b["section_name"]:
                score += 1
            if score > 0:
                candidates.append((score, b))
                seen_ids.add(b["id"])
        candidates.sort(key=lambda x: (-x[0], parse_dt(x[1]["created_at"])))
        related_map[a["id"]] = [c[1] for c in candidates[:MAX_RELATED]]
    return related_map


def export_index():
    if not articles:
        return
    latest = sorted(articles, key=lambda a: parse_dt(a["created_at"]), reverse=True)[:MAX_GLOBAL_FEED]
    html = render_index_page(latest, articles)
    (SITE_DIR / "index.html").write_text(html, encoding="utf-8")


def export_article_pages():
    if not articles:
        return
    related_map = compute_related_articles(articles)
    for a in articles:
        rel = related_map.get(a["id"], [])
        html = render_article_page(a, rel, articles)
        path = SITE_DIR / "news" / f"{a['slug']}.html"
        path.write_text(html, encoding="utf-8")


def export_topic_pages():
    if not articles:
        return
    by_tag = defaultdict(dict)  # tag -> {id: article}
    for a in articles:
        for t in a["tags"]:
            by_tag[t][a["id"]] = a
    for tag, id_map in by_tag.items():
        tag_articles = list(id_map.values())
        tag_articles_sorted = sorted(tag_articles, key=lambda a: parse_dt(a["created_at"]), reverse=True)
        html = render_topic_page(tag, tag_articles_sorted, articles)
        path = SITE_DIR / "topic" / f"{slugify(tag)}.html"
        path.write_text(html, encoding="utf-8")


def export_sitemap():
    urls = [
        f"<url><loc>{BASE_URL}/index.html</loc></url>",
        f"<url><loc>{BASE_URL}/sitemap.xml</loc></url>",
    ]
    for a in articles:
        urls.append(f"<url><loc>{BASE_URL}/news/{a['slug']}.html</loc></url>")
    tags = set()
    for a in articles:
        for t in a["tags"]:
            tags.add(t)
    for t in tags:
        urls.append(f"<url><loc>{BASE_URL}/topic/{slugify(t)}.html</loc></url>")
    xml = (
        '<?xml version="1.0" encoding="UTF-8"?>'
        '<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">'
        + "".join(urls)
        + "</urlset>"
    )
    (SITE_DIR / "sitemap.xml").write_text(xml, encoding="utf-8")


def export_rss():
    if not articles:
        return
    latest = sorted(articles, key=lambda a: parse_dt(a["created_at"]), reverse=True)[:MAX_RSS_ITEMS]
    items = []
    for a in latest:
        link = f"{BASE_URL}/news/{a['slug']}.html"
        pubdate = parse_dt(a["created_at"]).strftime("%a, %d %b %Y %H:%M:%S GMT")
        desc = a["deck"]
        items.append(
            "<item>"
            f"<title><![CDATA[{a['headline']}]]></title>"
            f"<link>{link}</link>"
            f"<guid>{link}</guid>"
            f"<pubDate>{pubdate}</pubDate>"
            f"<description><![CDATA[{desc}]]></description>"
            "</item>"
        )
    rss = (
        '<?xml version="1.0" encoding="UTF-8"?>'
        "<rss version=\"2.0\"><channel>"
        f"<title>{SITE_NAME} AI Wire</title>"
        f"<link>{BASE_URL}/index.html</link>"
        f"<description>{SITE_TAGLINE}</description>"
        + "".join(items)
        + "</channel></rss>"
    )
    (SITE_DIR / "feed.xml").write_text(rss, encoding="utf-8")


def export_robots():
    txt = (
        "User-agent: *\n"
        "Allow: /\n"
        f"Sitemap: {BASE_URL}/sitemap.xml\n"
    )
    (SITE_DIR / "robots.txt").write_text(txt, encoding="utf-8")


def export_all():
    ensure_site_dirs()
    export_index()
    export_article_pages()
    export_topic_pages()
    export_sitemap()
    export_rss()
    export_robots()
    export_knowledge()
    print("Static export complete.")

@bot.event
async def on_ready():
    print("Bot connected.")

    load_knowledge()

    guild = bot.get_guild(GUILD_ID)
    if guild is None:
        print("Guild not found.")
        return

    msgs_articles = []
    global used_images
    used_images.clear()

    for cid in CHANNEL_IDS:
        channel = guild.get_channel(cid)
        if channel is None or not isinstance(channel, discord.TextChannel):
            print(f"Channel {cid} not found or not text.")
            continue

        print(f"Syncing channel {channel.name} ({cid})...")
        try:
            async for msg in channel.history(limit=MAX_MESSAGES_PER_CHANNEL, oldest_first=True):
                if not msg.content:
                    continue
                article = build_article_from_message(
                    msg,
                    channel.name,
                    msgs_articles,
                    used_images,
                    use_live_images=False,
                )
                if article is None:
                    continue
                msgs_articles.append(article)
                await asyncio.sleep(0.01)
        except Exception as e:
            print(f"Error during history scan for channel {cid}: {e}")

    msgs_sorted = sorted(msgs_articles, key=lambda a: parse_dt(a["created_at"]))
    global articles
    articles = msgs_sorted[-MAX_MESSAGES_PER_CHANNEL * len(CHANNEL_IDS):]

    export_all()
    print("Initial sync complete.")


@bot.event
async def on_message(message: discord.Message):
    if message.guild is None or message.guild.id != GUILD_ID:
        return
    if not isinstance(message.channel, discord.TextChannel):
        return
    if message.channel.id not in CHANNEL_IDS:
        return
    if not message.content:
        return

    global used_images
    try:
        article = build_article_from_message(
            message,
            message.channel.name,
            articles,
            used_images,
            use_live_images=True,
        )
    except Exception as e:
        print(f"Error building article from live message: {e}")
        return

    if article is None:
        return

    articles.append(article)
    articles_sorted = sorted(articles, key=lambda a: parse_dt(a["created_at"]))
    articles[:] = articles_sorted[-MAX_MESSAGES_PER_CHANNEL * len(CHANNEL_IDS):]

    export_all()
    await bot.process_commands(message)
