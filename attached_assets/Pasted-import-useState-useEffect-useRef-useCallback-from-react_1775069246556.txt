import { useState, useEffect, useRef, useCallback } from "react";
import { motion, AnimatePresence } from "framer-motion";
import { base44 } from "@/api/base44Client";
import { getRelevantSources } from "./OmegaSources";
import { ArrowLeft, CheckCircle2, Circle, Loader2 } from "lucide-react";

function ResPanel({ title, items, icon }) {
  return (
    <motion.div initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }}
      className="rounded-xl border border-border bg-card/30 p-4">
      <div className="flex items-center gap-2 mb-3">
        <span>{icon}</span>
        <span className="text-xs font-semibold">{title}</span>
      </div>
      <div className="space-y-1.5">
        {items?.length > 0 ? items.slice(0, 5).map((item, i) => (
          <div key={i} className="text-xs text-muted-foreground flex items-start gap-1.5">
            <span className="text-primary/60 mt-0.5 shrink-0">·</span>
            <span className="line-clamp-2">{typeof item === "string" ? item : JSON.stringify(item)}</span>
          </div>
        )) : (
          <div className="flex items-center gap-2">
            <span className="w-3 h-0.5 bg-muted rounded animate-pulse" />
            <span className="w-8 h-0.5 bg-muted rounded animate-pulse" />
            <span className="w-5 h-0.5 bg-muted rounded animate-pulse" />
          </div>
        )}
      </div>
    </motion.div>
  );
}

const STEPS = [
  "Researching idea across internet",
  "Finding similar apps & market data",
  "Scanning 100 Omega sources",
  "Generating full application with AI",
  "Build complete!",
];

export default function BuildView({ appId, onComplete, onBack }) {
  const [app, setApp] = useState(null);
  const [step, setStep] = useState(0);
  const [logs, setLogs] = useState([]);
  const [research, setResearch] = useState(null);
  const [similarApps, setSimilarApps] = useState(null);
  const [omegaSources, setOmegaSources] = useState([]);
  const logsRef = useRef(null);
  const hasRun = useRef(false);

  const log = useCallback((msg, type = "info") => {
    setLogs((p) => [...p, { msg, type, id: Date.now() + Math.random() }]);
    setTimeout(() => logsRef.current?.scrollTo({ top: 99999, behavior: "smooth" }), 50);
  }, []);

  useEffect(() => {
    if (hasRun.current) return;
    hasRun.current = true;
    base44.entities.GeneratedApp.list("-created_date", 200).then((data) => {
      const found = data.find((a) => a.id === appId);
      if (found) { setApp(found); runPipeline(found); }
    });
  }, [appId]);

  const runPipeline = async (a) => {
    log("◆ ForgeAI Build Engine initializing...", "accent");
    log(`Analyzing: "${a.prompt}"`, "info");
    log(`Project type: ${a.project_type || "fullstack"}`, "info");
    setStep(0);

    // STEP 1 — Research
    const researchResult = await base44.integrations.Core.InvokeLLM({
      prompt: `You are an elite product researcher. Research this app/SaaS idea thoroughly: "${a.prompt}"\n\nProject type: ${a.project_type || "fullstack"}\n\nProvide:\n1. What this product does and why it matters\n2. Key features (at least 10)\n3. Target audience and use cases\n4. Common UI/UX patterns for this type of app\n5. Technical requirements (languages, frameworks, APIs)\n6. Market trends and opportunities\n7. Monetization strategies\n8. Data models needed\n\nBe thorough. Return JSON.`,
      add_context_from_internet: true,
      model: "gemini_3_flash",
      response_json_schema: {
        type: "object",
        properties: {
          summary: { type: "string" },
          key_features: { type: "array", items: { type: "string" } },
          target_audience: { type: "string" },
          ux_patterns: { type: "array", items: { type: "string" } },
          technical_requirements: { type: "array", items: { type: "string" } },
          market_trends: { type: "array", items: { type: "string" } },
          data_models: { type: "array", items: { type: "string" } },
        },
      },
    });

    setResearch(researchResult);
    log(`✓ Research complete — ${researchResult.key_features?.length || 0} features identified`, "success");
    researchResult.key_features?.slice(0, 4).forEach((f) => log(`  → ${f}`, "dim"));
    await base44.entities.GeneratedApp.update(a.id, { research_data: JSON.stringify(researchResult), status: "analyzing" });

    // STEP 2 — Similar apps
    setStep(1);
    log("◆ Cross-referencing app stores & web apps...", "accent");

    const appsResult = await base44.integrations.Core.InvokeLLM({
      prompt: `Find real apps similar to: "${a.prompt}"\n\nSearch for:\n1. Top Google Play Store apps\n2. Popular web apps / SaaS products\n3. Key differentiators\n\nReturn JSON with google_play_apps (name, description, rating, differentiator) and web_apps (name, description).`,
      add_context_from_internet: true,
      model: "gemini_3_flash",
      response_json_schema: {
        type: "object",
        properties: {
          google_play_apps: { type: "array", items: { type: "object", properties: { name: { type: "string" }, description: { type: "string" }, rating: { type: "string" }, differentiator: { type: "string" } } } },
          web_apps: { type: "array", items: { type: "object", properties: { name: { type: "string" }, description: { type: "string" } } } },
        },
      },
    });

    setSimilarApps(appsResult);
    log(`✓ Found ${appsResult.google_play_apps?.length || 0} Play Store apps, ${appsResult.web_apps?.length || 0} web apps`, "success");
    await base44.entities.GeneratedApp.update(a.id, { similar_apps: JSON.stringify(appsResult) });

    // STEP 3 — Omega sources
    setStep(2);
    const matched = getRelevantSources(a.prompt, 20);
    setOmegaSources(matched);
    log(`◆ Omega scan: ${matched.length} sources matched`, "accent");
    matched.slice(0, 5).forEach((s) => log(`  ⚡ ${s.name} [${s.cat}]`, "dim"));

    const codeResult = await base44.integrations.Core.InvokeLLM({
      prompt: `Find open-source resources for: "${a.prompt}"\n\nOmega sources:\n${matched.slice(0, 10).map((s) => `- ${s.name}: ${s.desc}`).join("\n")}\n\nReturn JSON: repositories (name,url,description,stars), libraries (name,purpose), apis (name,description).`,
      add_context_from_internet: true,
      model: "gemini_3_flash",
      response_json_schema: {
        type: "object",
        properties: {
          repositories: { type: "array", items: { type: "object", properties: { name: { type: "string" }, url: { type: "string" }, description: { type: "string" }, stars: { type: "string" } } } },
          libraries: { type: "array", items: { type: "object", properties: { name: { type: "string" }, purpose: { type: "string" } } } },
          apis: { type: "array", items: { type: "object", properties: { name: { type: "string" }, description: { type: "string" } } } },
        },
      },
    });

    log(`✓ ${codeResult.repositories?.length || 0} repos + ${codeResult.libraries?.length || 0} libs found`, "success");
    await base44.entities.GeneratedApp.update(a.id, { open_source_refs: JSON.stringify({ ...codeResult, omega_sources: matched }) });

    matched.slice(0, 6).forEach((src) => {
      base44.entities.ResourceLibrary.create({ name: src.name, url: src.url, category: src.cat, description: src.desc, source_build_id: a.id, usage_count: 1 }).catch(() => {});
    });

    // STEP 4 — Build
    setStep(3);
    log("◆ Generating full application code...", "accent");
    log("  Synthesizing research + market data + Omega sources...", "dim");

    const buildResult = await base44.integrations.Core.InvokeLLM({
      prompt: `You are the world's best full-stack developer. Build a COMPLETE, FULLY FUNCTIONAL single-page web application.

IDEA: "${a.prompt}"
PROJECT TYPE: ${a.project_type || "fullstack"}

RESEARCH:
${JSON.stringify(researchResult).slice(0, 1500)}

SIMILAR APPS:
${JSON.stringify(appsResult).slice(0, 800)}

OMEGA SOURCES:
${matched.slice(0, 8).map((s) => `- ${s.name}: ${s.desc}`).join("\n")}

REQUIREMENTS:
1. COMPLETE functional app — every button, form, interaction WORKS
2. Beautiful premium dark design — gradients, glassmorphism, micro-animations
3. Responsive — mobile and desktop
4. localStorage for state persistence
5. Rich demo data
6. Smooth CSS animations
7. Clean vanilla JS (no external dependencies)
8. Production-quality typography and spacing
9. Loading states, empty states, error states
10. At minimum 10 core features
11. If SaaS: include auth UI, dashboard, settings, sidebar nav
12. Proper form validation and user feedback

Return JSON: { app_name, app_description, app_type, html, css, js }`,
      model: "claude_sonnet_4_6",
      response_json_schema: {
        type: "object",
        properties: {
          app_name: { type: "string" },
          app_description: { type: "string" },
          app_type: { type: "string" },
          html: { type: "string" },
          css: { type: "string" },
          js: { type: "string" },
        },
      },
    });

    log(`✓ App "${buildResult.app_name}" generated!`, "success");

    await base44.entities.GeneratedApp.update(a.id, {
      generated_html: buildResult.html,
      generated_css: buildResult.css,
      generated_js: buildResult.js,
      app_name: buildResult.app_name,
      app_description: buildResult.app_description,
      app_type: buildResult.app_type,
      status: "complete",
    });

    await base44.entities.BuildMemory.create({
      build_id: a.id,
      prompt: a.prompt,
      app_type: buildResult.app_type,
      patterns_learned: JSON.stringify(researchResult?.ux_patterns || []),
      resources_used: JSON.stringify(matched.map((s) => s.name)),
      success_score: 100,
      version: 1,
    }).catch(() => {});

    setStep(4);
    log("◆ Build complete! Your app is ready.", "accent");
    setTimeout(onComplete, 1500);
  };

  const typeColors = { info: "text-muted-foreground", success: "text-primary", accent: "text-accent", dim: "text-muted-foreground/60" };

  return (
    <div className="max-w-5xl mx-auto px-4 sm:px-6 pt-8 pb-20">
      <button onClick={onBack} className="text-xs text-muted-foreground hover:text-foreground mb-6 flex items-center gap-1.5 transition-colors">
        <ArrowLeft className="w-3.5 h-3.5" /> Back
      </button>

      <div className="mb-7">
        <div className="flex items-center gap-2 mb-1">
          <Loader2 className="w-4 h-4 text-accent animate-spin" />
          <span className="text-xs font-mono text-accent uppercase tracking-widest">Building</span>
        </div>
        <h1 className="text-xl sm:text-2xl font-bold truncate">{app?.prompt || "Loading..."}</h1>
        <p className="text-xs text-muted-foreground mt-1">Project type: {app?.project_type || "fullstack"}</p>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-5 gap-5">
        <div className="lg:col-span-2 space-y-4">
          {/* Pipeline steps */}
          <div className="rounded-xl border border-border bg-card/30 p-5">
            <p className="text-xs font-mono text-muted-foreground uppercase tracking-widest mb-4">Pipeline</p>
            {STEPS.map((s, i) => (
              <motion.div key={i} initial={{ opacity: 0, x: -10 }} animate={{ opacity: 1, x: 0 }} transition={{ delay: i * 0.1 }}
                className="flex items-center gap-3 mb-3.5 last:mb-0">
                <div className={`w-6 h-6 rounded-full border flex items-center justify-center transition-all duration-500 ${
                  i < step ? "bg-primary/20 border-primary" : i === step ? "border-accent bg-accent/10" : "border-border"
                }`}>
                  {i < step ? <CheckCircle2 className="w-3.5 h-3.5 text-primary" /> :
                    i === step ? <Loader2 className="w-3 h-3 text-accent animate-spin" /> :
                    <Circle className="w-3 h-3 text-muted-foreground/30" />}
                </div>
                <span className={`text-sm ${i < step ? "text-primary" : i === step ? "text-foreground" : "text-muted-foreground/40"}`}>{s}</span>
              </motion.div>
            ))}
          </div>

          {/* Logs */}
          <div className="rounded-xl border border-border bg-card/30 p-4">
            <p className="text-xs font-mono text-muted-foreground uppercase tracking-widest mb-3">Build Log</p>
            <div ref={logsRef} className="max-h-56 overflow-y-auto space-y-0.5 pr-1">
              <AnimatePresence>
                {logs.map((l) => (
                  <motion.div key={l.id} initial={{ opacity: 0, x: -5 }} animate={{ opacity: 1, x: 0 }}
                    className={`text-xs font-mono ${typeColors[l.type] || "text-muted-foreground"}`}>
                    {l.msg}
                  </motion.div>
                ))}
              </AnimatePresence>
            </div>
          </div>
        </div>

        {/* Research panels */}
        <div className="lg:col-span-3 space-y-3">
          <p className="text-xs font-mono text-muted-foreground uppercase tracking-widest">Live Research Feed</p>
          <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">
            <ResPanel title="Key Features" items={research?.key_features} icon="🔍" />
            <ResPanel title="Similar Apps" items={similarApps?.google_play_apps?.map((a) => a.name + " — " + a.description)} icon="📱" />
            <ResPanel title="Omega Sources" items={omegaSources.slice(0, 6).map((s) => s.name + ": " + s.desc)} icon="⚛️" />
            <ResPanel title="UX Patterns" items={research?.ux_patterns} icon="🎨" />
          </div>
        </div>
      </div>
    </div>
  );
}
