import { useState, useEffect } from "react";
import { motion } from "framer-motion";
import { base44 } from "@/api/base44Client";
import { ArrowLeft, Play, Code2, Download, Zap, FlaskConical, History, PanelLeftClose, PanelLeft } from "lucide-react";
import FileTree from "./FileTree";
import CodeEditor from "./CodeEditor";
import PreviewPanel from "./PreviewPanel";
import DownloadManager from "./DownloadManager";
import UpgradePanel from "./UpgradePanel";
import ResearchSection from "./ResearchSection";

export default function EditorView({ appId, onBack }) {
  const [app, setApp] = useState(null);
  const [loading, setLoading] = useState(true);
  const [html, setHtml] = useState("");
  const [css, setCss] = useState("");
  const [js, setJs] = useState("");
  const [activeFile, setActiveFile] = useState("index.html");
  const [rightPanel, setRightPanel] = useState("preview"); // preview | code | download | upgrade | research
  const [sidebarOpen, setSidebarOpen] = useState(true);
  const [buildMemory, setBuildMemory] = useState([]);
  const [upgradeResult, setUpgradeResult] = useState(null);

  useEffect(() => {
    Promise.all([
      base44.entities.GeneratedApp.list("-created_date", 200),
      base44.entities.BuildMemory.list("-created_date", 50).catch(() => []),
    ]).then(([allApps, allMem]) => {
      const found = allApps.find((a) => a.id === appId);
      if (found) {
        setApp(found);
        setHtml(found.generated_html || "");
        setCss(found.generated_css || "");
        setJs(found.generated_js || "");
      }
      const mem = allMem.filter((m) => m.build_id === appId);
      setBuildMemory(mem || []);
      setLoading(false);
    }).catch(() => setLoading(false));
  }, [appId]);

  const getFiles = () => {
    const files = [
      { name: "index.html", lang: "html" },
      { name: "styles.css", lang: "css" },
      { name: "app.js", lang: "js" },
      { name: "forge_launcher.py", lang: "py" },
      { name: "LAUNCH.bat", lang: "bat" },
      { name: "forge_install.ps1", lang: "ps1" },
      { name: "package.json", lang: "json" },
      { name: "requirements.txt", lang: "txt" },
      { name: "README.md", lang: "md" },
    ];
    return files;
  };

  const getFileContent = (name) => {
    if (name === "index.html") return html;
    if (name === "styles.css") return css;
    if (name === "app.js") return js;
    return `// ${name} — auto-generated by ForgeAI\n// Download the full package to get this file`;
  };

  const setFileContent = (name, content) => {
    if (name === "index.html") setHtml(content);
    else if (name === "styles.css") setCss(content);
    else if (name === "app.js") setJs(content);
  };

  const saveToDb = async () => {
    if (!app) return;
    await base44.entities.GeneratedApp.update(app.id, {
      generated_html: html,
      generated_css: css,
      generated_js: js,
    });
  };

  const handleUpgrade = async (result) => {
    setHtml(result.html);
    setCss(result.css);
    setJs(result.js);
    setUpgradeResult(result);
    await base44.entities.GeneratedApp.update(app.id, {
      generated_html: result.html,
      generated_css: result.css,
      generated_js: result.js,
    });
    setApp((p) => ({ ...p, generated_html: result.html, generated_css: result.css, generated_js: result.js }));
    setRightPanel("preview");
  };

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="w-8 h-8 border-2 border-primary/30 border-t-primary rounded-full animate-spin" />
      </div>
    );
  }

  const TABS = [
    { id: "preview", icon: Play, label: "Preview" },
    { id: "code", icon: Code2, label: "Code" },
    { id: "download", icon: Download, label: "Download" },
    { id: "upgrade", icon: Zap, label: "Upgrade" },
    { id: "research", icon: FlaskConical, label: "Research" },
  ];

  return (
    <div className="h-screen flex flex-col overflow-hidden">
      {/* Top bar */}
      <div className="h-12 border-b border-border bg-card/80 backdrop-blur-sm flex items-center justify-between px-3 shrink-0">
        <div className="flex items-center gap-3">
          <button onClick={onBack} className="text-xs text-muted-foreground hover:text-foreground transition-colors flex items-center gap-1">
            <ArrowLeft className="w-3.5 h-3.5" /> Home
          </button>
          <div className="h-4 w-px bg-border" />
          <button onClick={() => setSidebarOpen(!sidebarOpen)}
            className="text-muted-foreground hover:text-foreground transition-colors p-1">
            {sidebarOpen ? <PanelLeftClose className="w-4 h-4" /> : <PanelLeft className="w-4 h-4" />}
          </button>
          <div className="flex items-center gap-2">
            <span className="text-primary text-sm">◆</span>
            <span className="text-sm font-semibold truncate max-w-[200px]">{app?.app_name}</span>
            {app?.app_type && (
              <span className="px-2 py-0.5 rounded-full bg-chart-3/10 text-chart-3 text-[10px] font-mono hidden sm:inline">
                {app.app_type}
              </span>
            )}
          </div>
        </div>
        <div className="flex items-center gap-1">
          {TABS.map((t) => {
            const Icon = t.icon;
            return (
              <button key={t.id} onClick={() => setRightPanel(t.id)}
                className={`flex items-center gap-1.5 px-3 py-1.5 rounded-md text-xs font-medium transition-all ${
                  rightPanel === t.id ? "bg-primary/10 text-primary" : "text-muted-foreground hover:text-foreground"
                }`}>
                <Icon className="w-3.5 h-3.5" />
                <span className="hidden sm:inline">{t.label}</span>
              </button>
            );
          })}
        </div>
      </div>

      {/* Main content */}
      <div className="flex-1 flex overflow-hidden">
        {/* Sidebar — File tree */}
        {sidebarOpen && (
          <motion.div
            initial={{ width: 0, opacity: 0 }}
            animate={{ width: 220, opacity: 1 }}
            exit={{ width: 0, opacity: 0 }}
            className="w-[220px] border-r border-border bg-card/50 overflow-y-auto shrink-0"
          >
            <div className="p-3 border-b border-border">
              <p className="text-[10px] font-mono text-muted-foreground uppercase tracking-widest">Explorer</p>
            </div>
            <FileTree
              files={getFiles()}
              activeFile={activeFile}
              onSelect={(name) => {
                setActiveFile(name);
                if (rightPanel !== "code") setRightPanel("code");
              }}
            />
            {buildMemory.length > 0 && (
              <div className="border-t border-border p-3">
                <p className="text-[10px] font-mono text-muted-foreground uppercase tracking-widest mb-2">
                  History ({buildMemory.length})
                </p>
                {buildMemory.slice(0, 3).map((m) => (
                  <div key={m.id} className="text-[10px] text-muted-foreground/60 mb-1.5">
                    v{m.version || 1} · {new Date(m.created_date).toLocaleDateString()}
                  </div>
                ))}
              </div>
            )}
          </motion.div>
        )}

        {/* Main panels */}
        <div className="flex-1 flex overflow-hidden">
          {rightPanel === "preview" && (
            <div className="flex-1">
              <PreviewPanel html={html} css={css} js={js} appName={app?.app_name} />
              {upgradeResult && (
                <motion.div initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }}
                  className="absolute bottom-4 left-1/2 -translate-x-1/2 rounded-xl border border-primary/20 bg-card/95 backdrop-blur-sm p-4 shadow-xl max-w-lg z-10">
                  <p className="text-sm font-semibold text-primary mb-1">Upgrade Applied!</p>
                  <p className="text-xs text-muted-foreground mb-2">{upgradeResult.summary}</p>
                  <div className="flex flex-wrap gap-1.5">
                    {upgradeResult.new_features?.slice(0, 5).map((f, i) => (
                      <span key={i} className="px-2 py-0.5 bg-primary/10 text-primary rounded text-[10px]">{f}</span>
                    ))}
                  </div>
                  <button onClick={() => setUpgradeResult(null)} className="absolute top-2 right-2 text-muted-foreground hover:text-foreground text-xs">✕</button>
                </motion.div>
              )}
            </div>
          )}

          {rightPanel === "code" && (
            <div className="flex-1">
              <CodeEditor
                fileName={activeFile}
                content={getFileContent(activeFile)}
                onChange={(content) => {
                  setFileContent(activeFile, content);
                  saveToDb();
                }}
              />
            </div>
          )}

          {rightPanel === "download" && (
            <div className="flex-1 overflow-y-auto p-5">
              <DownloadManager app={app} html={html} css={css} js={js} />
            </div>
          )}

          {rightPanel === "upgrade" && (
            <div className="flex-1 overflow-y-auto p-5">
              <UpgradePanel app={app} html={html} css={css} js={js} onUpgradeComplete={handleUpgrade} />
            </div>
          )}

          {rightPanel === "research" && (
            <div className="flex-1 overflow-y-auto p-5">
              <ResearchSection app={app} />
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
