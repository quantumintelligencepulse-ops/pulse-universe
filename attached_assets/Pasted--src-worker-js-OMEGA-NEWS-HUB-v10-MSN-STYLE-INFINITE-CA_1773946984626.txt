function flattenRoutes() {
  const routes = new Map();
  routes.set("/", { path: "/", title: SITE_NAME, desc: SITE_DESCRIPTION, label: "Home" });
  routes.set("/explore", {
    path: "/explore",
    title: `Explore - ${SITE_NAME}`,
    desc: "Full index of domains and categories.",
    label: "Explore",
  });
  for (const d of SPINE) {
    const dPath = `/${d.key}`;
    routes.set(dPath, {
      path: dPath,
      title: `${d.label} - ${SITE_NAME}`,
      desc: `Top + recent stories for ${d.label}.`,
      label: d.label,
    });
    for (const c of d.children || []) {
      const cPath = `/${d.key}/${c.key}`;
      routes.set(cPath, {
        path: cPath,
        title: `${d.label} - ${c.label} - ${SITE_NAME}`,
        desc: `Top + recent stories for ${d.label} -> ${c.label}.`,
        label: `${d.label} -> ${c.label}`,
      });
    }
  }
  return routes;
}

    .replace(/&quot;/g,'"').replace(/&amp;/g,"&");
}
function hash32(str) {
  let h = 2166136261;
  const s = String(str ?? "");
  for (let i = 0; i < s.length; i++) {
    h ^= s.charCodeAt(i);
    h = Math.imul(h, 16777619);
  }
  return (h >>> 0).toString(16);
}
function canonicalizeUrl(maybeUrl) {
  try {
    const u = new URL(maybeUrl);
    const kill = ["utm_source","utm_medium","utm_campaign","utm_term","utm_content","fbclid","gclid","ref","ref_src","cmpid","output"];
    for (const k of kill) u.searchParams.delete(k);
    u.hash = "";
    if (u.pathname.length > 1) u.pathname = u.pathname.replace(/\/+$/, "");
    if (u.pathname.endsWith("/amp")) u.pathname = u.pathname.replace(/\/amp$/i, "");
    if (u.pathname.endsWith(".amp")) u.pathname = u.pathname.replace(/\.amp$/i, "");
    return u.toString();
  } catch {
    return String(maybeUrl ?? "");
  }
}

function isGoogleHostUrl(u) {
  try {

    return false;
  }
}
function uniqueBy(items, keyFn) {
  const seen = new Set();
  const out = [];
  for (const it of items || []) {
    const k = String(keyFn(it) ?? "").trim().toLowerCase();
    if (!k || seen.has(k)) continue;
    seen.add(k);
    out.push(it);
  }
  return out;
}
async function fetchWithTimeout(url, init, ms) {
  const controller = new AbortController();
  const t = setTimeout(() => controller.abort(), ms);
  try {
    return await fetch(url, { ...init, signal: controller.signal, redirect: "follow" });
  } finally {
    clearTimeout(t);
  }
}

function mulberry32(seed) {
  let t = seed >>> 0;
  return function () {
    t += 0x6D2B79F5;
    let r = Math.imul(t ^ (t >>> 15), 1 | t);
    r ^= r + Math.imul(r ^ (t >>> 7), 61 | r);
    return ((r ^ (r >>> 14)) >>> 0) / 4294967296;
  };
}
async function mapWithConcurrency(items, limit, fn) {
  const out = new Array(items.length);
  let i = 0;
  async function worker() {
    while (true) {
      const idx = i++;
      if (idx >= items.length) return;
      try {
        out[idx] = await fn(items[idx], idx);
      } catch {
        out[idx] = null;
      }
    }
  }
  const n = Math.max(1, Math.min(limit, items.length || 1));
  await Promise.all(Array.from({ length: n }, () => worker()));
  return out;
}
function stableShuffle(arr, seedKey) {
  const a = arr.slice();
  const rng = mulberry32(parseInt(hash32(seedKey), 16) >>> 0);
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(rng() * (i + 1));
    const tmp = a[i];
    a[i] = a[j];
    a[j] = tmp;
  }
  return a;
}

function escapeRegex(str) { return String(str).replace(/[.*+?^${}()|[\]\\]/g, "\\$&"); }
function normalizeBrokenMarkupText(input) {
  let t = String(input || "");
  for (let i = 0; i < 4; i++) t = decodeEntities(t);
  return t
    .replace(/\\n|\\t|\\r/g," ")
    .replace(/<\/?(p|br|div|span|strong|em|b|i|li|ul|ol|h[1-6])[^>]*>/gi," ")
    .replace(/<[^>]+>/g," ")
    .replace(/&[a-z0-9#]+;/gi," ")
    .replace(/[<>]/g," ")
    .replace(/\u00a0/g," ")
    .replace(/\s+/g," ")
    .trim();
}
function cleanDescription(str) { return normalizeBrokenMarkupText(stripCDATA(stripHTML(str || ""))); }
function deepCleanText(text) {
  return normalizeBrokenMarkupText(text).replace(/[ ]{2,}/g," ").replace(/\n{3,}/g,"\n\n").trim();
}
function cleanFinalText(text) {
  let t = deepCleanText(text);
  const banned = ["use the original link","open the original","read the full story","for full context","go to the publisher","click through"];
  for (const phrase of banned) t = t.replace(new RegExp(escapeRegex(phrase),"gi"), "");
  return t.replace(/\s+/g," ").trim();
}
function fixGrammarLite(text) {
  let t = String(text || "").trim();
  t = t.replace(/\s+/g," ").replace(/\s+([,.;:?!])/g,"$1").replace(/([,;:])([A-Za-z])/g,"$1 $2")
    .replace(/\b(i)\b/g,"I").replace(/\b(dont)\b/gi,"don't").replace(/\b(cant)\b/gi,"can't").replace(/\b(wont)\b/gi,"won't")
    .replace(/\b(doesnt)\b/gi,"doesn't").replace(/\b(im)\b/gi,"I'm").replace(/\b(ive)\b/gi,"I've").replace(/\b(youre)\b/gi,"you're")
    .replace(/\bthe the\b/gi,"the").replace(/\bto to\b/gi,"to").trim();
  return t.replace(/(^|[.?!]\s+)([a-z])/g, (m,p1,p2)=>p1+p2.toUpperCase());
}
function hardCap(text, maxChars) {
  const s = String(text || "");
  if (s.length <= maxChars) return s;
  return s.slice(0,maxChars).replace(/\s+\S*$/,"").trim();
}

function edgeKey(url) { return new Request(url, { method: "GET" }); }
async function edgeGetJson(cache, keyUrl) {
  const res = await cache.match(edgeKey(keyUrl));
  if (!res) return null;
  try { return await res.json(); } catch { return null; }
}
async function edgePutJson(cache, keyUrl, obj, ttlSeconds) {
  const res = new Response(JSON.stringify(obj), {
    headers: {
      "content-type": "application/json; charset=utf-8",
      "cache-control": `public, max-age=0, s-maxage=${ttlSeconds}`,
    },
  });
  await cache.put(edgeKey(keyUrl), res);
}
async function edgeHas(cache, keyUrl) { return !!(await cache.match(edgeKey(keyUrl))); }
async function edgePutFlag(cache, keyUrl, ttlSeconds) {
  await cache.put(
    edgeKey(keyUrl),
    new Response("1", {
      headers: { "cache-control": `public, max-age=0, s-maxage=${ttlSeconds}` },
    })
  );
}

function seenResponseUrl(origin, routePath, h) {

function seenRouteRecentUrl(origin, routePath, h) {
  return `${origin}/__edge/seen/route-recent/${VERSION}/${hash32(routePath)}/${h}`;

async function isSeenEdge(cache, origin, routePath, h, allowRecirculation = false) {
  if (await edgeHas(cache, seenResponseUrl(origin, routePath, h))) return true;
  if (!allowRecirculation && await edgeHas(cache, seenRouteRecentUrl(origin, routePath, h))) return true;
  return false;
}
async function markSeenEdge(cache, origin, routePath, h) {
  await edgePutFlag(cache, seenResponseUrl(origin, routePath, h), RESPONSE_DEDUPE_TTL_SECONDS);
  await edgePutFlag(cache, seenRouteRecentUrl(origin, routePath, h), ROUTE_RECENT_TTL_SECONDS);
}

}

async function readArchiveIndex(cache, origin, routePath) {

}

    { v: 1, routePath, buckets: buckets.slice(0, MAX_ARCHIVE_BUCKETS_PER_ROUTE), updatedAt: nowIso() },
    ARCHIVE_INDEX_TTL_SECONDS
  );
}
async function appendArchiveBucket(cache, origin, routePath, items) {
  const bucketId = currentArchiveBucketId();
  const bucketUrl = archiveBucketUrl(origin, routePath, bucketId);
  const existing = (await edgeGetJson(cache, bucketUrl)) || { v: 1, routePath, bucketId, items: [], updatedAt: nowIso() };

  const normalized = uniqueBy(
    [...(existing.items || []), ...(items || [])].filter(Boolean),
    (x) => canonicalizeUrl(x.link) || x.id || x.title
  ).slice(0, MAX_ARCHIVE_ITEMS_PER_BUCKET);

  await edgePutJson(
    cache,
    bucketUrl,
    { v: 1, routePath, bucketId, items: normalized, updatedAt: nowIso() },
    ARCHIVE_BUCKET_TTL_SECONDS
  );

  const index = await readArchiveIndex(cache, origin, routePath);
  const merged = uniqueBy([bucketId, ...index], (x) => String(x)).slice(0, MAX_ARCHIVE_BUCKETS_PER_ROUTE);
  await writeArchiveIndex(cache, origin, routePath, merged);
}
async function readArchiveBuckets(cache, origin, routePath, cursorPage) {
  const index = await readArchiveIndex(cache, origin, routePath);
  if (!index.length) return [];

  const bucketCountBase = 8 + Math.floor(cursorPage / 4) * 2;
  const bucketCount = clamp(bucketCountBase, 8, Math.min(24, index.length));
  const selected = index.slice(0, bucketCount);

  const loaded = await mapWithConcurrency(selected, 8, async (bucketId) =>
    edgeGetJson(cache, archiveBucketUrl(origin, routePath, bucketId))
  );

  const items = [];
  for (const row of loaded || []) {
    if (row?.items?.length) items.push(...row.items);
  }
  return uniqueBy(items, (x) => canonicalizeUrl(x.link) || x.id || x.title).slice(0, ARCHIVE_BLEND_LIMIT);
}

function hiveUrl(origin, key) { return `${origin}/__edge/hive/${VERSION}/${hash32(key)}`; }
function defaultHiveMemory() {
  return {
    v: 3,
    n: 0,
    overused_phrases: {},
    penalties: {
      repetition: 1.0,
      shortness: 1.0,
      offroute: 1.0,
      low_entity_diversity: 1.0,
      low_fact_density: 1.0,
      opener_repetition: 1.0,
    },
    targets: { min_paras: STORY_MIN_PARAGRAPHS, min_chars: STORY_MIN_CHARS, max_paras: STORY_MAX_PARAGRAPHS },
    style: { transition_rate: 0.18, transition_pool: [], avoid_openers: {}, lead_openers: {} },
    updated_at: nowIso(),
  };
}
async function readHive(cache, origin, routePath) {
  const global = await edgeGetJson(cache, hiveUrl(origin, "global"));
  const route = await edgeGetJson(cache, hiveUrl(origin, `route:${routePath}`));
  return { global: global || defaultHiveMemory(), route: route || defaultHiveMemory() };
}
async function writeHive(cache, origin, routePath, mem) {
  const toWriteGlobal = mem?.global || defaultHiveMemory();
  const toWriteRoute = mem?.route || defaultHiveMemory();
  toWriteGlobal.updated_at = nowIso();
  toWriteRoute.updated_at = nowIso();
  await edgePutJson(cache, hiveUrl(origin, "global"), toWriteGlobal, HIVE_MEM_TTL_SECONDS);
  await edgePutJson(cache, hiveUrl(origin, `route:${routePath}`), toWriteRoute, HIVE_MEM_TTL_SECONDS);
}
function bumpPhrase(map, phrase) {
  const p = String(phrase || "").trim().toLowerCase();
  if (!p) return;
  map[p] = (map[p] || 0) + 1;
}
function topOverusedPhrases(mem, limit = 12) {
  return Object.entries(mem?.overused_phrases || {}).sort((a,b)=>b[1]-a[1]).slice(0,limit).map(([k])=>k);
}
function bumpMap(map, k, amt = 1) {
  const key = String(k || "").trim().toLowerCase();
  if (!key) return;
  map[key] = (map[key] || 0) + amt;
}
function topKeys(map, limit = 12) {
  return Object.entries(map || {}).sort((a,b)=>b[1]-a[1]).slice(0,limit).map(([k])=>k);
}
function ensureRouteStyle(routePath, hiveRouteMem) {
  const mem = hiveRouteMem || defaultHiveMemory();
  mem.style = mem.style || defaultHiveMemory().style;
  mem.style.avoid_openers = mem.style.avoid_openers || {};
  mem.style.lead_openers = mem.style.lead_openers || {};
  if (Array.isArray(mem.style.transition_pool) && mem.style.transition_pool.length >= 10) return mem;

  const { domainKey } = routeInfo(routePath);
  const domain = (domainKey || "general").toLowerCase();
  const seed = parseInt(hash32(`style:${routePath}`), 16) >>> 0;
  const rng = mulberry32(seed);

  const POOLS = {
    "real-estate": ["In the housing market,","For buyers,","For sellers,","In mortgage pricing,","In rate markets,","In inventory data,","In rental trends,","In homebuilding,","Across metro areas,"],
    "financials": ["In markets,","In rates,","In inflation data,","In earnings,","In dealmaking,","In credit,","In volatility,"],
    "sports": ["In the matchup,","On the schedule,","In the standings,","In the locker room,","In coaching decisions,"],
    "health-care": ["In clinical guidance,","For patients,","In public health,","In new research,"],
    "science-and-mathematics": ["In the research,","In the data,","In peer review,","In new models,"],
    "government-law-and-civilization": ["In policy,","In legal terms,","In official statements,","In negotiations,"],
    "information-technology": ["In product updates,","In security terms,","In developer circles,","In the rollout,"],
    "energy": ["In energy markets,","In power grids,","In emissions policy,","In fuel demand,"],
    "utilities": ["In grid planning,","In reliability concerns,","In generation mix,"],
    "materials": ["In commodity pricing,","In supply chains,","In mining output,"],
    "industrials": ["In logistics,","In shipping lanes,","In factory output,"],
  };
  const generic = ["Still,","That said,","In contrast,","By comparison,","In response,","As a result,","In the short term,","Over the longer term,"];
  const basePool = (POOLS[domain] || []).slice();
  const pool = uniqueBy([...basePool, ...generic.filter(()=>rng()<0.85), ...generic.filter(()=>rng()<0.35)], (x)=>String(x).toLowerCase()).slice(0, 22);

  mem.style.transition_pool = pool;
  mem.style.transition_rate = clamp01(0.10 + rng() * 0.14);
  return mem;
}
function addTransitions(paras, routePath, hiveRouteMem) {
  const out = [];
  const mem = ensureRouteStyle(routePath, hiveRouteMem || defaultHiveMemory());
  const style = mem.style || defaultHiveMemory().style;
  const seed = parseInt(hash32(`trans:${routePath}:${mem.n || 0}`), 16) >>> 0;
  const rng = mulberry32(seed);
  const pool = Array.isArray(style.transition_pool) ? style.transition_pool : [];
  const rate = clamp01(style.transition_rate);

  const overused = new Set(
    [...topKeys(style.avoid_openers || {}, 14), ...topOverusedPhrases(mem, 10)].map((x)=>String(x).toLowerCase())
  );

  let prefixesUsed = 0;
  const MAX_PREFIXES_PER_STORY = 1;

  function alreadyHasOpener(p) {
    return /^(Still,|That said,|In contrast,|By comparison,|In response,|As a result,|In the short term,|Over the longer term,|In the housing market,|For buyers,|For sellers,|In mortgage pricing,|In markets,|In rates,|In the matchup,|On the schedule,)/i.test(p);
  }
  function maybePrefix(i, p) {
    if (i === 0) return p;
    if (prefixesUsed >= MAX_PREFIXES_PER_STORY) return p;
    if (rng() > rate) return p;
    if (alreadyHasOpener(p)) return p;

    let picked = "";
    for (let k = 0; k < 8; k++) {
      const cand = pool[Math.floor(rng() * pool.length)] || "";
      if (!cand) break;
      const low = String(cand).toLowerCase();
      if (!overused.has(low)) { picked = cand; break; }
      picked = cand;
    }
    if (!picked) return p;
    prefixesUsed++;
    const next = p.charAt(0).toLowerCase() + p.slice(1);
    return `${picked} ${next}`;
  }

  for (let i = 0; i < (paras || []).length; i++) {
    const p = String(paras[i] || "").trim();
    if (!p) continue;
    out.push(maybePrefix(i, p));
  }
  return out;
}

function groupForPath(path) {
  const parts = String(path || "/").split("/").filter(Boolean);
  const domain = (parts[0] || "").toLowerCase();
  const cat = (parts[1] || "").toLowerCase();

  if (!domain) return "latest";
  if (domain.includes("education-and-knowledge")) return "education";
  if (domain.includes("financial")) return cat.includes("crypto") ? "crypto" : "business";
  if (domain.includes("real-estate")) return "realestate";
  if (domain.includes("information-technology") || domain.includes("communication-services")) return "technology";
  if (domain.includes("health")) return "health";
  if (domain.includes("science")) return "science";
  if (domain.includes("sports")) return "sports";
  if (domain.includes("energy") || domain.includes("utilities") || domain.includes("materials") || domain.includes("industrials")) return "business";
  return "latest";
}
function buildBaseUniverseForPath(path) {
  const parts = String(path || "/").split("/").filter(Boolean);
  const domainKey = (parts[0] || "").trim();
  const catKey = (parts[1] || "").trim();
  const group = groupForPath(path);

  const base = FEEDS_BASE[group] || FEEDS_BASE.latest;
  const addDomain = domainKey && DOMAIN_FEED_ADDONS[domainKey] ? DOMAIN_FEED_ADDONS[domainKey] : [];
  const addCat = catKey && CATEGORY_FEED_ADDONS[catKey] ? CATEGORY_FEED_ADDONS[catKey] : [];

  const cross = [];
  if (domainKey === "financials") cross.push(...(FEEDS_BASE.business || []));
  if (catKey === "markets") cross.push(...(FEEDS_BASE.business || []));
  if (domainKey === "real-estate") cross.push(...(FEEDS_BASE.business || []));
  if (domainKey === "government-law-and-civilization") cross.push(...(FEEDS_BASE.latest || []));
  if (domainKey === "science-and-mathematics") cross.push(...(FEEDS_BASE.science || []));
  if (domainKey === "education-and-knowledge") cross.push(...(FEEDS_BASE.education || []));
  if (domainKey === "sports") cross.push(...(FEEDS_BASE.sports || []));
  if (domainKey === "information-technology") cross.push(...(FEEDS_BASE.technology || []));
  if (domainKey === "health-care") cross.push(...(FEEDS_BASE.health || []));
  if (domainKey === "communication-services") cross.push(...(FEEDS_BASE.technology || []));
  if (domainKey === "energy") cross.push(...(FEEDS_BASE.business || []));
  if (domainKey === "industrials") cross.push(...(FEEDS_BASE.business || []));
  if (catKey === "ai") cross.push(...(FEEDS_BASE.technology || []));

}
function extraFeedsForRoute(routePath, baseUniverse) {
  const baseSet = new Set((baseUniverse || []).map((u)=>String(u).trim()));
  const candidates = EXTRA_FEED_POOL.filter((u)=>u && !baseSet.has(u));
  if (!candidates.length) return [];

  const epochBucket = Math.floor(Date.now() / (1000 * 60 * 60 * 6));
  const shuffled = stableShuffle(candidates, `extras:${routePath}:${epochBucket}`);
  return shuffled.slice(0, Math.min(EXTRA_SOURCES_PER_ROUTE, shuffled.length));
}
function buildFeedUniverseForPath(path) {
  const baseUniverse = buildBaseUniverseForPath(path);
  const extras = extraFeedsForRoute(path, baseUniverse);
  return uniqueBy([...extras, ...baseUniverse], (x)=>x);
}
function selectFeedSlice(universe, routePath, cursorPage, roundIndex, passIndex, sliceSize) {
  const feeds = universe || [];
  if (!feeds.length) return [];

  const epochBucket = Math.floor(Date.now() / (1000 * 60 * 10));
  const shuffled = stableShuffle(feeds, `feedsel:${routePath}:${epochBucket}:p${cursorPage}:r${roundIndex}:k${passIndex}`);
  return shuffled.slice(0, Math.min(sliceSize, shuffled.length));
}

function pickFirst(text, patterns) {
  for (const p of patterns) {
    const m = text.match(p);
    if (m && m[1]) return m[1].trim();
  }
  return "";
}
function parseRssItems(xml, feedUrl) {
  const out = [];

  const itemMatches = (xml.match(/<item[\s\S]*?<\/item>/gi) || []).slice(0, MAX_ITEMS_PER_FEED_PARSE);
  for (const block of itemMatches) {
    const title = normalizeBrokenMarkupText(pickFirst(block, [/<title[^>]*>([\s\S]*?)<\/title>/i])) || "";
    const linkRaw = normalizeBrokenMarkupText(pickFirst(block, [/<link[^>]*>([\s\S]*?)<\/link>/i])) || "";
    const link = canonicalizeUrl(linkRaw);
    if (!title || !link || isGoogleHostUrl(link)) continue;

    const guid = normalizeBrokenMarkupText(pickFirst(block, [/<guid[^>]*>([\s\S]*?)<\/guid>/i])) || link || title;
    const pubDate = normalizeBrokenMarkupText(pickFirst(block, [/<pubDate[^>]*>([\s\S]*?)<\/pubDate>/i])) || "";
    const desc = pickFirst(block, [/<description[^>]*>([\s\S]*?)<\/description>/i, /<content:encoded[^>]*>([\s\S]*?)<\/content:encoded>/i]) || "";
    const mediaThumb = pickFirst(block, [/<media:thumbnail[^>]*url="([^"]+)"/i, /<media:content[^>]*url="([^"]+)"/i, /<enclosure[^>]*url="([^"]+)"/i]) || "";

    out.push({
      id: `rss:${hash32(feedUrl + "::" + guid)}`,
      title: safeText(normalizeBrokenMarkupText(title), 180),
      link,
      publishedAt: toIsoDate(pubDate || Date.now()),
      description: safeText(cleanDescription(desc), 700),
      image: mediaThumb ? canonicalizeUrl(mediaThumb) : "",
      feedUrl,
      sourceKind: "rss",
    });
  }

  const entryMatches = (xml.match(/<entry[\s\S]*?<\/entry>/gi) || []).slice(0, MAX_ITEMS_PER_FEED_PARSE);
  for (const block of entryMatches) {
    const title = normalizeBrokenMarkupText(pickFirst(block, [/<title[^>]*>([\s\S]*?)<\/title>/i])) || "";
    const link = pickFirst(block, [
      /<link[^>]*href=["']([^"']+)["'][^>]*rel=["']alternate["'][^>]*\/?>/i,
      /<link[^>]*href=["']([^"']+)["'][^>]*\/?>/i,
    ]) || "";
    const url = canonicalizeUrl(normalizeBrokenMarkupText(link));
    if (!title || !url || isGoogleHostUrl(url)) continue;

    const guid = normalizeBrokenMarkupText(pickFirst(block, [/<id[^>]*>([\s\S]*?)<\/id>/i])) || url || title;
    const updated = normalizeBrokenMarkupText(pickFirst(block, [/<updated[^>]*>([\s\S]*?)<\/updated>/i, /<published[^>]*>([\s\S]*?)<\/published>/i])) || "";
    const summary = pickFirst(block, [/<summary[^>]*>([\s\S]*?)<\/summary>/i, /<content[^>]*>([\s\S]*?)<\/content>/i]) || "";

    out.push({
      id: `atom:${hash32(feedUrl + "::" + guid)}`,
      title: safeText(normalizeBrokenMarkupText(title), 180),
      link: url,
      publishedAt: toIsoDate(updated || Date.now()),
      description: safeText(cleanDescription(summary), 700),
      image: "",
      feedUrl,
      sourceKind: "atom",
    });
  }

  return out;
}
function sortNewest(items) {
  return [...(items || [])].sort((a, b) => {
    const ta = a?.publishedAt ? new Date(a.publishedAt).getTime() : 0;
    const tb = b?.publishedAt ? new Date(b.publishedAt).getTime() : 0;
    return tb - ta;
  });
}
async function fetchTextCached(feedUrl) {
  const cache = caches.default;
  const req = new Request(feedUrl, { method: "GET" });

  const cached = await cache.match(req);
  if (cached) return await cached.text();

  const res = await fetchWithTimeout(feedUrl, {
    method: "GET",
    cf: { cacheTtl: FEED_CACHE_TTL_SECONDS, cacheEverything: true },
    headers: {
      "user-agent": "OMEGA-HIVE/1.0 (+https://myaigpt.online/)",
      accept: "application/rss+xml, application/xml;q=0.9, text/xml;q=0.8, */*;q=0.1",
    },
  }, FEED_FETCH_TIMEOUT_MS);

  if (!res.ok) throw new Error(`Feed fetch failed ${res.status}`);

  const clone = new Response(res.body, res);
  clone.headers.set("cache-control", `public, s-maxage=${FEED_CACHE_TTL_SECONDS}`);
  await cache.put(req, clone);

  return await res.text();
}
async function loadFeedItems(feedUrl) {
  try {
    if (!feedUrl) return [];
    const xml = await fetchTextCached(feedUrl);
    return parseRssItems(xml, feedUrl);
  } catch {
    return [];
  }
}

function parseDdgHtmlResults(html) {
  const out = [];
  const re = /<a[^>]+class="result__a"[^>]+href="([^"]+)"[^>]*>([\s\S]*?)<\/a>/gi;
  let m;
  while ((m = re.exec(html)) !== null) {
    const href = canonicalizeUrl(normalizeBrokenMarkupText(m[1] || ""));
    const title = safeText(normalizeBrokenMarkupText(m[2] || ""), 180);
    if (!href || !title) continue;
    if (isGoogleHostUrl(href)) continue;
    out.push({ href, title });
    if (out.length >= DDG_MAX_RESULTS) break;
  }
  return uniqueBy(out, (x) => x.href);
}
async function ddgSearchOnce(query) {
  if (!DDG_ENABLED) return [];
  const q = String(query || "").trim();
  if (!q || q.length < 4) return [];
  try {
    const body = new URLSearchParams({ q, kl: "us-en", df: DDG_TIME_FILTER });
    const res = await fetchWithTimeout("https://html.duckduckgo.com/html", {
      method: "POST",
      headers: {
        "content-type": "application/x-www-form-urlencoded",
        "user-agent": "OMEGA-HIVE/1.0 (+https://myaigpt.online/)",
        accept: "text/html,*/*;q=0.8",
      },
      body,
    }, DDG_TIMEOUT_MS);
    if (!res.ok) return [];
    return parseDdgHtmlResults(await res.text());
  } catch {
    return [];
  }
}
function ddgQueriesForPath(routePath) {
  const { domainLabel, catLabel, domainKey } = routeInfo(routePath);
  const d = domainLabel || "news";
  const c = catLabel || "";
  const domain = String(domainKey || "").toLowerCase();

  const base = [
    c ? `${d} ${c}` : d,
    c ? `${c} latest news` : `${d} latest news`,
    c ? `${c} analysis` : `${d} analysis`,
    c ? `${c} updates` : `${d} updates`,
  ];

  const boosts = [];
  if (domain === "real-estate") boosts.push(
    `${c ? c+" " : ""}housing market mortgage rates`,
    `${c ? c+" " : ""}home prices inventory housing supply`,
    "rent inflation housing supply",
  );
  if (domain === "financials") boosts.push(
    "stock market futures bonds yields",
    "earnings outlook markets rates",
    "inflation jobs report central bank",
  );
  if (domain === "information-technology") boosts.push(
    "software security patch vulnerability",
    "AI model release update",
  );
  if (domain === "sports") boosts.push(
    "scores injuries roster trade",
    "playoff standings schedule",
  );

  return uniqueBy([...boosts, ...base].map((x)=>safeText(x, 120)), (x)=>x).filter(Boolean).slice(0, DDG_QUERY_COUNT);
}
async function ddgExpandForPath(routePath) {
  const qs = ddgQueriesForPath(routePath);
  const all = [];
  for (const q of qs) {
    const hits = await ddgSearchOnce(q);
    for (const h of hits) all.push(h);
  }
  return uniqueBy(all, (x)=>x.href);
}
function ddgHitsToItems(hits, routePath) {
  return (hits || []).map((x) => ({
    id: `ddg:${hash32(x.href + "::" + x.title + "::" + routePath)}`,
    title: safeText(normalizeBrokenMarkupText(x.title), 180),
    link: x.href,
    publishedAt: "",
    description: "",
    image: "",
    feedUrl: "duckduckgo-html",
    sourceKind: "ddg",
  }));
}

function routeTokens(routePath) {
  const { domainLabel, catLabel, domainKey, catKey } = routeInfo(routePath);
  const raw = `${domainLabel} ${catLabel} ${domainKey} ${catKey}`.toLowerCase().replace(/[^a-z0-9\s-]/g," ").replace(/\s+/g," ").trim();
  return uniqueBy(raw.split(" ").filter((w)=>w.length>=3), (x)=>x).slice(0, 40);
}
function routeSignalTerms(routePath) {
  const { domainKey, catKey, domainLabel, catLabel } = routeInfo(routePath);
  const parts = [];
  if (domainKey && DOMAIN_SIGNAL_MAP[domainKey]) parts.push(...DOMAIN_SIGNAL_MAP[domainKey]);
  if (catKey && CATEGORY_SIGNAL_MAP[catKey]) parts.push(...CATEGORY_SIGNAL_MAP[catKey]);
  if (domainLabel) parts.push(domainLabel.toLowerCase());
  if (catLabel) parts.push(catLabel.toLowerCase());
  parts.push(...routeTokens(routePath));
  return uniqueBy(parts.map((x)=>String(x||"").trim().toLowerCase()).filter(Boolean), (x)=>x);
}
function itemHaystack(item) {
  return `${item?.title || ""} ${item?.description || ""} ${item?.link || ""} ${item?.feedUrl || ""}`.toLowerCase();
}
function countSignalHits(item, terms) {
  const hay = itemHaystack(item);
  let hits = 0;
  for (const t of (terms || [])) if (t && hay.includes(t)) hits++;
  return hits;
}
function isLooseGenericRoute(routePath) {
  const { domainKey } = routeInfo(routePath);
  return !domainKey || routePath === "/" || routePath === "/explore";
}
function requiredSignalHitsForRoute(routePath) {
  if (routePath === "/" || routePath === "/explore") return 0;
  return 1;
}
function passesHardRouteLock(item, routePath) {
  if (isLooseGenericRoute(routePath)) return true;
  const terms = routeSignalTerms(routePath);
  const hits = countSignalHits(item, terms);
  return hits >= requiredSignalHitsForRoute(routePath);
}
function filterItemsForRouteLock(items, routePath) {
  if (routePath === "/" || routePath === "/explore") return items || [];
  return (items || []).filter((it)=>passesHardRouteLock(it, routePath));
}

function titleQuality(title) {
  const t = String(title || "").trim();
  if (!t) return -30;
  if (t.length < MIN_TITLE_LEN) return -20;
  if (t.length > MAX_TITLE_LEN) return -8;
  if (/^untitled\b/i.test(t)) return -30;
  if (/^here'?s the latest\.?$/i.test(t)) return -24;
  if (/^(latest|story|news update)$/i.test(t)) return -24;
  if (/^[^a-zA-Z0-9]+$/.test(t)) return -20;
  return 8;
}
function descQuality(desc) {
  const d = String(desc || "").trim();
  if (!d) return -8;
  if (d.length < MIN_DESC_LEN) return -3;
  if (/^no summary available\.?$/i.test(d)) return -20;
  if (/^loading/i.test(d)) return -20;
  return 6;
}
function sourceTier(host) {
  const h = String(host || "").toLowerCase();
  if (!h) return 0;
  if (/(reuters|apnews|bbc|npr|ft\.com|theguardian|cnbc|marketwatch|fortune|aljazeera|dw\.com|france24|cbc\.ca|economist|nytimes\.com|cnn\.com|nasdaq\.com)\b/.test(h)) return 10;
  if (/(axios|politico|wired|techcrunch|theverge|arstechnica|krebsonsecurity|securityweek|bleepingcomputer|who\.int|cdc\.gov|nih\.gov|statnews|scientificamerican|nature\.com|science\.org|sec\.gov|federalreserve\.gov|bankofengland\.co\.uk|ecb\.europa\.eu|bis\.org|imf\.org|worldbank\.org|oecd\.org|news\.un\.org|lawfaremedia|scotusblog|courthousenews)\b/.test(h)) return 8;
  return 5;
}
function matchHits(item, tokens) {
  const hay = itemHaystack(item);
  let hits = 0;
  for (const t of tokens) if (t && hay.includes(t)) hits++;
  return hits;
}
function safeAgeHours(publishedAt) {
  try {
    if (!publishedAt) return 999;
    const t = new Date(publishedAt).getTime();
    if (!Number.isFinite(t) || Number.isNaN(t)) return 999;
    return Math.max(0, (Date.now() - t) / 3600000);
  } catch {
    return 999;
  }
}
function buildFallbackDesc(item) {
  const t = normalizeBrokenMarkupText(item?.title || "");
  if (!t) return "";
  const host = hostnameOf(item?.link || "");
  const hint = host ? `Source: ${host}.` : "Source coverage.";
  return safeText(`${t}. ${hint} On-site writing for this route.`, 240);
}
function passesCardQualityStrict(item) {
  if (!item?.link) return false;
  if (isGoogleHostUrl(item.link)) return false;
  const title = String(item?.title || "").trim();
  const desc = String(item?.description || "").trim();
  const q = titleQuality(title) + descQuality(desc) + sourceTier(hostnameOf(item.link));
  if (!title || title.length < MIN_TITLE_LEN) return false;
  if (!desc || desc.length < MIN_DESC_LEN) return false;
  if (q < MIN_CARD_QUALITY_SCORE) return false;
  return true;
}
function passesCardQualityRelaxed(item) {
  if (!item?.link) return false;
  if (isGoogleHostUrl(item.link)) return false;
  const title = String(item?.title || "").trim();
  if (!title || title.length < MIN_TITLE_LEN) return false;
  return true;
}
function scoreItemForRoute(item, routePath) {
  const host = hostnameOf(item?.link || "");
  const sourceScore = sourceTier(host);
  const hits = matchHits(item, routeTokens(routePath));
  const signalHits = countSignalHits(item, routeSignalTerms(routePath));
  const ageHours = safeAgeHours(item?.publishedAt || "");
  const freshness = Math.max(0, 96 - ageHours);
  const titleScore = titleQuality(item?.title || "");
  const descriptionScore = descQuality(item?.description || "");
  const hardPenalty = passesHardRouteLock(item, routePath) ? 0 : 1000;
  return sourceScore * 4 + hits * 10 + signalHits * 22 + freshness * 0.6 + titleScore + descriptionScore - hardPenalty;
}

function applyRecency(items) {
  const strictCutoff = daysAgoIso(RECENCY_DAYS_STRICT);
  let strict = (items || []).filter((it)=>it?.publishedAt && isNewerThan(it.publishedAt, strictCutoff));
  if (strict.length < 20) {
    const fallbackCutoff = daysAgoIso(RECENCY_DAYS_FALLBACK);
    strict = (items || []).filter((it)=>it?.publishedAt && isNewerThan(it.publishedAt, fallbackCutoff));
  }
  return strict.length ? strict : (items || []);
}
function normalizeCandidateItem(it) {
  if (!it?.link) return null;
  const link = canonicalizeUrl(it.link);
  if (!link || isGoogleHostUrl(link)) return null;
  const title = safeText(normalizeBrokenMarkupText(it.title || ""), 180);
  if (!title) return null;
  const description = String(it.description || "").trim();
  return {
    ...it,
    id: it.id || hash32(link + "::" + title),
    link,
    title,
    description: description || buildFallbackDesc({ ...it, title, link }),
  };
}
async function collectLiveCandidateBag(routePath, cursorPage, roundIndex, timeBudgetMs) {
  const startMs = Date.now();
  const universe = buildFeedUniverseForPath(routePath);

  let all = [];
  let pass = 0;

  while (pass < MAX_SLICE_PASSES && Date.now() - startMs < timeBudgetMs) {
    const sliceSize = SLICE_SIZE_BASE + pass * SLICE_SIZE_BOOST;
    const feedSlice = selectFeedSlice(universe, routePath, cursorPage, roundIndex, pass, sliceSize);

    const loaded = await mapWithConcurrency(feedSlice, MAX_FEED_CONCURRENCY, async (u) => loadFeedItems(u));
    const items = uniqueBy((loaded || []).flat().filter(Boolean), (x)=>canonicalizeUrl(x.link) || x.id || x.title)
      .filter((x)=>x?.link && !isGoogleHostUrl(x.link));

    all = uniqueBy([...all, ...items], (x)=>canonicalizeUrl(x.link) || x.id || x.title);
    if (all.length >= 1200) break;
    pass++;
  }

  all = sortNewest(all);
  all = applyRecency(all);

  let strict = filterItemsForRouteLock(all.filter(passesCardQualityStrict), routePath);

  if (strict.length < 24) {
    let relaxed = filterItemsForRouteLock(
      all.filter(passesCardQualityRelaxed).map((it)=> {
        const desc = String(it.description || "").trim();
        if (!desc || desc.length < MIN_DESC_LEN) return { ...it, description: buildFallbackDesc(it) };
        return it;
      }),
      routePath
    );

    if (DDG_ENABLED && relaxed.length < DDG_MIN_FILL_THRESHOLD && Date.now() - startMs < timeBudgetMs) {
      const ddgItems = filterItemsForRouteLock(
        ddgHitsToItems(await ddgExpandForPath(routePath), routePath).map((it)=>({ ...it, description: buildFallbackDesc(it) })),
        routePath
      );
      relaxed = uniqueBy([...relaxed, ...ddgItems], (x)=>canonicalizeUrl(x.link) || x.id || x.title);
    }

    const finalRelaxed = relaxed
      .map(normalizeCandidateItem)
      .filter(Boolean)
      .slice(0, MAX_CANDIDATES_PER_BAG)
      .sort((a,b)=>scoreItemForRoute(b, routePath) - scoreItemForRoute(a, routePath));

    return finalRelaxed;
  }

  const finalStrict = strict
    .map(normalizeCandidateItem)
    .filter(Boolean)
    .slice(0, MAX_CANDIDATES_PER_BAG)
    .sort((a,b)=>scoreItemForRoute(b, routePath) - scoreItemForRoute(a, routePath));

  return finalStrict;
}
function shouldAllowRecirculation(cursor) {
  const c = Math.max(0, parseInt(cursor || 0, 10) || 0);
  return c >= RECIRCULATION_START_CURSOR;
}
function computeSeekWindow(cursor) {
  const c = Math.max(0, parseInt(cursor || 0, 10) || 0);
  const growth = Math.floor(c / 120) * 900;
  return clamp(SEEK_WINDOW_BASE + growth, SEEK_WINDOW_BASE, SEEK_WINDOW_MAX);
}
function recirculationTargetCount(limit, cursor) {
  if (!shouldAllowRecirculation(cursor)) return 0;
  const depthFactor = Math.floor((cursor - RECIRCULATION_START_CURSOR) / RECIRCULATION_INTERVAL);
  const rate = clamp01(RECIRCULATION_RATE_DEEP + depthFactor * 0.03);
  return clamp(Math.floor(limit * rate), 1, Math.max(1, Math.floor(limit * 0.55)));
}
function mergeBagsForInfiniteFeed(liveBag, archiveBag, routePath, cursorPage, roundIndex) {
  const live = (liveBag || []).map(normalizeCandidateItem).filter(Boolean);
  const archive = (archiveBag || []).map(normalizeCandidateItem).filter(Boolean);

  const freshWeighted = [];
  for (const it of live) {
    freshWeighted.push({ ...it, _layer: "live", _score: scoreItemForRoute(it, routePath) + 28 });
  }
  for (const it of archive) {
    freshWeighted.push({ ...it, _layer: "archive", _score: scoreItemForRoute(it, routePath) + 6 });
  }

  const deduped = uniqueBy(freshWeighted, (x)=>canonicalizeUrl(x.link) || x.id || x.title);
  const sorted = deduped.sort((a,b)=>b._score - a._score);

  const overlapCount = clamp(Math.floor(sorted.length * WINDOW_OVERLAP_RATE), 0, 160);
  if (overlapCount > 0 && cursorPage > 0) {
    const overlapSource = sorted.slice(0, overlapCount);
    const overlapShuffle = stableShuffle(overlapSource, `overlap:${routePath}:${cursorPage}:${roundIndex}`);
    return uniqueBy([...sorted, ...overlapShuffle], (x)=>`${canonicalizeUrl(x.link)}::${x._layer || ""}`);
  }
  return sorted;
}

function cardImg(origin, seed) { return `${origin}/api/img?seed=${encodeURIComponent(seed)}&w=1200&h=800`; }
function buildCleanDescription(item) {
  const desc = normalizeBrokenMarkupText(item?.description || "");
  const title = normalizeBrokenMarkupText(item?.title || "");
  if (desc && !/^no summary available\.?$/i.test(desc)) return safeText(desc, 260);
  if (title.length > 18) return safeText(`${title} - on-site coverage on this route.`, 260);
  return safeText(buildFallbackDesc(item), 260);
}
function toCard(origin, routePath, feedItem) {
  const title = safeText(normalizeBrokenMarkupText(feedItem?.title || "").trim(), 180);
  const desc = buildCleanDescription(feedItem);
  if (!title || !desc) return null;
  if (!passesHardRouteLock(feedItem, routePath)) return null;

  const can = canonicalizeUrl(feedItem.link || "");
  const h = dedupeKeyForUrl(can);
  const img = feedItem.image
    ? `${origin}/api/imgproxy?u=${encodeURIComponent(feedItem.image)}&seed=${encodeURIComponent(`rss:${h}`)}`
    : cardImg(origin, `hero:${hash32(routePath)}:${h}`);

  return {
    id: h,
    title,
    description: desc,
    pubDate: feedItem.publishedAt || "",
    link: can,
    image: img,
    hub: routePath,
    layer: feedItem._layer || "live",
  };
}
function cardUniqueKey(card) { return canonicalizeUrl(card?.link || "") || card?.id || card?.title || ""; }

async function buildNonStopWindow(cache, origin, routePath, cursor, limit) {
  const startMs = Date.now();
  const start = clamp(parseInt(cursor || 0, 10) || 0, 0, 10000000);
  const cursorPage = Math.floor(start / Math.max(1, limit));
  const seekWindow = computeSeekWindow(start);
  const recircTarget = recirculationTargetCount(limit, start);

  const out = [];
  let roundIndex = 0;
  let recircAdded = 0;

  while (out.length < limit && roundIndex < MAX_FETCH_ROUNDS_PER_REQUEST && Date.now() - startMs < LIVE_TIME_BUDGET_MS) {
    const roundBudget = Math.min(LIVE_FETCH_ROUND_BUDGET_MS, Math.max(1200, LIVE_TIME_BUDGET_MS - (Date.now() - startMs)));

    const liveBag = await collectLiveCandidateBag(routePath, cursorPage, roundIndex, roundBudget);
    if (liveBag.length) {
      // archive every live round so the route builds history
      try { await appendArchiveBucket(cache, origin, routePath, liveBag.slice(0, MAX_ARCHIVE_ITEMS_PER_BUCKET)); } catch {}
    }

    const archiveBag = await readArchiveBuckets(cache, origin, routePath, cursorPage);
    const bag = mergeBagsForInfiniteFeed(liveBag, archiveBag, routePath, cursorPage, roundIndex);

    const overlapOffset = Math.floor(Math.max(0, start - limit) * WINDOW_OVERLAP_RATE);
    const scanStart = roundIndex === 0 ? Math.max(0, start - overlapOffset) : 0;
    const scanEnd = Math.min(bag.length, scanStart + seekWindow);

    for (let i = scanStart; i < scanEnd && out.length < limit; i++) {
      const it = bag[i];
      if (!it?.link) continue;

      const can = canonicalizeUrl(it.link);
      const h = dedupeKeyForUrl(can);

      const allowRecirculation = shouldAllowRecirculation(start) && recircAdded < recircTarget && (it._layer === "archive" || i > Math.floor(scanEnd * 0.55));
      const seen = await isSeenEdge(cache, origin, routePath, h, allowRecirculation);
      if (seen) continue;

      const card = toCard(origin, routePath, it);
      if (!card) continue;

      await markSeenEdge(cache, origin, routePath, h);
      if (allowRecirculation) recircAdded++;
      out.push(card);
    }

    roundIndex++;
  }

  return {
    items: uniqueBy(out, (x)=>cardUniqueKey(x)).slice(0, limit),
    nextCursor: start + Math.max(1, limit),
    hasMore: true,
    source: "msn-style-fresh-archive-nonstop",
  };
}

async function handleLive(request, origin, routePath) {
  const url = new URL(request.url);
  const cursor = clamp(parseInt(url.searchParams.get("cursor") || "0", 10) || 0, 0, 10000000);
  const limit = clamp(parseInt(url.searchParams.get("limit") || String(API_PAGE_SIZE), 10) || API_PAGE_SIZE, 6, 80);
  const cache = caches.default;

  const stream = await buildNonStopWindow(cache, origin, routePath, cursor, limit);
  return json({
    ok: true,
    hub: routePath,
    items: stream.items,
    nextCursor: stream.nextCursor,
    hasMore: true,
    source: stream.source,
  }, 200, 5);
}

function extractTag(html, tag) {
  const m = html.match(new RegExp(`<${tag}[^>]*>([\\s\\S]*?)<\\/${tag}>`, "i"));
  return m ? normalizeBrokenMarkupText(m[1]) : "";
}
function extractMeta(html, attrName, attrValue) {
  const regex = new RegExp(`<meta[^>]+${attrName}=["']${escapeRegex(attrValue)}["'][^>]+content=["']([^"']+)["'][^>]*>`, "i");
  const reverseRegex = new RegExp(`<meta[^>]+content=["']([^"']+)["'][^>]+${attrName}=["']${escapeRegex(attrValue)}["'][^>]*>`, "i");
  const m1 = html.match(regex), m2 = html.match(reverseRegex);
  return normalizeBrokenMarkupText(m1 ? m1[1] : m2 ? m2[1] : "");
}
function isBoilerplate(text) {
  const t = String(text || "").toLowerCase();
  return ["all rights reserved","subscribe","newsletter","advertisement","privacy policy","terms of service","cookie policy","recommended stories","sign up","manage cookies"].some((x)=>t.includes(x));
}
function looksLikeLooseCaption(text) {
  const s = String(text || "").trim();
  return /^[A-Z][^.]{0,120}\s[A-Z][^.]{0,120}$/.test(s) && !/[.?!:]/.test(s);
}
function extractArticleContent(html, url = "") {
  const title = extractMeta(html,"property","og:title") || extractMeta(html,"name","twitter:title") || extractTag(html,"title") || "";
  const byline = extractMeta(html,"name","author") || extractMeta(html,"property","article:author") || "";

  let body = html
    .replace(/<script\b[^>]*>[\s\S]*?<\/script>/gi," ")
    .replace(/<style\b[^>]*>[\s\S]*?<\/style>/gi," ")
    .replace(/<noscript\b[^>]*>[\s\S]*?<\/noscript>/gi," ")
    .replace(/<iframe\b[^>]*>[\s\S]*?<\/iframe>/gi," ")
    .replace(/<svg\b[^>]*>[\s\S]*?<\/svg>/gi," ")
    .replace(/<!--[\s\S]*?-->/g," ");

  const articleMatch =
    body.match(/<article\b[^>]*>([\s\S]*?)<\/article>/i) ||
    body.match(/<main\b[^>]*>([\s\S]*?)<\/main>/i) ||
    body.match(/<body\b[^>]*>([\s\S]*?)<\/body>/i);
  if (articleMatch) body = articleMatch[1];

  body = body
    .replace(/<\/p>/gi,"\n\n").replace(/<\/div>/gi,"\n").replace(/<\/section>/gi,"\n")
    .replace(/<\/h1>|<\/h2>|<\/h3>/gi,"\n\n").replace(/<br\s*\/?>/gi,"\n")
    .replace(/<\/li>/gi,"\n").replace(/<li\b[^>]*>/gi,"* ")
    .replace(/<[^>]+>/g," ");

  body = deepCleanText(body);

  const paragraphs = body.split(/\n{2,}/)
    .map((x)=>deepCleanText(x)).filter(Boolean)
    .filter((x)=>x.length>70).filter((x)=>!isBoilerplate(x)).filter((x)=>!looksLikeLooseCaption(x))
    .slice(0, EXTRACT_MAX_PARAGRAPHS);

  return { url, title: cleanFinalText(title), byline: cleanFinalText(byline), text: hardCap(paragraphs.join("\n\n"), EXTRACT_MAX_CHARS) };
}
function sentenceSplit(text) {
  return String(text || "")
    .replace(/([.?!])\s+(?=[A-Z0-9])/g,"$1|")
    .split("|")
    .map((x)=>fixGrammarLite(cleanFinalText(x)))
    .filter((x)=>x.length>35 && x.length<420);
}
function tokenizeForFrequency(text) {
  const stop = new Set(["the","a","an","and","or","but","if","then","this","that","these","those","to","of","in","on","for","with","as","at","by","from","is","are","was","were","be","been","being","it","its","their","there","they","them","he","she","his","her","you","your","we","our","has","have","had","will","would","can","could","should","may","might","after","before","during","about","into","over","under","more","most","less","least","than","also","not","no","yes","do","does","did"]);
  return String(text || "").toLowerCase().replace(/[^a-z0-9\s-]/g," ").split(/\s+/)
    .filter(Boolean).filter((w)=>w.length>2 && !stop.has(w));
}
function wordFrequency(text) {
  const words = tokenizeForFrequency(text);
  const map = {};
  for (const w of words) map[w] = (map[w] || 0) + 1;
  return map;
}
function entityDiversityScore(text) {
  const s = String(text || "");
  const caps = s.match(/\b[A-Z][a-z]{2,}\b/g) || [];
  const acr = s.match(/\b[A-Z]{2,}\b/g) || [];
  return new Set([...caps, ...acr].map((x)=>x.toLowerCase())).size;
}
function repetitionScore(text) {
  const s = String(text || "").toLowerCase();
  if (!s) return 0;
  const grams = s.replace(/[^a-z0-9\s]/g," ").split(/\s+/).filter(Boolean);
  const seen = {};
  let rep = 0;
  for (let i = 0; i < grams.length - 3; i++) {
    const g = `${grams[i]} ${grams[i+1]} ${grams[i+2]} ${grams[i+3]}`;
    seen[g] = (seen[g] || 0) + 1;
    if (seen[g] >= 2) rep += 1;
  }
  return rep;
}
function routeMatchConfidence(routePath, title, desc, url) {
  const tokens = routeSignalTerms(routePath);
  const hay = `${title || ""} ${desc || ""} ${url || ""}`.toLowerCase();
  let hits = 0;
  for (const t of tokens) if (t && hay.includes(t)) hits++;
  return hits / Math.max(1, tokens.length);
}
function toneForRoute(routePath) {
  const { domainKey } = routeInfo(routePath);
  if (/sports/.test(domainKey)) return "energetic";
  if (/government|law/.test(domainKey)) return "formal";
  if (/health/.test(domainKey)) return "calm";
  if (/finance/.test(domainKey)) return "analytical";
  if (/science/.test(domainKey)) return "curious";
  if (/education/.test(domainKey)) return "clear";
  if (/real-estate/.test(domainKey)) return "practical";
  return "neutral";
}
function applyTone(sentence, tone) {
  const s = String(sentence || "").trim();
  if (!s) return "";
  if (tone === "formal") return s.replace(/\bguys\b/gi,"people");
  if (tone === "analytical") return s.replace(/\bhuge\b/gi,"significant");
  if (tone === "calm") return s.replace(/\bpanic\b/gi,"concern");
  return s;
}
function failsStoryQuality(draft, routePath, feedTitle, feedDesc, articleUrl) {
  const headline = String(draft?.headline || "").trim();
  const lead = String(draft?.lead || "").trim();
  const paras = Array.isArray(draft?.paragraphs) ? draft.paragraphs : [];
  const joined = [headline, lead, ...paras].join("\n\n");

  const tooShort = paras.length < STORY_MIN_PARAGRAPHS || joined.length < STORY_MIN_CHARS;
  const tooRepetitive = repetitionScore(joined) >= 4;
  const lowEntities = entityDiversityScore(joined) <= 3;
  const conf = routeMatchConfidence(routePath, feedTitle, feedDesc, articleUrl);
  const offRoute = routeDepth(routePath) >= 1 && conf < 0.01;
  const echo = headline && joined.toLowerCase().includes(headline.toLowerCase().slice(0, Math.min(60, headline.length))) && paras.length <= 2;
  return tooShort || tooRepetitive || lowEntities || offRoute || echo;
}
function writeCandidate({ routePath, feedTitle, feedDesc, pageTitle, byline, text, url, mut, hive }) {
  const tone = toneForRoute(routePath);
  const raw = deepCleanText(text);
  const sents = sentenceSplit(raw).slice(0, STORY_MAX_SENTENCES);

  const freq = wordFrequency(sents.join(" "));
  const overused = new Set([...(topOverusedPhrases(hive?.global, 10)), ...(topOverusedPhrases(hive?.route, 10))]);

  function sentScore(s, idx) {
    const words = tokenizeForFrequency(s);
    let score = 0;
    for (const w of words) score += freq[w] || 0;
    if (/\bsaid\b|\baccording\b|\breported\b|\bannounced\b|\bconfirmed\b/i.test(s)) score += 5 + mut.factBoost;
    if (/\b\d{4}\b|\b\d+(?:\.\d+)?%|\$\d+|\bmillion\b|\bbillion\b/i.test(s)) score += 3 + mut.numberBoost;
    for (const p of overused) if (p && s.toLowerCase().includes(p)) score -= 2 + mut.overusePenalty;
    if (idx < 3) score += 4;
    if (s.length > 290) score -= 2;
    if (s.length < 70) score -= 1;
    score += idx % 2 === 0 ? mut.evenBoost : mut.oddBoost;
    return score;
  }

  const ranked = sents.map((s,i)=>({ s, sc: sentScore(s,i) })).sort((a,b)=>b.sc-a.sc).map((x)=>x.s);
  const headlineSeed = pageTitle && pageTitle.length > 12 ? pageTitle : ranked[0] || feedTitle || "News Update";
  const headline = safeText(cleanFinalText(String(headlineSeed).replace(/\s*[-|:]\s*[^-|:]+$/, "").replace(/\.$/,"").trim()), 105);
  const subheadlineSeed = cleanFinalText(feedDesc || ranked[1] || "");
  const subheadline = safeText(subheadlineSeed, 220);

  const leadCore = ranked.slice(0,2).join(" ");
  const lead = applyTone(fixGrammarLite(cleanFinalText(leadCore || "")), tone) || applyTone("New developments are shaping the latest coverage.", tone);

  const body = ranked.slice(2,16);
  const buckets = [body.slice(0,2), body.slice(2,4), body.slice(4,6), body.slice(6,8), body.slice(8,10)].filter((b)=>b.length);
  const paragraphs = [];
  for (const b of buckets) {
    const p = b.map((x)=>applyTone(fixGrammarLite(cleanFinalText(x)), tone)).filter(Boolean).join(" ");
    if (p) paragraphs.push(p);
    if (paragraphs.length >= (hive?.route?.targets?.max_paras || STORY_MAX_PARAGRAPHS) || paragraphs.length >= STORY_MAX_PARAGRAPHS) break;
  }

  const finalParas = addTransitions(paragraphs, routePath, hive?.route).slice(0, hive?.route?.targets?.max_paras || STORY_MAX_PARAGRAPHS);
  const whySeed = body[0] ? safeText(cleanFinalText(body[0]), 240) : safeText(cleanFinalText(feedDesc || ""), 240);

  return {
    headline,
    subheadline,
    byline: cleanFinalText(byline || ""),
    lead: cleanFinalText(lead),
    paragraphs: finalParas,
    why_it_matters: cleanFinalText(`Why it matters: ${whySeed || "this story highlights a shift unfolding in real time."}`),
    source_url: canonicalizeUrl(url || ""),
    article_label: "AI-written story",
    article_fine_print: "This is an on-site AI story built for this universe route. It's designed to be read here.",
    seo_title: headline ? `${headline} | ${SITE_NAME}` : `${SITE_NAME}`,
    seo_description: subheadline || cleanFinalText(lead) || SITE_DESCRIPTION,
  };
}
function scoreCandidate(candidate, routePath, feedTitle, feedDesc, articleUrl, hive) {
  const paras = Array.isArray(candidate?.paragraphs) ? candidate.paragraphs : [];
  const joined = [candidate?.headline || "", candidate?.lead || "", ...paras].join("\n\n");

  const rep = repetitionScore(joined);
  const ent = entityDiversityScore(joined);
  const conf = routeMatchConfidence(routePath, feedTitle, feedDesc, articleUrl);
  const offRoutePenalty = routeDepth(routePath) >= 1 && conf < 0.01 ? 28 : 0;
  const shortPenalty = paras.length < (hive?.targets?.min_paras || STORY_MIN_PARAGRAPHS) || joined.length < (hive?.targets?.min_chars || STORY_MIN_CHARS) ? 35 : 0;
  const lowEntityPenalty = ent <= 3 ? 16 : ent <= 5 ? 8 : 0;

  const factSignals = (joined.match(/\bsaid\b|\baccording\b|\breported\b|\bannounced\b|\bconfirmed\b/gi) || []).length;
  const numbers = (joined.match(/\b\d{4}\b|\b\d+(?:\.\d+)?%|\$\d+|\bmillion\b|\bbillion\b/gi) || []).length;
  const factDensity = factSignals + numbers;
  const lowFactPenalty = factDensity <= 1 ? 10 : 0;

  const penalties = hive?.penalties || defaultHiveMemory().penalties;
  const base = 100 + paras.length * 6 + Math.min(joined.length / 120, 25) + Math.min(ent * 2, 22) + Math.min(factDensity * 3, 24);

  return (
    base -
    rep * 7 * (penalties.repetition || 1) -
    shortPenalty * (penalties.shortness || 1) -
    offRoutePenalty * (penalties.offroute || 1) -
    lowEntityPenalty * (penalties.low_entity_diversity || 1) -
    lowFactPenalty * (penalties.low_fact_density || 1)
  );
}
function learnFromOutput(mem, candidate) {
  const m = mem || defaultHiveMemory();
  m.style = m.style || defaultHiveMemory().style;
  m.style.avoid_openers = m.style.avoid_openers || {};
  m.style.lead_openers = m.style.lead_openers || {};

  const joined = [candidate?.headline || "", candidate?.lead || "", ...(candidate?.paragraphs || [])].join(" ");
  const low = joined.toLowerCase();

  for (const p of ["new details may be limited","coverage is evolving fast","will keep updating","in real time"])
    if (low.includes(p)) bumpPhrase(m.overused_phrases, p);

  const paras = Array.isArray(candidate?.paragraphs) ? candidate.paragraphs : [];
  let openerHits = 0;
  for (const p of paras) {
    const s = String(p || "").trim();
    if (/^(Still,|That said,|In contrast,|By comparison,|In response,|As a result,|In the short term,|Over the longer term,)/i.test(s)) {
      openerHits++;
      bumpMap(m.style.avoid_openers, s.split(" ")[0]);
    }
  }
  if (openerHits >= 2) m.style.transition_rate = clamp01((m.style.transition_rate || 0.18) - 0.03);
  else m.style.transition_rate = clamp01((m.style.transition_rate || 0.18) + 0.003);

  const rep = repetitionScore(joined);
  if (rep >= 5) m.penalties.repetition = Math.min(2.5, (m.penalties.repetition || 1) + 0.05);
  if (paras.length < STORY_MIN_PARAGRAPHS || joined.length < STORY_MIN_CHARS) m.penalties.shortness = Math.min(2.5, (m.penalties.shortness || 1) + 0.06);
  const ent = entityDiversityScore(joined);
  if (ent <= 3) m.penalties.low_entity_diversity = Math.min(2.5, (m.penalties.low_entity_diversity || 1) + 0.05);

  m.n = (m.n || 0) + 1;
  return m;
}
async function hiveWriteStory(cache, origin, routePath, inputs) {
  const hive = await readHive(cache, origin, routePath);
  hive.route = ensureRouteStyle(routePath, hive.route);
  hive.global = hive.global || defaultHiveMemory();

  const muts = [
    { factBoost: 0, numberBoost: 0, overusePenalty: 0, evenBoost: 0, oddBoost: 0 },
    { factBoost: 2, numberBoost: 1, overusePenalty: 1, evenBoost: 1, oddBoost: 0 },
    { factBoost: 1, numberBoost: 2, overusePenalty: 2, evenBoost: 0, oddBoost: 1 },
    { factBoost: 3, numberBoost: 1, overusePenalty: 2, evenBoost: 1, oddBoost: 1 },
  ].slice(0, MUTATION_COUNT);

  const candidates = muts.map((mut)=>writeCandidate({ ...inputs, mut, hive: { global: hive.global, route: hive.route } }));

  let best = candidates[0], bestScore = -1e9;
  for (const c of candidates) {
    const sc = scoreCandidate(c, routePath, inputs.feedTitle, inputs.feedDesc, inputs.url, hive.route);
    if (sc > bestScore) { bestScore = sc; best = c; }
  }

  hive.global = learnFromOutput(hive.global, best);
  hive.route = learnFromOutput(hive.route, best);
  await writeHive(cache, origin, routePath, hive);

  return { story: best, debug: { score: bestScore } };
}
async function handleArticle(request) {
  const cache = caches.default;

  try {
    const reqUrl = new URL(request.url);
    const articleUrl = reqUrl.searchParams.get("url") || "";
    const feedTitle = cleanFinalText(reqUrl.searchParams.get("title") || "");
    const feedDesc = cleanFinalText(reqUrl.searchParams.get("desc") || "");
    const hub = String(reqUrl.searchParams.get("hub") || "/").trim() || "/";
    const routePath = ROUTES.has(hub) ? hub : "/";
    const origin = reqUrl.origin;

    if (!articleUrl) return json({ ok: false, error: "Missing article URL" }, 400);
    if (!/^https?:\/\//i.test(articleUrl)) return json({ ok: false, error: "Article URL must start with http:// or https://" }, 400);
    if (isGoogleHostUrl(articleUrl)) return json({ ok: false, error: "Google is disabled." }, 400);

    const routeCheckItem = { title: feedTitle, description: feedDesc, link: articleUrl };
    if (!passesHardRouteLock(routeCheckItem, routePath)) {
      return json({ ok: false, error: "This story does not match the current page lock." }, 400);
    }

    const canonical = canonicalizeUrl(articleUrl);
    const cacheKeyUrl = articleCacheUrl(origin, routePath, canonical);
    const cached = await edgeGetJson(cache, cacheKeyUrl);
    if (cached?.ok && cached?.article?.headline) return json(cached, 200, 60);

    let fetchedUrl = canonical;
    let extracted = { url: fetchedUrl, title: "", byline: "", text: "" };
    let blocked = false;

    try {
      const res = await fetchWithTimeout(articleUrl, {
        headers: { "user-agent": "Mozilla/5.0 Cloudflare Article Reader", accept: "text/html,*/*;q=0.8" },
        cf: { cacheTtl: 300, cacheEverything: true },
      }, ARTICLE_FETCH_TIMEOUT_MS);

      if (!res.ok) blocked = true;
      else {
        fetchedUrl = canonicalizeUrl(res.url || articleUrl);
        const html = await res.text();
        extracted = extractArticleContent(html, fetchedUrl);
      }
    } catch {
      blocked = true;
    }

    const extractedCheckItem = {
      title: extracted.title || feedTitle,
      description: `${feedDesc} ${String(extracted.text || "").slice(0, 700)}`,
      link: fetchedUrl,
    };

    if (!passesHardRouteLock(extractedCheckItem, routePath)) {
      return json({ ok: false, error: "Fetched article content does not match the current page lock." }, 400);
    }

    const basePieces = [extracted.title || feedTitle, feedDesc || "", extracted.text || ""].filter(Boolean);
    const baseText = hardCap(deepCleanText(basePieces.join("\n\n")), EXTRACT_MAX_CHARS);

    let { story } = await hiveWriteStory(cache, origin, routePath, {
      routePath,
      feedTitle,
      feedDesc,
      pageTitle: extracted.title || feedTitle,
      byline: extracted.byline || "",
      text: baseText,
      url: fetchedUrl,
    });

    if (failsStoryQuality(story, routePath, extracted.title || feedTitle, feedDesc, fetchedUrl)) story._fallback = true;

    if (!Array.isArray(story.paragraphs)) story.paragraphs = [];
    story.paragraphs = story.paragraphs.filter(Boolean).slice(0, STORY_MAX_PARAGRAPHS);
    if (blocked) story._fallback = true;

    story.article_fine_print = "This is an on-site AI story built for this universe route. It's designed to be read here.";

    const payload = { ok: true, source: { url: fetchedUrl }, article: story };
    await edgePutJson(cache, cacheKeyUrl, payload, ARTICLE_CACHE_TTL_SECONDS);
    return json(payload, 200, 60);
  } catch {
    return json({
      ok: true,
      source: { url: "" },
      article: {
        headline: "Story brief",
        subheadline: "Coverage is updating on this route.",
        byline: "",
        lead: "The hive mind writer is generating a clean story for this page.",
        paragraphs: [
          "This route updates continuously and will refresh as coverage changes.",
          "If an upstream source restricts access, the route will still show live cards and on-site writing.",
          "That keeps the story readable inside this world - without sending readers away.",
        ],
        why_it_matters: "Why it matters: nonstop reading without broken text or empty stories.",
        source_url: "",
        article_label: "AI-written story",
        article_fine_print: "On-site fallback story.",
        seo_title: `Story brief | ${SITE_NAME}`,
        seo_description: SITE_DESCRIPTION,
        _fallback: true,
      },
    }, 200, 10);
  }
}

// Works on your workers.dev domain like:
//   /health     -> /health-care
//   /energy     -> /energy
//   /technology -> /information-technology
//
// Users visit:
//   https://myaigptnews.quantumintelligencepulse.workers.dev/health
// and that “site” is locked to that category (same template, different branding).
const PATH_LOCK = {
  "/health": "/health-care",
  "/energy": "/energy",
  "/materials": "/materials",
  "/industrials": "/industrials",
  "/sports": "/sports",
  "/realestate": "/real-estate",
  "/technology": "/information-technology",
  "/finance": "/financials",
  "/crypto": "/financials/crypto",
  "/science": "/science-and-mathematics",
  "/education": "/education-and-knowledge",
  "/government": "/government-law-and-civilization",
};

function _pathKey(pathname) {
  const p = String(pathname || "/").toLowerCase().replace(/\/+$/, "") || "/";
  return p;
}

// If request is /health or /health/... => return locked hub for that “site”.
function _lockedHubForPath(pathname) {
  const p = String(pathname || "/");
  const parts = p.split("/").filter(Boolean);
  if (!parts.length) return "";
  const root = "/" + parts[0].toLowerCase();
  const hub = PATH_LOCK[root];
  return hub && ROUTES.has(hub) ? hub : "";
}

// Keep users inside the locked “site” root (e.g. /health only)
function _lockedRootForPath(pathname) {
  const p = String(pathname || "/");
  const parts = p.split("/").filter(Boolean);
  if (!parts.length) return "";
  const root = "/" + parts[0].toLowerCase();
  return PATH_LOCK[root] ? root : "";
}

function _rewriteWithHub(request, hub) {
  const u = new URL(request.url);
  u.searchParams.set("hub", hub);
  return new Request(u.toString(), request);
}

// Per-site naming (swap branding per locked hub)
function _siteNameForHub(hub) {

}

// Post-process rendered HTML for per-site branding
function _swapHtmlBranding(html, siteName) {
  if (!siteName) return html;
  let out = String(html || "");

  // <title>...</title>
  out = out.replace(/<title>[\s\S]*?<\/title>/i, `<title>${escapeHtml(siteName)}</title>`);

  // og:site_name
  out = out.replace(
    /<meta\s+property="og:site_name"\s+content="[^"]*"\s*\/?>/i,
    `<meta property="og:site_name" content="${escapeHtml(siteName)}"/>`
  );

  // og:title
  out = out.replace(
    /<meta\s+property="og:title"\s+content="[^"]*"\s*\/?>/i,
    `<meta property="og:title" content="${escapeHtml(siteName)}"/>`
  );

  // Footer bold site name (first <b> inside footer block)
  out = out.replace(/<footer><div><b>[\s\S]*?<\/b>/i, `<footer><div><b>${escapeHtml(siteName)}</b>`);

  return out;
}

function _edNorm(s) { return String(s || "").replace(/\s+/g, " ").trim(); }
function _edLow(s) { return _edNorm(s).toLowerCase(); }
function _edHardCap(s, n) { return hardCap(_edNorm(s), n); }
function _edStripDup(a, b) {
  const A = _edLow(a), B = _edLow(b);
  if (!A || !B) return a;
  if (A === B) return "";
  if (A.startsWith(B)) return _edNorm(a.slice(b.length));
  return a;
}
function _edUniqParas(paras) {
  const seen = new Set();
  const out = [];
  for (const p of paras || []) {
    const t = _edNorm(p);
    const k = _edLow(t).slice(0, 180);
    if (!t || t.length < 70) continue;
    if (seen.has(k)) continue;
    seen.add(k);
    out.push(t);
  }
  return out;
}
function _edSentencePool(text) {
  return (sentenceSplit(deepCleanText(text || "")) || [])
    .map(_edNorm)
    .filter((x) => x.length >= 45 && x.length <= 360);
}
function _edBuildParas(sents, targetParas) {
  const out = [];
  let i = 0;
  while (out.length < targetParas && i < sents.length) {
    const p = _edNorm([sents[i++], sents[i++], sents[i++]].filter(Boolean).join(" "));
    if (p.length >= 160) out.push(p);
  }
  return out;
}
function _edFinalizeStory(story, routePath) {
  const minParas = STORY_MIN_PARAGRAPHS || 3;
  const minChars = STORY_MIN_CHARS || 900;
  const maxParas = STORY_MAX_PARAGRAPHS || 6;

  story.headline = _edHardCap(cleanFinalText(story?.headline || ""), 110);
  story.subheadline = _edHardCap(cleanFinalText(story?.subheadline || ""), 240);
  story.lead = _edHardCap(cleanFinalText(story?.lead || ""), 520);

  story.subheadline = _edStripDup(story.subheadline, story.headline);
  story.lead = _edStripDup(story.lead, story.headline);
  story.lead = _edStripDup(story.lead, story.subheadline);

  let paras = Array.isArray(story?.paragraphs) ? story.paragraphs : [];
  paras = _edUniqParas(paras).slice(0, maxParas);

  const seedText = deepCleanText(
    [story.headline, story.subheadline, story.lead, ...paras, story.why_it_matters].filter(Boolean).join("\n\n")
  );
  const sents = _edSentencePool(seedText);

  if (paras.length < minParas) {
    const built = _edBuildParas(sents, Math.max(minParas, 4));
    paras = _edUniqParas([...paras, ...built]).slice(0, maxParas);
    story._janitorial_fill = true;
  }

  const why = _edLow(story?.why_it_matters || "");
  const leadL = _edLow(story?.lead || "");
  const subL = _edLow(story?.subheadline || "");
  if (!why || why.includes(leadL.slice(0, 80)) || why.includes(subL.slice(0, 80))) {
    const pick = sents.find((x) => {
      const lx = _edLow(x);
      return lx && !lx.includes(leadL.slice(0, 60)) && !lx.includes(subL.slice(0, 60));
    });
    const seed = _edHardCap(pick || story.lead || story.subheadline || "", 240);
    story.why_it_matters = cleanFinalText(`Why it matters: ${seed || "this development changes expectations for this route."}`);
    story._janitorial_why = true;
  }

  story.paragraphs = _edUniqParas(paras).slice(0, maxParas);

  const joined = [story.headline, story.subheadline, story.lead, ...story.paragraphs].join("\n\n");
  if (story.paragraphs.length < minParas || joined.length < minChars) {
    const tail = _edBuildParas(sents.slice(6), 2);
    story.paragraphs = _edUniqParas([...story.paragraphs, ...tail]).slice(0, maxParas);
    story._janitorial_finish = true;
  }

  if (!story.paragraphs.length) {
    story.paragraphs = [
      "Developments are evolving and this route is updating as new details emerge.",
      "This page keeps coverage readable on-site even when upstream sources restrict access.",
      "Check back soon for additional context, reactions, and confirmed specifics.",
    ].slice(0, maxParas);
    story._janitorial_emergency = true;
  }

  story.article_fine_print = "This is an on-site AI story built for this universe route. It's designed to be read here.";
  return story;
}

function _hubCandidates() {
  const hubs = ["/", "/explore"];
  for (const d of SPINE || []) {
    const dh = `/${d.key}`;
    if (ROUTES.has(dh)) hubs.push(dh);
    for (const c of d.children || []) {
      const ch = `/${d.key}/${c.key}`;
      if (ROUTES.has(ch)) hubs.push(ch);
    }
  }
  return hubs;
}
function _hubScore(hub, item) {
  const info = routeInfo(hub);
  const hay = itemHaystack({
    title: item?.title || "",
    description: item?.description || "",
    link: item?.link || "",
    feedUrl: item?.feedUrl || "",
  });

  const routeTerms = routeSignalTerms(hub);
  const routeHits = countSignalHits({ ...item, feedUrl: item?.feedUrl || "" }, routeTerms);

  const domainTerms = info.domainKey ? (DOMAIN_SIGNAL_MAP[info.domainKey] || []) : [];
  const catTerms = info.catKey ? (CATEGORY_SIGNAL_MAP[info.catKey] || []) : [];

  const domainHits = domainTerms.length ? domainTerms.filter((t) => hay.includes(String(t).toLowerCase())).length : 0;
  const catHits = catTerms.length ? catTerms.filter((t) => hay.includes(String(t).toLowerCase())).length : 0;

  const depth = routeDepth(hub);
  const specificityBonus = depth >= 2 ? 2.0 : depth === 1 ? 0.8 : 0;

  return routeHits * 4.0 + catHits * 2.5 + domainHits * 1.8 + specificityBonus;
}
function _bestHubForItem(item, currentHub) {
  const cur = routeInfo(currentHub || "/");
  const hubs = _hubCandidates();

  let best = "/";
  let bestScore = -1e9;

  for (const h of hubs) {
    if (h === "/explore") continue;
    const inf = routeInfo(h);

    let sc = _hubScore(h, item);
    if (cur.domainKey && inf.domainKey && inf.domainKey !== cur.domainKey) sc -= 1.2;

    if (sc > bestScore) {
      bestScore = sc;
      best = h;
    }
  }

  const curScore = currentHub ? _hubScore(currentHub, item) : -1e9;
  const shouldReroute = best !== currentHub && bestScore >= curScore + 2.5 && bestScore >= 5.0;

  return { bestHub: best, bestScore, curScore, shouldReroute };
}
function _rewriteRequestHub(request, newHub) {
  const u = new URL(request.url);
  u.searchParams.set("hub", newHub);
  return new Request(u.toString(), request);
}

async function _handleArticleWithCrew(request) {
  const u = new URL(request.url);
  const hub = String(u.searchParams.get("hub") || "/").trim() || "/";
  const routePath = ROUTES.has(hub) ? hub : "/";

  const item = {
    title: cleanFinalText(u.searchParams.get("title") || ""),
    description: cleanFinalText(u.searchParams.get("desc") || ""),
    link: String(u.searchParams.get("url") || ""),
    feedUrl: "",
  };

  let res = await handleArticle(request);

  if (res && (res.status === 400 || res.status === 409)) {
    try {
      const data = await res.clone().json();
      const msg = String(data?.error || "");
      if (msg.includes("does not match the current page lock") || msg.includes("Fetched article content does not match")) {
        const pick = _bestHubForItem(item, routePath);
        if (pick.shouldReroute && ROUTES.has(pick.bestHub)) {
          const reroutedReq = _rewriteRequestHub(request, pick.bestHub);
          const reroutedRes = await handleArticle(reroutedReq);
          if (reroutedRes?.headers?.get("content-type")?.toLowerCase().includes("application/json")) {
            const reroutedData = await reroutedRes.clone().json();
            if (reroutedData?.ok && reroutedData?.article) {
              reroutedData.article._rerouted_from = routePath;
              reroutedData.article._rerouted_to = pick.bestHub;
              reroutedData.article = _edFinalizeStory(reroutedData.article, pick.bestHub);
              return json(reroutedData, 200, 60);
            }
          }
          res = reroutedRes;
        }
      }
    } catch {}
  }

  try {
    const ct = (res.headers.get("content-type") || "").toLowerCase();
    if (ct.includes("application/json")) {
      const data = await res.clone().json();
      if (!data?.ok) {
        const pick = _bestHubForItem(item, routePath);
        const fallbackHub = pick?.bestHub && ROUTES.has(pick.bestHub) ? pick.bestHub : routePath;

        const story = _edFinalizeStory(
          {
            headline: safeText(item.title || "Story brief", 105),
            subheadline: safeText(item.description || "Coverage is updating on this route.", 220),
            byline: "",
            lead: cleanFinalText(
              item.description ||
                "This route is generating on-site coverage using the best available signals from the feed card."
            ),
            paragraphs: [
              cleanFinalText("This story is being organized into the correct page world based on signal terms and route taxonomy."),
              cleanFinalText("If an upstream source restricts access or provides thin metadata, the hub will still produce a readable on-site story."),
              cleanFinalText("Check back as the route refreshes; additional sources and context are merged into the on-site write-up."),
            ],
            why_it_matters: cleanFinalText(`Why it matters: ${safeText(item.description || item.title || "route-locked coverage improves relevance.", 220)}`),
            source_url: canonicalizeUrl(item.link || ""),
            article_label: "AI-written story",
            article_fine_print: "On-site fallback story (auto-reroute enabled).",
            seo_title: `${safeText(item.title || "Story brief", 90)} | ${SITE_NAME}`,
            seo_description: safeText(item.description || SITE_DESCRIPTION, 200),
            _fallback: true,
            _rerouted_to: fallbackHub,
          },
          fallbackHub
        );

        return json({ ok: true, source: { url: item.link || "" }, article: story }, 200, 10);
      }

      if (data?.ok && data?.article) {
        data.article = _edFinalizeStory(data.article, routePath);
        return json(data, res.status, 60);
      }
    }
  } catch {}

  return res;
}

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);

    const lockedHub = _lockedHubForPath(url.pathname);     // "" if not locked site
    const lockedRoot = _lockedRootForPath(url.pathname);   // "/health" etc, "" if not locked

    const origin = String(env.BASE_URL || url.origin).replace(/\/$/, "");
    const path = url.pathname.replace(/\/+$/, "") || "/";

    if (request.method === "OPTIONS") {
      return new Response(null, { status: 204, headers: { ...SECURITY_HEADERS } });
    }

    if (path === "/api/img") {
      const seed = (url.searchParams.get("seed") || "news").slice(0, 140);
      const w = clamp(parseInt(url.searchParams.get("w") || "1200", 10) || 1200, 64, 1600);
      const h = clamp(parseInt(url.searchParams.get("h") || "800", 10) || 800, 64, 1200);
      return Response.redirect(`https://picsum.photos/seed/${encodeURIComponent(seed)}/${w}/${h}`, 302);
    }

    if (path === "/api/imgproxy") {
      const u = url.searchParams.get("u");
      const seed = (url.searchParams.get("seed") || `proxy:${hash32(u || "x")}`).slice(0, 140);
      if (!u) return Response.redirect(`${origin}/api/img?seed=${encodeURIComponent(seed)}&w=1200&h=800`, 302);

      try {
        const res = await fetchWithTimeout(
          u,
          { headers: { "user-agent": "Mozilla/5.0", accept: "image/*,*/*;q=0.8" } },
          6500
        );
        const ct = (res.headers.get("content-type") || "").toLowerCase();
        if (!res.ok || !ct.startsWith("image/")) {
          return Response.redirect(`${origin}/api/img?seed=${encodeURIComponent(seed)}&w=1200&h=800`, 302);
        }
        return new Response(res.body, {
          headers: {
            "content-type": ct || "image/jpeg",
            "cache-control": "public, max-age=86400, s-maxage=86400",
            ...SECURITY_HEADERS,
          },
        });
      } catch {
        return Response.redirect(`${origin}/api/img?seed=${encodeURIComponent(seed)}&w=1200&h=800`, 302);
      }
    }

    // ✅ PATH-LOCKED “site”: force hub for writer
    if (path === "/api/article") {
      const req = lockedHub ? _rewriteWithHub(request, lockedHub) : request;
      return _handleArticleWithCrew(req);
    }

    // ✅ PATH-LOCKED “site”: force hub for feed
    if (path === "/api/live") {
      const hub = lockedHub || String(url.searchParams.get("hub") || "/").trim() || "/";
      const routePath = ROUTES.has(hub) ? hub : "/";
      return handleLive(request, origin, routePath);
    }

    // ✅ PATH-LOCKED “site”: keep user inside that root (e.g. /health)
    if (lockedHub && lockedRoot) {
      // Any deep path like /health/something -> /health
      if (_pathKey(url.pathname) !== lockedRoot) {
        return Response.redirect(`${origin}${lockedRoot}`, 302);
      }

      // Render locked hub but keep requestUrl = /health for canonical
      const html = renderHTML(request.url, lockedHub, origin);
      const siteName = _siteNameForHub(lockedHub);
      return new Response(_swapHtmlBranding(html, siteName), {
        headers: {
          "content-type": "text/html; charset=UTF-8",
          "cache-control": "no-store",
          ...SECURITY_HEADERS,
        },
      });
    }

    // normal (non-locked) behavior
    const activePath = ROUTES.has(path) ? path : "/";
    return new Response(renderHTML(request.url, activePath, origin), {
      headers: {
        "content-type": "text/html; charset=UTF-8",
        "cache-control": "no-store",
        ...SECURITY_HEADERS,
      },
    });
  },

  async scheduled(event, env, ctx) {
    // No cron warmers required.
  },
};
