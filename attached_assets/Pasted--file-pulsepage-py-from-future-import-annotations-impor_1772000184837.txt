def db() -> sqlite3.Connection:

def init_db() -> None:
    con = db()
    cur = con.cursor()
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS config (
          id INTEGER PRIMARY KEY CHECK (id=1),
          json TEXT NOT NULL
        )
        """
    )
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS conversations (
          id TEXT PRIMARY KEY,
          title TEXT NOT NULL,
          created_at REAL NOT NULL,
          updated_at REAL NOT NULL
        )
        """
    )
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS messages (
          id INTEGER PRIMARY KEY AUTOINCREMENT,
          convo_id TEXT NOT NULL,
          role TEXT NOT NULL,
          content TEXT NOT NULL,
          created_at REAL NOT NULL
        )
        """
    )
    cur.execute("INSERT OR IGNORE INTO config (id, json) VALUES (1, ?)", (json.dumps(DEFAULT_CONFIG),))
    con.commit()
    con.close()

def load_config() -> Dict[str, Any]:
    con = db()
    row = con.execute("SELECT json FROM config WHERE id=1").fetchone()
    con.close()
    merged = dict(DEFAULT_CONFIG)
    if row:
        try:
            data = json.loads(row["json"])
            for k in merged:
                if k in data:
                    merged[k] = data[k]
        except Exception:
            pass

    # Enforce public env policy (env wins)
    if PUBLIC_DISABLE_MARKETS:
        merged["ENABLE_MARKETS"] = False
    if PUBLIC_DISABLE_SEARCH:
        merged["ENABLE_SEARCH"] = False

    return merged

def save_config(cfg: Dict[str, Any]) -> None:
    merged = dict(DEFAULT_CONFIG)
    for k in merged:
        if k in cfg:
            merged[k] = cfg[k]

    # Enforce public env policy (env wins)
    if PUBLIC_DISABLE_MARKETS:
        merged["ENABLE_MARKETS"] = False
    if PUBLIC_DISABLE_SEARCH:
        merged["ENABLE_SEARCH"] = False

    con = db()
    con.execute("UPDATE config SET json=? WHERE id=1", (json.dumps(merged),))
    con.commit()
    con.close()

def require_admin(req: Request) -> None:
    token = (req.headers.get("X-Admin-Token") or "").strip()
    if not token:
        raise HTTPException(401, "Unauthorized")
    if hashlib.sha256(token.encode("utf-8")).hexdigest() != ADMIN_PASSWORD_HASH:
        raise HTTPException(401, "Unauthorized")

@dataclass
class TokenBucket:
    tokens: float
    updated_at: float

CHAT_BUCKETS: Dict[str, TokenBucket] = {}
PRICE_BUCKETS: Dict[str, TokenBucket] = {}

def _ip(req: Request) -> str:
    xf = (req.headers.get("x-forwarded-for") or "").split(",")[0].strip()
    return xf or (req.client.host if req.client else "unknown")

def _take(bucket_map: Dict[str, TokenBucket], key: str, per_min: float, cost: float = 1.0) -> None:
    cap = max(1.0, per_min)
    refill_per_sec = per_min / 60.0
    now = time.time()
    b = bucket_map.get(key)
    if b is None:
        bucket_map[key] = TokenBucket(tokens=cap - cost, updated_at=now)
        return
    elapsed = max(0.0, now - b.updated_at)
    b.tokens = min(cap, b.tokens + elapsed * refill_per_sec)
    b.updated_at = now
    if b.tokens < cost:
        raise HTTPException(429, "Rate limited")
    b.tokens -= cost

def rate_limit_chat(req: Request) -> None:
    _take(CHAT_BUCKETS, _ip(req), CHAT_REQ_PER_MIN)

def rate_limit_price(req: Request) -> None:
    _take(PRICE_BUCKETS, _ip(req), PRICE_REQ_PER_MIN)

_PRICE_CACHE: Dict[str, Tuple[float, float]] = {}
_PRICE_CACHE_TTL = 20.0

def _cache_get(sym: str) -> Optional[float]:
    item = _PRICE_CACHE.get(sym.upper())
    if not item:
        return None
    ts, val = item
    return val if (time.time() - ts <= _PRICE_CACHE_TTL) else None

def _cache_set(sym: str, val: float) -> None:
    _PRICE_CACHE[sym.upper()] = (time.time(), float(val))

def _normalize_symbol(symbol: str) -> str:
    s = symbol.strip().upper()
    if s in {"BTC", "BITCOIN"}:
        return "BTC-USD"
    if s in {"ETH", "ETHEREUM"}:
        return "ETH-USD"
    return s

def _enforce_allowlist(symbol: str) -> None:
    if not TICKER_ALLOWLIST:
        return
    base = symbol.upper().strip()
    # allowlist can contain BTC-USD etc.
    if base not in TICKER_ALLOWLIST:
        raise HTTPException(403, "Ticker not allowed on this public deployment")

def _yf_price_sync(symbol: str) -> Optional[float]:
    sym = _normalize_symbol(symbol)
    try:
        t = yf.Ticker(sym)
        try:
            fi = getattr(t, "fast_info", None)
            if fi:
                lp = fi.get("last_price") or fi.get("lastPrice")
                if lp is not None:
                    return float(lp)
        except Exception:
            pass
        try:
            df = t.history(period="1d", interval="1m")
            if df is not None and not df.empty:
                return float(df["Close"].iloc[-1])
        except Exception:
            pass
        df = t.history(period="5d", interval="1d")
        if df is not None and not df.empty:
            return float(df["Close"].iloc[-1])
        return None
    except Exception:
        return None

async def get_price(symbol: str) -> Dict[str, Any]:
    sym = _normalize_symbol(symbol)
    _enforce_allowlist(sym)

    cached = _cache_get(sym)
    if cached is not None:
        return {"symbol": sym, "price": cached, "cached": True, "ts": time.time()}

    price = await asyncio.to_thread(_yf_price_sync, sym)
    if price is None:
        raise HTTPException(404, f"Could not fetch price for {sym}")

    _cache_set(sym, float(price))
    return {"symbol": sym, "price": float(price), "cached": False, "ts": time.time()}

def detect_finance_symbols(text: str) -> List[str]:
    symbols: set[str] = set()
    for w in text.replace(",", " ").replace("\n", " ").split():
        if w.startswith("$") and len(w) > 1:
            sym = "".join(ch for ch in w[1:] if ch.isalnum() or ch in ".-^")
            if sym:
                symbols.add(sym.upper())
    upper = text.upper()
    if "BITCOIN" in upper:
        symbols.add("BTC")
    if "ETHEREUM" in upper:
        symbols.add("ETH")
    for token in re.findall(r"\b[A-Za-z]{1,5}\b", text):
        if token.upper() in {"SPY", "QQQ", "BTC", "ETH"}:
            symbols.add(token.upper())
    return sorted(symbols)

_WEB_CACHE: Dict[str, Tuple[float, str]] = {}
_WEB_CACHE_TTL = 180.0

def looks_like_search_question(text: str) -> bool:
    t = text.lower()
    return any(x in t for x in ["what is", "who is", "when did", "why", "news", "latest", "define", "meaning of"])

async def duckduckgo_search(query: str, max_results: int = 5) -> str:
    q = query.strip()
    if not q:
        return ""
    cached = _WEB_CACHE.get(q)
    if cached and (time.time() - cached[0] <= _WEB_CACHE_TTL):
        return cached[1]

    def _run() -> List[str]:
        out: List[str] = []
        try:
            with DDGS() as ddgs:
                for r in ddgs.text(q, max_results=max_results):
                    body = (r.get("body") or r.get("snippet") or "").strip()
                    if body:
                        out.append(body)
        except Exception:
            pass
        return out

    snippets = await asyncio.to_thread(_run)
    text = "\n".join(snippets[:max_results]) if snippets else ""
    _WEB_CACHE[q] = (time.time(), text)
    return text

GROQ_URL = "https://api.groq.com/openai/v1/chat/completions"

def _build_prompt(history: List[Dict[str, str]], user_text: str, data_block: str, background_block: str) -> str:
    def compact(items: List[Dict[str, str]], max_turns: int = 8) -> str:
        tail = items[-(max_turns * 2) :]
        out: List[str] = []
        for it in tail:
            role = (it.get("role") or "").upper()
            content = (it.get("content") or "").strip()
            if content:
                out.append(f"{role}: {content}")
        return "\n".join(out)

    return "\n".join(
        [
            "You are the Pulse Deep Think engine.",
            "Rules:",
            "- If DATA contains prices, use them as truth.",
            "- Be direct, grounded, no tool talk.",
            "- If asked for financial advice, give educational info + risks, not guarantees.",
            "",
            "=== HISTORY ===",
            compact(history) or "(none)",
            "",
            "=== USER ===",
            user_text,
            "",
            "=== BACKGROUND ===",
            background_block or "(none)",
            "",
            "=== DATA ===",
            data_block or "(none)",
            "",
            "Answer now.",
        ]
    )

async def groq_stream(prompt: str) -> Iterable[str]:
    if not GROQ_API_KEY:
        yield "Server missing GROQ_API_KEY."
        return

    headers = {"Authorization": f"Bearer {GROQ_API_KEY}", "Content-Type": "application/json"}
    payload = {
        "model": GROQ_MODEL,
        "messages": [{"role": "user", "content": prompt}],
        "temperature": 0.4,
        "max_tokens": 900,
        "stream": True,
    }

    timeout = aiohttp.ClientTimeout(total=60)
    async with aiohttp.ClientSession(timeout=timeout) as session:
        async with session.post(GROQ_URL, json=payload, headers=headers) as resp:
            if resp.status >= 400:
                raw = await resp.text()
                yield f"GROQ error {resp.status}: {raw[:600]}"
                return

            async for line_bytes in resp.content:
                line = line_bytes.decode("utf-8", errors="ignore").strip()
                if not line:
                    continue
                if not line.startswith("data:"):
                    continue
                data = line[5:].strip()
                if data == "[DONE]":
                    break
                try:
                    obj = json.loads(data)
                    delta = obj["choices"][0]["delta"].get("content")
                    if delta:
                        yield delta
                except Exception:
                    continue

@app.post("/api/chat/stream")
async def api_chat_stream(req: Request, payload: ChatStartPayload) -> StreamingResponse:
    rate_limit_chat(req)

    cfg = load_config()
    convo_id = payload.convo_id.strip()
    user_text = payload.message.strip()
    if not convo_id or not user_text:
        raise HTTPException(400, "convo_id and message required")

    add_message(convo_id, "user", user_text)

    enable_markets = bool(cfg.get("ENABLE_MARKETS", True))
    enable_search = bool(cfg.get("ENABLE_SEARCH", True))

    symbols = detect_finance_symbols(user_text)[:6]
    data_lines: List[str] = []
    finance_lines: List[str] = []

    if enable_markets and symbols and not PUBLIC_DISABLE_MARKETS:
        normalized = [_normalize_symbol(s) for s in symbols]
        results = await asyncio.gather(*[get_price(s) for s in normalized], return_exceptions=True)
        for r in results:
            if isinstance(r, Exception):
                continue
            line = f"{r['symbol']} price: {r['price']}"
            finance_lines.append(line)
            data_lines.append(f"{r['symbol']}_DATA: {line}")

    web_context = ""
    if enable_search and not PUBLIC_DISABLE_SEARCH and looks_like_search_question(user_text):
        web_context = await duckduckgo_search(user_text, max_results=5)

    background_parts = [p for p in ["\n".join(finance_lines), web_context] if p]
    background_block = "\n\n".join(background_parts)
    data_block = "\n".join(data_lines) if data_lines else "(none)"

    messages = get_messages(convo_id, limit=160)
    history_for_prompt = [{"role": m["role"], "content": m["content"]} for m in messages[:-1]]
    prompt = _build_prompt(history_for_prompt, user_text, data_block, background_block)

    async def event_gen() -> Iterable[bytes]:
        full: List[str] = []
        meta = {"type": "meta", "data": {"prices": data_lines, "markets_enabled": enable_markets}}
        yield f"event: meta\ndata: {json.dumps(meta)}\n\n".encode("utf-8")

        try:
            async for chunk in groq_stream(prompt):
                full.append(chunk)
                yield f"event: delta\ndata: {json.dumps({'type':'delta','data':chunk})}\n\n".encode("utf-8")
        except Exception as e:
            yield f"event: error\ndata: {json.dumps({'type':'error','data':str(e)})}\n\n".encode("utf-8")
        finally:
            final_text = "".join(full).strip() or "(no reply)"
            add_message(convo_id, "assistant", final_text)
            yield f"event: done\ndata: {json.dumps({'type':'done'})}\n\n".encode("utf-8")

    return StreamingResponse(event_gen(), media_type="text/event-stream")
