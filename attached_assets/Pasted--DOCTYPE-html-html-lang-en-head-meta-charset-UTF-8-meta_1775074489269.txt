<script>
const CATS = {
  food:{label:'Food',icon:'🍔',color:'#f59e0b'},
  transport:{label:'Transport',icon:'🚗',color:'#06b6d4'},
  housing:{label:'Housing',icon:'🏠',color:'#8b5cf6'},
  entertainment:{label:'Fun',icon:'🎮',color:'#ec4899'},
  health:{label:'Health',icon:'💊',color:'#10b981'},
  shopping:{label:'Shopping',icon:'🛍️',color:'#f97316'},
  income:{label:'Income',icon:'💵',color:'#22c55e'},
  other:{label:'Other',icon:'📦',color:'#6b7280'},
};
let txs = JSON.parse(localStorage.getItem('fin_txs')||'[]');
if(!txs.length) txs = [
  {id:1,name:'Salary',amount:4500,cat:'income',date:'2026-04-01'},
  {id:2,name:'Rent',amount:1200,cat:'housing',date:'2026-04-01'},
  {id:3,name:'Groceries',amount:85,cat:'food',date:'2026-04-01'},
  {id:4,name:'Netflix',amount:15,cat:'entertainment',date:'2026-04-01'},
  {id:5,name:'Bus pass',amount:60,cat:'transport',date:'2026-04-01'},
  {id:6,name:'Gym',amount:40,cat:'health',date:'2026-04-01'},
  {id:7,name:'Amazon order',amount:120,cat:'shopping',date:'2026-04-01'},
];
function save(){localStorage.setItem('fin_txs',JSON.stringify(txs));}
function fmt(n){return '$'+n.toFixed(2);}
function render(){
  const income=txs.filter(t=>t.cat==='income').reduce((a,t)=>a+t.amount,0);
  const expenses=txs.filter(t=>t.cat!=='income').reduce((a,t)=>a+t.amount,0);
  const balance=income-expenses;
  document.getElementById('stats').innerHTML=[
    {label:'Balance',val:fmt(balance),cls:balance>=0?'green':'red'},
    {label:'Total Income',val:fmt(income),cls:'green'},
    {label:'Total Expenses',val:fmt(expenses),cls:'red'},
    {label:'Transactions',val:txs.length,cls:'blue'},
  ].map(s=>`<div class="card"><h3>${s.label}</h3><div class="val ${s.cls}">${s.val}</div></div>`).join('');
  const exTxs=txs.filter(t=>t.cat!=='income');
  const catTotals={};
  exTxs.forEach(t=>{catTotals[t.cat]=(catTotals[t.cat]||0)+t.amount;});
  renderChart(catTotals);
  const list=document.getElementById('txList');
  if(!txs.length){list.innerHTML='<div class="empty">No transactions yet. Add one above!</div>';return;}
  list.innerHTML=[...txs].reverse().map(t=>{
    const c=CATS[t.cat]||CATS.other;
    const isIncome=t.cat==='income';
    return `<div class="tx">
      <div class="tx-left">
        <div class="tx-icon" style="background:${c.color}22">${c.icon}</div>
        <div><div class="tx-name">${t.name}</div><div class="tx-cat">${c.label} · ${t.date}</div></div>
      </div>
      <div class="tx-right">
        <div class="tx-amount ${isIncome?'green':'red'}">${isIncome?'+':'-'}${fmt(t.amount)}</div>
        <button class="del" onclick="delTx(${t.id})">✕</button>
      </div>
    </div>`;
  }).join('');
}
let chartInst=null;
function renderChart(catTotals){
  const labels=Object.keys(catTotals).map(k=>CATS[k]?CATS[k].label:k);
  const vals=Object.values(catTotals);
  const colors=Object.keys(catTotals).map(k=>CATS[k]?CATS[k].color:'#888');
  if(!labels.length)return;
  const canvas=document.getElementById('chart');
  const ctx=canvas.getContext('2d');
  canvas.width=canvas.offsetWidth*2;
  canvas.height=440;
  const w=canvas.width,h=canvas.height,cx=w/2,cy=h/2,r=Math.min(cx,cy)-60;
  ctx.clearRect(0,0,w,h);
  const total=vals.reduce((a,v)=>a+v,0);
  if(!total)return;
  let angle=-Math.PI/2;
  vals.forEach((v,i)=>{
    const slice=(v/total)*2*Math.PI;
    ctx.beginPath();ctx.moveTo(cx,cy);ctx.arc(cx,cy,r,angle,angle+slice);ctx.closePath();
    ctx.fillStyle=colors[i];ctx.fill();
    ctx.strokeStyle='#0a0a0f';ctx.lineWidth=3;ctx.stroke();
    const mid=angle+slice/2;
    const lx=cx+Math.cos(mid)*(r+35),ly=cy+Math.sin(mid)*(r+35);
    ctx.fillStyle='#aaa';ctx.font='bold 22px Inter';ctx.textAlign='center';ctx.textBaseline='middle';
    ctx.fillText(labels[i],lx,ly);
    const pct=Math.round(v/total*100);
    ctx.font='18px Inter';ctx.fillStyle='#666';
    ctx.fillText(pct+'%',lx,ly+26);
    angle+=slice;
  });
}
function addTx(){
  const name=document.getElementById('tname').value.trim();
  const amount=parseFloat(document.getElementById('tamount').value);
  const cat=document.getElementById('tcat').value;
  if(!name||!amount||amount<=0){alert('Enter a description and valid amount');return;}
  txs.push({id:Date.now(),name,amount,cat,date:new Date().toISOString().slice(0,10)});
  save();render();
  document.getElementById('tname').value='';
  document.getElementById('tamount').value='';
}
function delTx(id){
  txs=txs.filter(t=>t.id!==id);save();render();
}
document.getElementById('monthLabel').textContent=new Date().toLocaleDateString('en-US',{month:'long',year:'numeric'});
document.getElementById('tcat').addEventListener('change',function(){
  document.getElementById('typeLabel').textContent=this.value==='income'?'Income':'Expense';
  document.getElementById('typeLabel').style.color=this.value==='income'?'#10b981':'#ef4444';
});
render();
</script>
