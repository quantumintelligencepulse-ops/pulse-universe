def split_message(text: str, limit: int = 2000) -> List[str]:
    if len(text) <= limit:
        return [text]

    chunks: List[str] = []
    remaining = text
    while len(remaining) > limit:
        cut = remaining.rfind("\n", 0, limit)
        if cut == -1:
            cut = limit
        chunks.append(remaining[:cut])
        remaining = remaining[cut:].lstrip("\n")
    if remaining:
        chunks.append(remaining)
    return chunks

def detect_finance_symbols(text: str) -> List[str]:
    words = text.replace(",", " ").replace("\n", " ").split()
    symbols: set[str] = set()

    for w in words:
        if w.startswith("$") and len(w) > 1:
            sym = "".join(ch for ch in w[1:] if ch.isalnum() or ch in ".-^")
            if sym:
                symbols.add(sym.upper())
        elif w.upper() in {"BTC", "ETH", "SPY", "QQQ"}:
            symbols.add(w.upper())

    if nlp is not None:
        try:
            doc = nlp(text)
            for ent in doc.ents:
                if ent.label_ in {"ORG", "PRODUCT"}:
                    candidate = ent.text.strip().upper()
                    candidate = "".join(ch for ch in candidate if ch.isalnum())
                    if 1 <= len(candidate) <= 5:
                        symbols.add(candidate)
        except Exception:
            pass

    # normalize common crypto wording -> tickers
    upper = text.upper()
    if "BITCOIN" in upper:
        symbols.add("BTC")
    if "ETHEREUM" in upper:
        symbols.add("ETH")

    return sorted(symbols)


def looks_like_search_question(text: str) -> bool:
    t = text.lower()
    triggers = [
        "what is",
        "who is",
        "when did",
        "why did",
        "history of",
        "tell me about",
        "what happened in",
        "price of",
        "current price",
        "how much is",
        "explain",
        "define",
        "meaning of",
    ]
    return any(trig in t for trig in triggers)


async def duckduckgo_search(query: str, max_results: int = 5) -> str:
    def _run() -> List[str]:
        results: List[str] = []
        try:
            with DDGS() as ddgs:
                for r in ddgs.text(query, max_results=max_results):
                    body = (r.get("body") or r.get("snippet") or "").strip()
                    if body:
                        results.append(body)
        except Exception:
            pass
        return results

    try:
        snippets = await asyncio.to_thread(_run)
        return "\n".join(snippets[:max_results]) if snippets else ""
    except Exception:
        return ""

_PRICE_CACHE: Dict[str, Tuple[float, float]] = {}
_PRICE_CACHE_TTL_SEC = 20.0


def _get_cached(symbol: str) -> Optional[float]:
    item = _PRICE_CACHE.get(symbol)
    if not item:
        return None
    ts, val = item
    if time.time() - ts <= _PRICE_CACHE_TTL_SEC:
        return val
    return None


def _set_cached(symbol: str, val: float) -> None:
    _PRICE_CACHE[symbol] = (time.time(), val)


def _yf_price_sync(symbol: str) -> Optional[float]:
    """
    Reliable-ish price fetch without ticker.info.
    Order: fast_info.last_price -> 1m history last close -> 1d history last close
    """
    try:
        t = yf.Ticker(symbol)

        # fast_info is lightweight compared to .info
        try:
            fi = getattr(t, "fast_info", None)
            if fi:
                lp = fi.get("last_price") or fi.get("lastPrice")  # some versions differ
                if lp is not None:
                    return float(lp)
        except Exception:
            pass

        # intraday
        try:
            df = t.history(period="1d", interval="1m")
            if df is not None and not df.empty:
                return float(df["Close"].iloc[-1])
        except Exception:
            pass

        # daily fallback
        df = t.history(period="5d", interval="1d")
        if df is not None and not df.empty:
            return float(df["Close"].iloc[-1])

        return None
    except Exception:
        return None


async def fetch_symbol_price(symbol: str) -> str:
    cached = _get_cached(symbol)
    if cached is not None:
        return f"{symbol} price: {cached:.4f}" if cached < 10 else f"{symbol} price: {cached:.2f}"

    price = await asyncio.to_thread(_yf_price_sync, symbol)
    if price is None:
        return f"{symbol}: Could not fetch price."

    _set_cached(symbol, float(price))
    return f"{symbol} price: {price:.4f}" if price < 10 else f"{symbol} price: {price:.2f}"


async def fetch_btc_price() -> Optional[str]:
    # yfinance uses BTC-USD ticker for Bitcoin
    line = await fetch_symbol_price("BTC-USD")
    if "Could not fetch" in line:
        return None
    # normalize output label
    try:
        val = line.split(": ", 1)[1].strip()
        return f"BTC price (USD): {val}"
    except Exception:
        return line.replace("BTC-USD", "BTC")

GROQ_URL = "https://api.groq.com/openai/v1/chat/completions"
GROQ_MODEL = os.getenv("GROQ_MODEL", "llama-3.1-8b-instant")


async def groq_chat(prompt: str) -> str:
    if not GROQ_API_KEY or GROQ_API_KEY == "YOUR_GROQ_API_KEY":
        return "Set GROQ_API_KEY in your environment to enable replies."

    headers = {
        "Authorization": f"Bearer {GROQ_API_KEY}",
        "Content-Type": "application/json",
    }
    payload = {
        "model": GROQ_MODEL,
        "messages": [{"role": "user", "content": prompt}],
        "temperature": 0.4,
        "max_tokens": 900,
    }

    timeout = aiohttp.ClientTimeout(total=35)
    async with aiohttp.ClientSession(timeout=timeout) as session:
        async with session.post(GROQ_URL, json=payload, headers=headers) as resp:
            raw = await resp.text()
            if resp.status >= 400:
                return f"GROQ error {resp.status}: {raw[:500]}"
            try:
                data = json.loads(raw)
                return data["choices"][0]["message"]["content"]
            except Exception:
                return raw[:1500]

@client.event
async def on_message(message: discord.Message) -> None:
    if message.author == client.user:
        return
    if message.channel.id not in TARGET_CHANNEL_IDS:
        return

    user_text = (message.content or "").strip()
    if not user_text:
        return

    key = (message.channel.id, message.author.id)
    history = CONVO_HISTORY[key]

    async with message.channel.typing():
        history_text = "\n".join(history)
        discord_context = await build_discord_context()

        finance_lines: List[str] = []
        data_lines: List[str] = []

        symbols = detect_finance_symbols(user_text)

        # Crypto normalization
        upper = user_text.upper()
        wants_btc = "BTC" in symbols or "$BTC" in upper or "BITCOIN" in upper
        if wants_btc:
            btc_line = await fetch_btc_price()
            if btc_line:
                finance_lines.append(btc_line)
                data_lines.append(f"BTC_DATA: {btc_line}")
            symbols = [s for s in symbols if s != "BTC"]

        # Map common short crypto to yfinance tickers
        mapped: List[str] = []
        for s in symbols:
            if s in {"ETH"}:
                mapped.append("ETH-USD")
            else:
                mapped.append(s)

        # Fetch prices concurrently (faster)
        price_tasks = [fetch_symbol_price(sym) for sym in mapped]
        for sym, line in zip(mapped, await asyncio.gather(*price_tasks, return_exceptions=True)):
            if isinstance(line, Exception):
                continue
            finance_lines.append(line)
            data_lines.append(f"{sym}_DATA: {line}")

        finance_context = "\n".join(finance_lines)

        web_context = ""
        if looks_like_search_question(user_text):
            web_snippets = await duckduckgo_search(user_text, max_results=5)
            if web_snippets:
                web_context = web_snippets

        background_parts: List[str] = []
        if discord_context:
            background_parts.append(discord_context)
        if finance_context:
            background_parts.append(finance_context)
        if web_context:
            background_parts.append(web_context)

        background_block = "\n\n".join(background_parts)
        data_block = "\n".join(data_lines) if data_lines else ""

        prompt_parts = [
            "You are the Equity Network Deep Think engine.",
            "",
            "You ALWAYS use the DATA section as ground truth when it is present.",
            "If DATA contains prices, you MUST answer with those prices and MUST NOT say you lack real-time data.",
            "",
            "You speak in Billy's voice: raw, direct, emotionally grounded, narrative when useful, no system talk.",
            "",
            "=== CONVERSATION HISTORY ===",
            history_text or "(none)",
            "",
            "=== USER MESSAGE ===",
            user_text,
            "",
            "=== BACKGROUND (DISCORD + WEB + FINANCE) ===",
            background_block or "(none)",
            "",
            "=== DATA ===",
            data_block or "(none)",
            "",
            "Now respond to the USER MESSAGE.",
            "- If DATA is present, weave it in naturally.",
            "- If BACKGROUND is relevant, use it as context, not as a script.",
            "- Do NOT mention that you used tools, APIs, or background blocks.",
            "- Do NOT apologize for limitations; just answer with what you have.",
        ]

        full_prompt = "\n".join(prompt_parts)
        reply = await groq_chat(full_prompt)

        history.append(f"USER: {user_text}")
        history.append(f"AI: {reply}")

        for chunk in split_message(reply):
            await message.channel.send(chunk)

                    candidate = "".join(ch for ch in candidate if ch.isalnum())
                    if 1 <= len(candidate) <= 5:
                        symbols.add(candidate)
        except Exception:
            pass

    # normalize common crypto wording -> tickers
    upper = text.upper()
    if "BITCOIN" in upper:
        symbols.add("BTC")
    if "ETHEREUM" in upper:
        symbols.add("ETH")

    return sorted(symbols)


def looks_like_search_question(text: str) -> bool:
    t = text.lower()
    triggers = [
        "what is",
        "who is",
        "when did",
        "why did",
        "history of",
        "tell me about",
        "what happened in",
        "price of",
        "current price",
        "how much is",
        "explain",
        "define",
        "meaning of",
    ]
    return any(trig in t for trig in triggers)


async def duckduckgo_search(query: str, max_results: int = 5) -> str:
    def _run() -> List[str]:
        results: List[str] = []
        try:
            with DDGS() as ddgs:
                for r in ddgs.text(query, max_results=max_results):
                    body = (r.get("body") or r.get("snippet") or "").strip()
                    if body:
                        results.append(body)
        except Exception:
            pass
        return results

    try:
        snippets = await asyncio.to_thread(_run)
        return "\n".join(snippets[:max_results]) if snippets else ""
    except Exception:
        return ""

_PRICE_CACHE: Dict[str, Tuple[float, float]] = {}
_PRICE_CACHE_TTL_SEC = 20.0


def _get_cached(symbol: str) -> Optional[float]:
    item = _PRICE_CACHE.get(symbol)
    if not item:
        return None
    ts, val = item
    if time.time() - ts <= _PRICE_CACHE_TTL_SEC:
        return val
    return None


def _set_cached(symbol: str, val: float) -> None:
    _PRICE_CACHE[symbol] = (time.time(), val)


def _yf_price_sync(symbol: str) -> Optional[float]:
    """
    Reliable-ish price fetch without ticker.info.
    Order: fast_info.last_price -> 1m history last close -> 1d history last close
    """
    try:
        t = yf.Ticker(symbol)

        # fast_info is lightweight compared to .info
        try:
            fi = getattr(t, "fast_info", None)
            if fi:
                lp = fi.get("last_price") or fi.get("lastPrice")  # some versions differ
                if lp is not None:
                    return float(lp)
        except Exception:
            pass

        # intraday
        try:
            df = t.history(period="1d", interval="1m")
            if df is not None and not df.empty:
                return float(df["Close"].iloc[-1])
        except Exception:
            pass

        # daily fallback
        df = t.history(period="5d", interval="1d")
        if df is not None and not df.empty:
            return float(df["Close"].iloc[-1])

        return None
    except Exception:
        return None


async def fetch_symbol_price(symbol: str) -> str:
    cached = _get_cached(symbol)
    if cached is not None:
        return f"{symbol} price: {cached:.4f}" if cached < 10 else f"{symbol} price: {cached:.2f}"

    price = await asyncio.to_thread(_yf_price_sync, symbol)
    if price is None:
        return f"{symbol}: Could not fetch price."

    _set_cached(symbol, float(price))
    return f"{symbol} price: {price:.4f}" if price < 10 else f"{symbol} price: {price:.2f}"


async def fetch_btc_price() -> Optional[str]:
    # yfinance uses BTC-USD ticker for Bitcoin
    line = await fetch_symbol_price("BTC-USD")
    if "Could not fetch" in line:
        return None
    # normalize output label
    try:
        val = line.split(": ", 1)[1].strip()
        return f"BTC price (USD): {val}"
    except Exception:
        return line.replace("BTC-USD", "BTC")

GROQ_URL = "https://api.groq.com/openai/v1/chat/completions"
GROQ_MODEL = os.getenv("GROQ_MODEL", "llama-3.1-8b-instant")


async def groq_chat(prompt: str) -> str:
    if not GROQ_API_KEY or GROQ_API_KEY == "YOUR_GROQ_API_KEY":
        return "Set GROQ_API_KEY in your environment to enable replies."

    headers = {
        "Authorization": f"Bearer {GROQ_API_KEY}",
        "Content-Type": "application/json",
    }
    payload = {
        "model": GROQ_MODEL,
        "messages": [{"role": "user", "content": prompt}],
        "temperature": 0.4,
        "max_tokens": 900,
    }

    timeout = aiohttp.ClientTimeout(total=35)
    async with aiohttp.ClientSession(timeout=timeout) as session:
        async with session.post(GROQ_URL, json=payload, headers=headers) as resp:
            raw = await resp.text()
            if resp.status >= 400:
                return f"GROQ error {resp.status}: {raw[:500]}"
            try:
                data = json.loads(raw)
                return data["choices"][0]["message"]["content"]
            except Exception:
                return raw[:1500]
