const NOTE_NAMES = ['C','C#','D','Eb','E','F','F#','G','Ab','A','Bb','B'];
const SCALES = {
  'Major':[0,2,4,5,7,9,11],'Minor':[0,2,3,5,7,8,10],
  'Pentatonic':[0,2,4,7,9],'Blues':[0,3,5,6,7,10],

function midiToFreq(midi) { return 440 * Math.pow(2, (midi - 69) / 12); }
function midiNoteName(midi) { return NOTE_NAMES[midi % 12] + Math.floor(midi / 12 - 1); }
function getScaleNotes(keyStr) {
  const parts = keyStr.split(' ');
  const rootName = parts[0];
  const modeName = parts.slice(1).join(' ') || 'Minor';
  const rootIdx = NOTE_NAMES.indexOf(rootName);
  const intervals = SCALES[modeName] || SCALES['Minor'];
  return intervals.map(i => (rootIdx + i) % 12);
}
function getScaleNote(keyStr, noteIndex, octaveShift = 0) {
  const parts = keyStr.split(' ');
  const noteName = parts[0];
  const modeName = parts.slice(1).join(' ') || 'Minor';
  const rootIdx = NOTE_NAMES.indexOf(noteName);
  const rootFreq = midiToFreq(48 + rootIdx);
  const intervals = SCALES[modeName] || SCALES['Minor'];
  const oct = Math.floor(noteIndex / intervals.length) + octaveShift;
  const pos = noteIndex % intervals.length;
  return rootFreq * Math.pow(2, (intervals[pos] + oct * 12) / 12);
}

function makeDistortionCurve(amount) {
  const n = 512, curve = new Float32Array(n), k = amount * 150;
  for (let i = 0; i < n; i++) {
    const x = (i * 2) / n - 1;
    curve[i] = k > 0 ? ((Math.PI + k) * x) / (Math.PI + k * Math.abs(x)) : x;
  }
  return curve;
}
function createReverbImpulse(ctx, dur = 2.5, decay = 2.2) {
  const len = ctx.sampleRate * dur;
  const buf = ctx.createBuffer(2, len, ctx.sampleRate);
  for (let ch = 0; ch < 2; ch++) {
    const d = buf.getChannelData(ch);
    for (let i = 0; i < len; i++) d[i] = (Math.random() * 2 - 1) * Math.pow(1 - i / len, decay);
  }
  return buf;
}

  const o = ctx.createOscillator(), g = ctx.createGain();
  o.connect(g); g.connect(dest);
  o.frequency.setValueAtTime(220, t); o.frequency.exponentialRampToValueAtTime(35, t+0.1);
  g.gain.setValueAtTime(v*2.2, t); g.gain.exponentialRampToValueAtTime(0.001, t+0.55);
  o.start(t); o.stop(t+0.6);
  const cl=ctx.sampleRate*0.01, cb=ctx.createBuffer(1,cl,ctx.sampleRate), cd=cb.getChannelData(0);
  for(let i=0;i<cl;i++) cd[i]=(Math.random()*2-1)*(1-i/cl);
  const cs=ctx.createBufferSource(); cs.buffer=cb;
  const cg=ctx.createGain(); cg.gain.setValueAtTime(v*0.5,t);
  cs.connect(cg); cg.connect(dest); cs.start(t);
}

  const s=ctx.createBufferSource(); s.buffer=buf;
  const f=ctx.createBiquadFilter(); f.type='bandpass'; f.frequency.value=600+noteIdx*500; f.Q.value=3.5;
  const g=ctx.createGain(); g.gain.setValueAtTime(v*0.45,t); g.gain.exponentialRampToValueAtTime(0.001,t+0.3);

function dispatch(id, ctx, dest, t, v, freq, dur, noteIdx, params={}) {
  const vel = typeof v === 'number' && v <= 1 ? v : v / 127;
  switch(id) {
    case 'kick':      playKick(ctx, dest, t, vel); break;
    case 'snare':     playSnare(ctx, dest, t, vel); break;
    case 'hhc':       playHHC(ctx, dest, t, vel); break;
    case 'hho':       playHHO(ctx, dest, t, vel); break;
    case 'clap':      playClap(ctx, dest, t, vel); break;
    case 'tom_hi':    playTom(ctx, dest, t, vel, 170); break;
    case 'tom_lo':    playTom(ctx, dest, t, vel, 100); break;
    case 'cymbal':    playCymbal(ctx, dest, t, vel); break;
    case 'bass808':   play808Bass(ctx, dest, t, vel, freq, dur); break;
    case 'basssynth': playBassSynth(ctx, dest, t, vel, freq, dur, params); break;
    case 'lead':      playLead(ctx, dest, t, vel, freq, dur, params); break;
    case 'pad':       playPad(ctx, dest, t, vel, freq * 0.5, dur, params); break;
    case 'pluck':     playPluck(ctx, dest, t, vel, freq, params); break;
    case 'keys':      playKeys(ctx, dest, t, vel, freq, dur, params); break;
    case 'arp':       playArp(ctx, dest, t, vel, freq * 2); break;
    case 'fx':        playFX(ctx, dest, t, vel, noteIdx); break;
  }
}
function buildChannel(ctx, masterGain, reverb) {

  const comp=ctx.createDynamicsCompressor();
  comp.threshold.value=-24; comp.ratio.value=4; comp.attack.value=0.003; comp.release.value=0.25; comp.knee.value=6;

  inp.connect(mute); mute.connect(vol); vol.connect(pan);
  pan.connect(eqLo); eqLo.connect(eqMid); eqMid.connect(eqHi); eqHi.connect(filt); filt.connect(dist);
  dist.connect(chorusDry); chorusDly1.connect(chorusMix); chorusDly2.connect(chorusMix);
  dist.connect(chorusDly1); dist.connect(chorusDly2);
  chorusDry.connect(comp); chorusMix.connect(chorusWet); chorusWet.connect(comp);
  comp.connect(phaserDry); comp.connect(phaserFilters[0]);
  phaserDry.connect(masterGain);
  phaserFilters[phaserFilters.length-1].connect(phaserWet); phaserWet.connect(masterGain);
  comp.connect(rvbS); rvbS.connect(reverb);
  comp.connect(dlyS); dlyS.connect(dlyN); dlyN.connect(dlyF); dlyF.connect(dlyN); dlyN.connect(masterGain);
  return { inp, mute, vol, pan, eqLo, eqMid, eqHi, filt, dist, chorusDry, chorusWet, phaserDry, phaserWet, phaserLFOG, phaserLFO, comp, rvbS, dlyS, dlyN, dlyF };
}

function wavFromBuffer(audioBuffer) {
  const numCh=audioBuffer.numberOfChannels, len=audioBuffer.length, sr=audioBuffer.sampleRate;
  const buf=new ArrayBuffer(44+len*numCh*2), v=new DataView(buf);
  const ws=(off,s)=>{ for(let i=0;i<s.length;i++) v.setUint8(off+i,s.charCodeAt(i)); };
  ws(0,'RIFF'); v.setUint32(4,36+len*numCh*2,true); ws(8,'WAVE'); ws(12,'fmt ');
  v.setUint32(16,16,true); v.setUint16(20,1,true); v.setUint16(22,numCh,true);
  v.setUint32(24,sr,true); v.setUint32(28,sr*numCh*2,true); v.setUint16(32,numCh*2,true);
  v.setUint16(34,16,true); ws(36,'data'); v.setUint32(40,len*numCh*2,true);
  let off=44;
  for(let i=0;i<len;i++) for(let ch=0;ch<numCh;ch++) {
    const s=Math.max(-1,Math.min(1,audioBuffer.getChannelData(ch)[i]));
    v.setInt16(off,s<0?s*0x8000:s*0x7FFF,true); off+=2;
  }
  return buf;
}
async function exportWav({ patterns, mixer, bpm, steps, musicalKey, activePattern, swing=0 }) {
  const stepDur=60/bpm/4, totalDur=stepDur*steps*4+3;
  const offCtx=new OfflineAudioContext(2,Math.ceil(44100*totalDur),44100);
  const master=offCtx.createGain(); master.gain.value=mixer.__master?.volume??0.75;
  const comp=offCtx.createDynamicsCompressor(); comp.threshold.value=-12; comp.ratio.value=4;
  master.connect(comp); comp.connect(offCtx.destination);
  const reverb=offCtx.createConvolver(); reverb.buffer=createReverbImpulse(offCtx); reverb.connect(master);
  const channels={};
  INSTRUMENTS.forEach(inst => {
    const vol=offCtx.createGain(); vol.gain.value=mixer[inst.id]?.mute?0:(mixer[inst.id]?.volume??0.85);
    const pan=offCtx.createStereoPanner(); pan.pan.value=mixer[inst.id]?.pan??0;
    const rvb=offCtx.createGain(); rvb.gain.value=mixer[inst.id]?.reverb??0;
    vol.connect(pan); pan.connect(master); pan.connect(rvb); rvb.connect(reverb);
    channels[inst.id]=vol;
  });
  const patData=patterns[activePattern]||{};
  for(let loop=0;loop<4;loop++) for(let step=0;step<steps;step++) {
    const swingOff=step%2===1?swing*stepDur*0.5:0, t=(loop*steps+step)*stepDur+swingOff;
    INSTRUMENTS.forEach(inst => {
      const track=patData[inst.id]; if(!track) return;
      const s=track[step]; if(!s?.active) return;
      if(mixer[inst.id]?.mute) return;
      const freq=getScaleNote(musicalKey,s.note??0), dur=stepDur*(s.length??1)*0.95;
      dispatch(inst.id,offCtx,channels[inst.id],t,s.velocity??100,freq,dur,s.note??0,mixer[inst.id]?.synth||{});
    });
  }
  const rendered=await offCtx.startRendering();
  const wav=wavFromBuffer(rendered), blob=new Blob([wav],{type:'audio/wav'}), url=URL.createObjectURL(blob);
  const a=document.createElement('a'); a.href=url; a.download=`beatforge_${Date.now()}.wav`; a.click(); URL.revokeObjectURL(url);
}

function exportMidi({ notes, bpm=140, instrumentName='Piano Roll' }) {
  if(!notes||notes.length===0){ alert('No notes to export.'); return; }
  const ticksPerBeat=480, msPerBeat=Math.round(60000000/bpm);
  function varLen(n) {
    if(n===0) return [0];
    const bytes=[];
    while(n>0){ bytes.unshift(n&0x7F); n>>=7; }
    for(let i=0;i<bytes.length-1;i++) bytes[i]|=0x80;
    return bytes;
  }
  const events=[];
  notes.forEach(note => {
    const startTick=Math.round(note.startStep*ticksPerBeat/4);
    const endTick=startTick+Math.max(1,Math.round(note.lengthSteps*ticksPerBeat/4));
    events.push({tick:startTick,type:'on',pitch:note.pitch,velocity:note.velocity||100});
    events.push({tick:endTick,type:'off',pitch:note.pitch});
  });
  events.sort((a,b)=>a.tick-b.tick||(a.type==='off'?-1:1));
  const trackData=[];
  trackData.push(...varLen(0),0xFF,0x51,0x03,(msPerBeat>>16)&0xFF,(msPerBeat>>8)&0xFF,msPerBeat&0xFF);
  const nameBytes=[...instrumentName].map(c=>c.charCodeAt(0));
  trackData.push(...varLen(0),0xFF,0x03,...varLen(nameBytes.length),...nameBytes);
  let lastTick=0;
  events.forEach(ev => {
    const delta=ev.tick-lastTick; lastTick=ev.tick;
    trackData.push(...varLen(delta));
    trackData.push(ev.type==='on'?0x90:0x80,ev.pitch,ev.type==='on'?ev.velocity:0);
  });
  trackData.push(0,0xFF,0x2F,0);
  const header=[0x4D,0x54,0x68,0x64,0,0,0,6,0,0,0,1,(ticksPerBeat>>8)&0xFF,ticksPerBeat&0xFF];
  const tLen=trackData.length;
  const trackHeader=[0x4D,0x54,0x72,0x6B,(tLen>>24)&0xFF,(tLen>>16)&0xFF,(tLen>>8)&0xFF,tLen&0xFF];
  const bytes=new Uint8Array([...header,...trackHeader,...trackData]);
  const blob=new Blob([bytes],{type:'audio/midi'}), url=URL.createObjectURL(blob);
  const a=document.createElement('a'); a.href=url; a.download=`${instrumentName.replace(/ /g,'_')}_${Date.now()}.mid`; a.click(); URL.revokeObjectURL(url);
}

function useAudioEngine() {
  const ctxRef=useRef(null), masterRef=useRef(null), reverbRef=useRef(null);
  const chansRef=useRef({}), timerRef=useRef(null), stepRef=useRef(0), nextRef=useRef(0);
  const [isPlaying,setIsPlaying]=useState(false);
  const [currentStep,setCurrentStep]=useState(-1);
  const patternsRef=useRef({}), mixerRef=useRef({}), bpmRef=useRef(140), patternRef=useRef('A');
  const stepsRef=useRef(16), keyRef=useRef('C Minor'), swingRef=useRef(0);
  const pianoRollRef=useRef({}), arrangementRef=useRef([]), arrangeModeRef=useRef(false);
  const updateRefs=useCallback((patterns,mixer,bpm,activePattern,steps,musicalKey,masterVol,swing=0,pianoRoll={},arrangement=[],arrangeMode=false)=>{
    patternsRef.current=patterns; mixerRef.current=mixer; bpmRef.current=bpm; patternRef.current=activePattern;
    stepsRef.current=steps; keyRef.current=musicalKey; swingRef.current=swing;
    pianoRollRef.current=pianoRoll; arrangementRef.current=arrangement; arrangeModeRef.current=arrangeMode;
    if(masterRef.current) masterRef.current.gain.setTargetAtTime(masterVol??0.75,ctxRef.current.currentTime,0.02);
  },[]);
  const applyMixer=useCallback(()=>{
    const ctx=ctxRef.current; if(!ctx) return;
    const mixer=mixerRef.current, t=ctx.currentTime;
    Object.entries(mixer).forEach(([id,ch])=>{
      const c=chansRef.current[id]; if(!c) return;
      c.mute.gain.setTargetAtTime(ch.mute?0:1,t,0.01);
      c.vol.gain.setTargetAtTime(ch.mute?0:(ch.volume??0.85),t,0.01);
      c.pan.pan.setTargetAtTime(ch.pan??0,t,0.01);
      c.filt.frequency.setTargetAtTime(ch.filterCutoff??20000,t,0.01);
      c.filt.Q.value=ch.filterRes??1; c.filt.type=ch.filterType??'lowpass';
      c.rvbS.gain.setTargetAtTime(ch.reverb??0,t,0.01);
      c.dlyS.gain.setTargetAtTime(ch.delay??0,t,0.01);
      c.dlyF.gain.value=ch.delayFeedback??0.35;
      c.dist.curve=makeDistortionCurve(ch.distortion??0);
      c.eqLo.gain.setTargetAtTime(ch.eqLo??0,t,0.01);
      c.eqMid.gain.setTargetAtTime(ch.eqMid??0,t,0.01);
      c.eqHi.gain.setTargetAtTime(ch.eqHi??0,t,0.01);
      if(ch.eqMidFreq) c.eqMid.frequency.setTargetAtTime(ch.eqMidFreq,t,0.01);
      c.chorusWet.gain.setTargetAtTime(ch.chorus??0,t,0.01);
      c.chorusDry.gain.setTargetAtTime(1-(ch.chorus??0)*0.5,t,0.01);
      c.phaserWet.gain.setTargetAtTime(ch.phaser??0,t,0.01);
      c.phaserLFOG.gain.setTargetAtTime((ch.phaser??0)*800,t,0.01);
      if(ch.compThreshold!==undefined) c.comp.threshold.setTargetAtTime(ch.compThreshold??-24,t,0.01);
      if(ch.compRatio!==undefined) c.comp.ratio.setTargetAtTime(ch.compRatio??4,t,0.01);
    });
  },[]);
  const getPatternForStep=useCallback((globalStep)=>{
    if(!arrangeModeRef.current||!arrangementRef.current.length) return patternRef.current;
    const bar=Math.floor(globalStep/stepsRef.current);
    const sorted=[...arrangementRef.current].sort((a,b)=>a.bar-b.bar);
    let activeP=sorted[0]?.patternId||patternRef.current;
    for(const block of sorted){ if(block.bar<=bar) activeP=block.patternId; else break; }
    return activeP;
  },[]);
  const scheduleStep=useCallback((step,time,globalStep)=>{
    const ctx=ctxRef.current; if(!ctx) return;
    const swing=swingRef.current;
    const swingOffset=step%2===1?swing*(60/bpmRef.current/4)*0.5:0;
    const t=time+swingOffset;
    const currentPattern=arrangeModeRef.current?getPatternForStep(globalStep):patternRef.current;
    const patData=patternsRef.current[currentPattern]; if(!patData) return;
    const stepDur=60/bpmRef.current/4;
    INSTRUMENTS.forEach(inst=>{
      const mixer=mixerRef.current[inst.id]; if(mixer?.mute) return;
      const ch=chansRef.current[inst.id]; if(!ch) return;
      const params=mixer?.synth||{};
      const prNotes=pianoRollRef.current?.[currentPattern]?.[inst.id];
      if(prNotes?.length){
        prNotes.forEach(note=>{
          if(note.startStep===step){
            const freq=midiToFreq(note.pitch), dur=stepDur*note.lengthSteps*0.95;
            dispatch(inst.id,ctx,ch.inp,t+(Math.random()-0.5)*0.002,note.velocity??100,freq,dur,0,params);
          }
        });
        return;
      }
      const track=patData[inst.id]; if(!track) return;
      const s=track[step]; if(!s?.active) return;
      const freq=getScaleNote(keyRef.current,s.note??0), dur=stepDur*(s.length??1)*0.95;
      dispatch(inst.id,ctx,ch.inp,t+(Math.random()-0.5)*0.003,s.velocity??100,freq,dur,s.note??0,params);
    });
    applyMixer();
  },[applyMixer,getPatternForStep]);
  const globalStepRef=useRef(0);
  const scheduler=useCallback(()=>{
    const ctx=ctxRef.current; if(!ctx) return;
    const stepDur=60/bpmRef.current/4, totalSteps=stepsRef.current;
    while(nextRef.current<ctx.currentTime+0.12){
      const step=stepRef.current%totalSteps;
      scheduleStep(step,nextRef.current,globalStepRef.current);
      setCurrentStep(step);
      stepRef.current=(step+1)%totalSteps;
      globalStepRef.current++;
      nextRef.current+=stepDur;
    }
  },[scheduleStep]);
  const init=useCallback(()=>{
    if(ctxRef.current) return;
    const ctx=new (window.AudioContext||window.webkitAudioContext)();
    ctxRef.current=ctx;
    const master=ctx.createGain(); master.gain.value=0.75; masterRef.current=master;
    const comp=ctx.createDynamicsCompressor();
    comp.threshold.value=-12; comp.knee.value=6; comp.ratio.value=4; comp.attack.value=0.003; comp.release.value=0.25;
    master.connect(comp); comp.connect(ctx.destination);
    const reverb=ctx.createConvolver(); reverb.buffer=createReverbImpulse(ctx);
    reverb.connect(master); reverbRef.current=reverb;
    INSTRUMENTS.forEach(inst=>{ chansRef.current[inst.id]=buildChannel(ctx,master,reverb); });
  },[]);
  const play=useCallback(()=>{
    init();
    const ctx=ctxRef.current;
    if(ctx.state==='suspended') ctx.resume();
    stepRef.current=0; globalStepRef.current=0; nextRef.current=ctx.currentTime+0.05;
    timerRef.current=setInterval(scheduler,25);
    setIsPlaying(true);
  },[init,scheduler]);
  const stop=useCallback(()=>{
    clearInterval(timerRef.current);
    setIsPlaying(false); setCurrentStep(-1); stepRef.current=0; globalStepRef.current=0;
  },[]);
  useEffect(()=>()=>{ clearInterval(timerRef.current); ctxRef.current?.close(); },[]);
  return { isPlaying, currentStep, play, stop, updateRefs };
}

function StepSequencer({ patterns,activePattern,currentStep,steps,onToggleStep,onSetNote,onSetVelocity,selectedChannel,onSelectChannel }) {
  const scrollRef=useRef(null);
  const patData=patterns[activePattern]||{};
  const groups=Object.entries(INSTRUMENTS.reduce((acc,inst)=>{ if(!acc[inst.type])acc[inst.type]=[]; acc[inst.type].push(inst); return acc; },{}));
  return (
    <div className="flex-1 overflow-y-auto bg-[#0a0a0f]">
      <div ref={scrollRef} className="overflow-x-auto min-w-0">
        <div className="min-w-max p-3 space-y-1">
          {groups.map(([type,insts])=>(
            <div key={type}>
              <div className="text-[9px] font-black tracking-widest text-white/20 uppercase px-2 py-1">{TYPE_LABELS[type]}</div>
              {insts.map(inst=>{
                const track=patData[inst.id]||[], isSelected=selectedChannel===inst.id;
                return (
                  <div key={inst.id} className={cn("flex items-center gap-1 rounded-lg px-1 py-0.5 transition-colors cursor-pointer",isSelected?"bg-white/5":"hover:bg-white/3")} onClick={()=>onSelectChannel(isSelected?null:inst.id)}>
                    <div className="w-20 flex-shrink-0 flex items-center gap-1.5 px-1">
                      <div className="w-1.5 h-1.5 rounded-full flex-shrink-0" style={{backgroundColor:inst.color}}/>
                      <span className="text-[10px] font-semibold truncate" style={{color:inst.color}}>{inst.name}</span>
                    </div>
                    <div className="flex items-center">
                      {Array(steps).fill(0).map((_,si)=>{
                        const s=track[si]||{active:false,velocity:100,note:0};
                        const isActive=s.active, isCurrent=currentStep===si, showBar=si>0&&si%4===0;
                        const velHex=Math.round(((s.velocity??100)/127)*215+40).toString(16).padStart(2,'0');
                        return (
                          <div key={si} className="flex items-center">
                            {showBar&&<div className="w-px h-5 bg-white/8 mx-0.5"/>}
                            <button
                              onClick={e=>{e.stopPropagation();onToggleStep(inst.id,si);}}
                              onContextMenu={e=>{
                                e.preventDefault(); e.stopPropagation();
                                if(isActive){
                                  if(inst.type==='drum'){const cur=s.velocity??100;const idx=VELOCITY_CYCLE.findIndex(v=>v>=cur);onSetVelocity(inst.id,si,VELOCITY_CYCLE[(idx+1)%VELOCITY_CYCLE.length]);}
                                  else{onSetNote(inst.id,si,((s.note||0)+1)%8);}
                                }
                              }}
                              className="w-7 h-7 rounded-sm transition-all duration-75 flex-shrink-0 relative overflow-hidden border border-transparent"
                              style={{backgroundColor:isActive?isCurrent?inst.color:`${inst.color}${velHex}`:isCurrent?`${inst.color}20`:'rgba(255,255,255,0.04)',boxShadow:isActive&&isCurrent?`0 0 10px ${inst.color}80`:'none',borderColor:isCurrent?`${inst.color}60`:'transparent',transform:isActive&&isCurrent?'scale(1.08)':'scale(1)'}}
                            >
                              {isActive&&<div className="absolute inset-0 opacity-20" style={{background:'linear-gradient(to bottom, white 0%, transparent 100%)'}}/>}
                              {isActive&&inst.type==='drum'&&<div className="absolute bottom-0 left-0 right-0 h-0.5" style={{backgroundColor:`rgba(255,255,255,${((s.velocity??100)/127)*0.6})`}}/>}
                            </button>
                          </div>
                        );
                      })}
                    </div>
                    <div className="w-10 flex-shrink-0 text-[9px] text-white/25 text-center ml-1">
                      {inst.type==='drum'?(()=>{const a=track.find(s=>s?.active);return a?`v${a.velocity??100}`:'';})():(()=>{const a=track.find(s=>s?.active);return a?SEQ_NOTES[a.note||0]:'';})()}
                    </div>
                  </div>
                );
              })}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}

//  PIANO ROLL COMPONENT
// ─────────────────────────────────────────────────────────────
const NOTE_HEIGHT=14, STEP_WIDTH=32, MIN_MIDI=36, MAX_MIDI=96, TOTAL_NOTES=MAX_MIDI-MIN_MIDI;
const BLACK_NOTES=[1,3,6,8,10];
function isBlack(midi){ return BLACK_NOTES.includes(midi%12); }

function PianoRoll({ notes,onNotesChange,steps,musicalKey,activePattern,activeInstrument,onInstrumentChange,bpm }) {
  const gridRef=useRef(null);
  const [scaleLock,setScaleLock]=useState(false);
  const [quantize,setQuantize]=useState(1);
  const inst=INSTRUMENTS.find(i=>i.id===activeInstrument)||INSTRUMENTS[10];
  const scaleNotes=getScaleNotes(musicalKey);
  const notesForInst=notes||[];
  const snapToScale=useCallback((pitch)=>{
    const noteInOctave=pitch%12;
    if(scaleNotes.includes(noteInOctave)) return pitch;
    let bestNote=scaleNotes[0]??0, bestDist=13;
    scaleNotes.forEach(sn=>{ const dist=Math.min(Math.abs(noteInOctave-sn),12-Math.abs(noteInOctave-sn)); if(dist<bestDist){bestDist=dist;bestNote=sn;} });
    const octaveBase=Math.floor(pitch/12)*12;
    let finalPitch=octaveBase+bestNote;
    if(finalPitch<MIN_MIDI) finalPitch+=12;
    if(finalPitch>MAX_MIDI) finalPitch-=12;
    return finalPitch;
  },[scaleNotes]);
  const addNote=useCallback((pitch,startStep)=>{
    const finalPitch=scaleLock?snapToScale(pitch):pitch;
    const snapStep=Math.floor(startStep/quantize)*quantize;
    const existing=notesForInst.find(n=>n.pitch===finalPitch&&n.startStep===snapStep);
    if(existing){ onNotesChange(notesForInst.filter(n=>n!==existing)); return; }
    onNotesChange([...notesForInst,{id:`${Date.now()}_${Math.random()}`,pitch:finalPitch,startStep:snapStep,lengthSteps:quantize,velocity:100}]);
  },[notesForInst,onNotesChange,scaleLock,snapToScale,quantize]);
  const removeNote=useCallback((id)=>onNotesChange(notesForInst.filter(n=>n.id!==id)),[notesForInst,onNotesChange]);
  const resizeNote=useCallback((id,newLen)=>onNotesChange(notesForInst.map(n=>n.id===id?{...n,lengthSteps:Math.max(quantize,newLen)}:n)),[notesForInst,onNotesChange,quantize]);
  const gridH=TOTAL_NOTES*NOTE_HEIGHT, gridW=steps*STEP_WIDTH;
  return (
    <div className="flex flex-col h-full bg-[#080810]">
      <div className="flex items-center gap-2 px-4 py-2 border-b border-white/5 bg-[#0d0d12] flex-wrap">
        <span className="text-[10px] text-white/30 uppercase tracking-widest font-bold">Piano Roll</span>
        <Select value={activeInstrument} onValueChange={onInstrumentChange}>
          <SelectTrigger className="h-7 w-32 bg-white/5 border-white/10 text-xs" style={{color:inst.color}}><SelectValue/></SelectTrigger>
          <SelectContent>{INSTRUMENTS.filter(i=>i.type!=='drum'&&i.type!=='fx').map(i=><SelectItem key={i.id} value={i.id} style={{color:i.color}}>{i.name}</SelectItem>)}</SelectContent>
        </Select>
        <div className="flex items-center gap-1">
          <span className="text-[9px] text-white/30 uppercase">Q:</span>
          {[1,2,4].map(q=>(
            <button key={q} onClick={()=>setQuantize(q)} className={cn("h-6 px-1.5 rounded text-[9px] font-bold border transition-all",quantize===q?"bg-primary/20 border-primary/40 text-primary":"border-white/10 text-white/30 hover:text-white/60")}>
              {q===1?'1/16':q===2?'1/8':'1/4'}
            </button>
          ))}
        </div>
        <div className="flex items-center gap-1 ml-auto">
          <button onClick={()=>setScaleLock(v=>!v)} className={cn("h-6 px-2.5 rounded text-[9px] font-bold uppercase tracking-wider border transition-all",scaleLock?"bg-green-500/20 border-green-500/40 text-green-400":"border-white/10 text-white/30 hover:text-white/60")}>
            {scaleLock?'🔒 Scale Lock ON':'Scale Lock'}
          </button>
          <button onClick={()=>exportMidi({notes:notesForInst,bpm:bpm||140,instrumentName:inst.name})} disabled={notesForInst.length===0} className="h-6 px-2.5 rounded text-[9px] font-bold uppercase tracking-wider border border-white/10 text-white/30 hover:text-cyan-400 hover:border-cyan-400/30 transition-all disabled:opacity-30">MIDI Export</button>
          <button onClick={()=>onNotesChange([])} className="h-6 px-2.5 rounded text-[9px] font-bold uppercase tracking-wider text-white/20 hover:text-red-400 transition-colors">Clear</button>
        </div>
      </div>
      <div className="flex flex-1 overflow-auto">
        <div className="flex-shrink-0 w-12 relative" style={{height:gridH}}>
          {Array.from({length:TOTAL_NOTES},(_,i)=>{
            const midi=MAX_MIDI-1-i, black=isBlack(midi), inScale=scaleNotes.includes(midi%12), isC=midi%12===0;
            return (
              <div key={midi} className={cn("absolute flex items-center justify-end pr-1 border-b select-none",black?"bg-[#1a1a22] text-white/30 border-white/5 z-10 w-8":"bg-[#111118] text-white/20 border-white/5 w-12",inScale&&!black&&"bg-[#0f1020]",scaleLock&&inScale&&"border-l border-green-500/20")} style={{top:i*NOTE_HEIGHT,height:NOTE_HEIGHT}}>
                {isC&&<span className="text-[8px] font-mono">{midiNoteName(midi)}</span>}
              </div>
            );
          })}
        </div>
        <div ref={gridRef} className="flex-1 overflow-auto relative" style={{height:gridH+24}}>
          <div className="sticky top-0 flex z-20 bg-[#0a0a10] border-b border-white/5" style={{width:gridW}}>
            {Array.from({length:steps},(_,i)=>(
              <div key={i} className={cn("border-r border-white/5 text-[8px] text-white/20 text-center flex-shrink-0",i%4===0&&"border-r-white/15 text-white/40")} style={{width:STEP_WIDTH,height:20,lineHeight:'20px'}}>
                {i%4===0&&Math.floor(i/4)+1}
              </div>
            ))}
          </div>
          <div className="relative" style={{width:gridW,height:gridH}}
            onMouseDown={e=>{
              if(e.button===2) return;
              const rect=gridRef.current.getBoundingClientRect();
              const y=e.clientY-rect.top-20+gridRef.current.scrollTop;
              const x=e.clientX-rect.left+gridRef.current.scrollLeft-48;
              const noteRow=Math.floor(y/NOTE_HEIGHT), stepCol=Math.floor(x/STEP_WIDTH);
              const midi=MAX_MIDI-1-noteRow;
              if(midi>=MIN_MIDI&&midi<=MAX_MIDI&&stepCol>=0&&stepCol<steps) addNote(midi,stepCol);
            }}
            onContextMenu={e=>e.preventDefault()}
          >
            {Array.from({length:TOTAL_NOTES},(_,i)=>{
              const midi=MAX_MIDI-1-i, black=isBlack(midi), inScale=scaleNotes.includes(midi%12);
              return <div key={i} className="absolute w-full border-b border-white/3" style={{top:i*NOTE_HEIGHT,height:NOTE_HEIGHT,backgroundColor:black?'#0c0c14':scaleLock&&inScale?'#0d1018':inScale?'#0d0d18':'#0a0a10'}}/>;
            })}
            {Array.from({length:steps},(_,i)=>(
              <div key={i} className="absolute top-0 bottom-0 border-r" style={{left:i*STEP_WIDTH,borderColor:i%4===0?'rgba(255,255,255,0.1)':'rgba(255,255,255,0.04)'}}/>
            ))}
            {notesForInst.map(note=>{
              const rowIdx=MAX_MIDI-1-note.pitch;
              const velAlpha=Math.round(((note.velocity||100)/127)*200+55).toString(16).padStart(2,'0');
              return (
                <div key={note.id} className="absolute rounded-sm flex items-center overflow-hidden cursor-pointer group"
                  style={{top:rowIdx*NOTE_HEIGHT+1,left:note.startStep*STEP_WIDTH+1,width:note.lengthSteps*STEP_WIDTH-2,height:NOTE_HEIGHT-2,backgroundColor:inst.color+velAlpha,boxShadow:`0 0 6px ${inst.color}40`}}
                  onContextMenu={e=>{e.preventDefault();removeNote(note.id);}}>
                  <div className="flex-1 px-1 text-[8px] font-bold text-black/70 truncate">{midiNoteName(note.pitch)}</div>
                  <div className="w-2 h-full flex-shrink-0 cursor-ew-resize bg-black/20 hover:bg-black/50"
                    onMouseDown={e=>{
                      e.stopPropagation();
                      const startX=e.clientX, origLen=note.lengthSteps;
                      const onMove=ev=>resizeNote(note.id,Math.max(1,origLen+Math.round((ev.clientX-startX)/STEP_WIDTH)));
                      const onUp=()=>{ document.removeEventListener('mousemove',onMove); document.removeEventListener('mouseup',onUp); };
                      document.addEventListener('mousemove',onMove); document.addEventListener('mouseup',onUp);
                    }}
                  />
                </div>
              );
            })}
          </div>
        </div>
      </div>
    </div>
  );
}

const PATTERN_COLORS = { A:'#8b5cf6',B:'#06b6d4',C:'#f97316',D:'#22c55e',E:'#f43f5e',F:'#eab308',G:'#14b8a6',H:'#ec4899' };
const TOTAL_BARS=64, BAR_WIDTH=48;

function SongArranger({ arrangement,onUpdate,activePattern,isPlaying,currentBar }) {
  const [selectedPattern,setSelectedPattern]=useState('A');
  const placeBlock=useCallback((bar)=>{
    const existing=arrangement.find(b=>b.bar===bar);
    if(existing){ onUpdate(arrangement.filter(b=>b!==existing)); }
    else{ onUpdate([...arrangement,{id:`${Date.now()}`,patternId:selectedPattern,bar}].sort((a,b)=>a.bar-b.bar)); }
  },[arrangement,onUpdate,selectedPattern]);
  const removeBlock=useCallback((id)=>onUpdate(arrangement.filter(b=>b.id!==id)),[arrangement,onUpdate]);
  return (
    <div className="flex flex-col h-full bg-[#080810]">
      <div className="flex items-center gap-3 px-4 py-2 border-b border-white/5 bg-[#0d0d12]">
        <span className="text-[10px] text-white/30 uppercase tracking-widest font-bold">Song Arranger</span>
        <div className="flex gap-1">
          {ALL_PATTERNS.map(p=>(
            <button key={p} onClick={()=>setSelectedPattern(p)} className={cn("w-8 h-7 rounded-md text-xs font-bold transition-all border",selectedPattern===p?"border-white/30 text-white":"border-white/10 text-white/40 hover:border-white/20")} style={selectedPattern===p?{backgroundColor:PATTERN_COLORS[p]+'33',borderColor:PATTERN_COLORS[p]+'80',color:PATTERN_COLORS[p]}:{}}>{p}</button>
          ))}
        </div>
        <span className="text-[10px] text-white/20">Click bar to place/remove selected pattern</span>
        <button onClick={()=>onUpdate([])} className="ml-auto flex items-center gap-1 text-[10px] text-white/30 hover:text-red-400"><Trash2 className="w-3 h-3"/> Clear</button>
      </div>
      <div className="flex-1 overflow-auto p-4">
        <div className="flex mb-1" style={{paddingLeft:80}}>
          {Array.from({length:TOTAL_BARS},(_,i)=>(
            <div key={i} className="flex-shrink-0 text-[9px] text-white/20 text-center border-r border-white/5" style={{width:BAR_WIDTH}}>{i+1}</div>
          ))}
        </div>
        {ALL_PATTERNS.map(p=>{
          const blocksForPattern=arrangement.filter(b=>b.patternId===p);
          return (
            <div key={p} className="flex items-center mb-1 h-10">
              <div className="w-20 flex-shrink-0 text-xs font-bold flex items-center gap-1.5 pr-2">
                <div className="w-2 h-2 rounded-sm" style={{backgroundColor:PATTERN_COLORS[p]}}/>
                <span style={{color:PATTERN_COLORS[p]}}>Pattern {p}</span>
              </div>
              <div className="flex relative h-full" style={{width:TOTAL_BARS*BAR_WIDTH}}>
                {Array.from({length:TOTAL_BARS},(_,i)=>(
                  <div key={i} className={cn("flex-shrink-0 h-full border-r border-white/5 cursor-pointer transition-colors",selectedPattern===p?"hover:bg-white/5":"")} style={{width:BAR_WIDTH}} onClick={()=>selectedPattern===p&&placeBlock(i)}/>
                ))}
                {blocksForPattern.map(block=>(
                  <div key={block.id} className="absolute top-0.5 bottom-0.5 rounded flex items-center justify-between px-2 group cursor-pointer"
                    style={{left:block.bar*BAR_WIDTH+1,width:BAR_WIDTH-2,backgroundColor:PATTERN_COLORS[p]+'40',borderLeft:`3px solid ${PATTERN_COLORS[p]}`,boxShadow:currentBar===block.bar&&isPlaying?`0 0 12px ${PATTERN_COLORS[p]}80`:'none'}}
                    onClick={()=>removeBlock(block.id)}>
                    <span className="text-[9px] font-bold" style={{color:PATTERN_COLORS[p]}}>{p}</span>
                    <Trash2 className="w-2.5 h-2.5 opacity-0 group-hover:opacity-60 transition-opacity" style={{color:PATTERN_COLORS[p]}}/>
                  </div>
                ))}
                {isPlaying&&<div className="absolute top-0 bottom-0 w-0.5 bg-white/40 pointer-events-none z-10" style={{left:(currentBar||0)*BAR_WIDTH}}/>}
              </div>
            </div>
          );
        })}
        {arrangement.length>0&&(
          <div className="mt-4 p-3 rounded-xl bg-white/3 border border-white/5 text-[10px] text-white/30">
            {arrangement.length} block{arrangement.length!==1?'s':''} placed · {Math.max(...arrangement.map(b=>b.bar))+1} bars total · Patterns: {[...new Set(arrangement.map(b=>b.patternId))].join(', ')}
          </div>
        )}
      </div>
    </div>
  );
}

const AI_GENRES=['trap','hip_hop','drill','lofi','rnb','electronic','pop','afrobeat','jazz','reggaeton','rock','synthwave','classical','ambient','drum_and_bass','house'];
const ARTISTS=['Kanye West','Metro Boomin','J Dilla','Flying Lotus','Daft Punk','Justice','Four Tet','Nujabes','Dr. Dre','9th Wonder','Madlib','Timbaland'];

function AIPanel({ patterns,mixer,bpm,musicalKey,genre,mood,steps,activePattern,onPatternsChange,onMixerChange,onArrangementChange }) {
  const [activeTab,setActiveTab]=useState('mix');
  const [loading,setLoading]=useState(false);
  const [chatMessages,setChatMessages]=useState([{role:'assistant',content:`Hey! I'm your AI beat collaborator. Tell me what you want — "make the bass heavier", "add more hi-hats", "make it sound darker", whatever you want!`}]);
  const [chatInput,setChatInput]=useState('');
  const [styleInput,setStyleInput]=useState('');
  const [selectedArtist,setSelectedArtist]=useState('');
  const [morphGenre1,setMorphGenre1]=useState('trap');
  const [morphGenre2,setMorphGenre2]=useState('jazz');
  const [morphBlend,setMorphBlend]=useState(50);
  const chatEndRef=useRef(null);
  const normalizePattern=(raw)=>{
    const normalized={};
    INSTRUMENTS.forEach(inst=>{
      const r=raw[inst.id];
      normalized[inst.id]=Array.isArray(r)&&r.length===steps?r.map(s=>({active:!!s.active,velocity:Math.min(127,Math.max(1,s.velocity??100)),note:Math.min(7,Math.max(0,s.note??0)),length:Math.min(4,Math.max(1,s.length??1))})):emptyTrack(steps);
    });
    return normalized;
  };
  const withLoading=async(fn)=>{ setLoading(true); try{await fn();}catch(e){toast.error("AI error: "+e.message);}finally{setLoading(false);} };
  const handleAIMix=()=>withLoading(async()=>{
    const patternSummary=Object.entries(patterns[activePattern]||{}).map(([id,track])=>`${id}: ${track.filter(s=>s?.active).length} active steps`).join(', ');
    const res=await base44.integrations.Core.InvokeLLM({
      prompt:`You are a world-class mixing engineer. Analyze this beat and set optimal mixer/FX settings.\nGenre: ${genre} | Mood: ${mood} | BPM: ${bpm} | Key: ${musicalKey}\nPattern density: ${patternSummary}\nSet professional mix settings. Consider gain staging, panning, EQ, reverb/delay, compression, genre-appropriate FX.`,
      response_json_schema:{type:"object",properties:{mixer:{type:"object",additionalProperties:{type:"object",properties:{volume:{type:"number"},pan:{type:"number"},eqLo:{type:"number"},eqMid:{type:"number"},eqHi:{type:"number"},reverb:{type:"number"},delay:{type:"number"},distortion:{type:"number"},chorus:{type:"number"},phaser:{type:"number"},compThreshold:{type:"number"},compRatio:{type:"number"}}}},notes:{type:"string"}}}
    });
    if(res.mixer){
      const updated={...mixer};
      Object.entries(res.mixer).forEach(([id,settings])=>{ if(updated[id]) updated[id]={...updated[id],...settings}; });
      onMixerChange(updated);
      toast.success(`AI Mix applied! ${res.notes?res.notes.slice(0,80):''}`);
    }
  });
  const handleStyleTransfer=()=>withLoading(async()=>{
    const target=styleInput||selectedArtist;
    if(!target){toast.error("Enter a style or artist first");return;}
    const res=await base44.integrations.Core.InvokeLLM({
      prompt:`You are a legendary music producer. Transform this beat to sound like "${target}". BPM: ${bpm}, Key: ${musicalKey}, Steps: ${steps}. Available: ${INSTRUMENTS.map(i=>`${i.id} (${i.name})`).join(', ')}`,
      response_json_schema:{type:"object",properties:{pattern:{type:"object",additionalProperties:{type:"array",items:{type:"object",properties:{active:{type:"boolean"},velocity:{type:"number"},note:{type:"number"},length:{type:"number"}}}}},description:{type:"string"}}}
    });
    if(res.pattern){onPatternsChange(prev=>({...prev,[activePattern]:normalizePattern(res.pattern)}));toast.success(`Style transferred to "${target}"!`);}
  });
  const handleGenreMorph=()=>withLoading(async()=>{
    const res=await base44.integrations.Core.InvokeLLM({
      prompt:`Create a beat blending ${morphGenre1} (${morphBlend}%) and ${morphGenre2} (${100-morphBlend}%). BPM: ${bpm}, Key: ${musicalKey}, Steps: ${steps}. Available: ${INSTRUMENTS.map(i=>`${i.id} (${i.name})`).join(', ')}`,
      response_json_schema:{type:"object",properties:{pattern:{type:"object",additionalProperties:{type:"array",items:{type:"object",properties:{active:{type:"boolean"},velocity:{type:"number"},note:{type:"number"},length:{type:"number"}}}}},description:{type:"string"}}}
    });
    if(res.pattern){onPatternsChange(prev=>({...prev,[activePattern]:normalizePattern(res.pattern)}));toast.success('Genre morph complete!');}
  });
  const handleAIArranger=()=>withLoading(async()=>{
    const res=await base44.integrations.Core.InvokeLLM({
      prompt:`Create a song arrangement using patterns A,B,C,D. Genre: ${genre}, Mood: ${mood}, BPM: ${bpm}. Design 16-32 bars with intro/buildup/drop/outro. A=verse, B=build, C=drop, D=bridge. Return array of {patternId, bar}.`,
      response_json_schema:{type:"object",properties:{arrangement:{type:"array",items:{type:"object",properties:{patternId:{type:"string"},bar:{type:"number"}}}},structure:{type:"string"}}}
    });
    if(res.arrangement?.length){ onArrangementChange(res.arrangement.map((b,i)=>({id:`ai_${i}`,patternId:b.patternId,bar:b.bar}))); toast.success(`Song arranged! ${res.structure?.slice(0,80)||''}`); }
  });
  const handleChat=async()=>{
    if(!chatInput.trim()) return;
    const userMsg=chatInput.trim(); setChatInput('');
    setChatMessages(prev=>[...prev,{role:'user',content:userMsg}]);
    setLoading(true);
    try {
      const patternSummary=JSON.stringify(Object.fromEntries(Object.entries(patterns[activePattern]||{}).map(([id,track])=>[id,track.filter(s=>s?.active).map(s=>({step:track.indexOf(s),note:s.note,vel:s.velocity}))])));
      const res=await base44.integrations.Core.InvokeLLM({
        prompt:`You are an AI music collaborator. Producer said: "${userMsg}"\nBeat: Genre: ${genre}, BPM: ${bpm}, Key: ${musicalKey}, Steps: ${steps}\nActive steps: ${patternSummary}\nRespond naturally. If they want beat changes, provide updated pattern data. Keep response brief.`,
        response_json_schema:{type:"object",properties:{response:{type:"string"},pattern:{type:"object",additionalProperties:{type:"array",items:{type:"object",properties:{active:{type:"boolean"},velocity:{type:"number"},note:{type:"number"},length:{type:"number"}}}}}}}
      });
      setChatMessages(prev=>[...prev,{role:'assistant',content:res.response||"Let me work on that..."}]);
      if(res.pattern&&Object.keys(res.pattern).length>0){
        const normalized={};
        INSTRUMENTS.forEach(inst=>{ const r=res.pattern[inst.id]; if(Array.isArray(r)&&r.length===steps) normalized[inst.id]=r.map(s=>({active:!!s.active,velocity:Math.min(127,Math.max(1,s.velocity??100)),note:Math.min(7,Math.max(0,s.note??0)),length:Math.min(4,Math.max(1,s.length??1))})); });
        if(Object.keys(normalized).length>0){onPatternsChange(prev=>({...prev,[activePattern]:{...patterns[activePattern],...normalized}}));toast.success("Beat updated by AI!");}
      }
    } catch(e) {
      setChatMessages(prev=>[...prev,{role:'assistant',content:"Hmm, ran into an issue. Try again!"}]);
    } finally { setLoading(false); setTimeout(()=>chatEndRef.current?.scrollIntoView({behavior:'smooth'}),100); }
  };
  const tabs=[{id:'mix',label:'Mix Engineer',icon:Sliders},{id:'style',label:'Style Transfer',icon:Wand2},{id:'morph',label:'Genre Morph',icon:Shuffle},{id:'arranger',label:'AI Arranger',icon:Music},{id:'chat',label:'AI Chat',icon:MessageSquare}];
  return (
    <div className="flex flex-col h-full bg-[#080810]">
      <div className="flex gap-1 px-4 py-2 border-b border-white/5 bg-[#0d0d12] overflow-x-auto">
        {tabs.map(tab=>{ const Icon=tab.icon; return (
          <button key={tab.id} onClick={()=>setActiveTab(tab.id)} className={cn("flex items-center gap-1.5 px-3 py-1.5 rounded-lg text-xs font-medium transition-all flex-shrink-0",activeTab===tab.id?"bg-primary/20 text-primary border border-primary/30":"text-white/40 hover:text-white/70 border border-transparent")}>
            <Icon className="w-3 h-3"/> {tab.label}
          </button>
        ); })}
      </div>
      <div className="flex-1 overflow-auto p-4">
        {activeTab==='mix'&&(
          <div className="max-w-lg space-y-4">
            <div className="p-4 rounded-xl bg-white/3 border border-white/5 space-y-2">
              <div className="text-sm font-bold text-white/80">AI Mix Engineer</div>
              <div className="text-xs text-white/40 leading-relaxed">AI analyzes your beat and automatically sets optimal volume, pan, EQ, compression, reverb, delay, and FX for every channel.</div>
            </div>
            <button onClick={handleAIMix} disabled={loading} className="flex items-center gap-2 px-5 py-3 rounded-xl bg-primary/20 border border-primary/30 text-primary font-bold text-sm hover:bg-primary/30 transition-all disabled:opacity-50">
              {loading?<Loader2 className="w-4 h-4 animate-spin"/>:<Sliders className="w-4 h-4"/>} Auto-Mix This Beat
            </button>
          </div>
        )}
        {activeTab==='style'&&(
          <div className="max-w-lg space-y-4">
            <div className="p-4 rounded-xl bg-white/3 border border-white/5"><div className="text-sm font-bold text-white/80">AI Style Transfer</div><div className="text-xs text-white/40 mt-1">Transform your beat to sound like any artist or genre.</div></div>
            <Select value={selectedArtist} onValueChange={setSelectedArtist}>
              <SelectTrigger className="bg-white/5 border-white/10 text-white/70 text-sm"><SelectValue placeholder="Choose an artist..."/></SelectTrigger>
              <SelectContent>{ARTISTS.map(a=><SelectItem key={a} value={a}>{a}</SelectItem>)}</SelectContent>
            </Select>
            <div className="text-[10px] text-white/20 text-center">OR</div>
            <input className="w-full bg-white/5 border border-white/10 rounded-lg px-3 py-2 text-sm text-white/70 placeholder-white/20 outline-none focus:border-primary/40" placeholder="Type any artist, genre, or style..." value={styleInput} onChange={e=>setStyleInput(e.target.value)}/>
            <button onClick={handleStyleTransfer} disabled={loading} className="flex items-center gap-2 px-5 py-3 rounded-xl bg-primary/20 border border-primary/30 text-primary font-bold text-sm hover:bg-primary/30 transition-all disabled:opacity-50">
              {loading?<Loader2 className="w-4 h-4 animate-spin"/>:<Wand2 className="w-4 h-4"/>} Transfer Style
            </button>
          </div>
        )}
        {activeTab==='morph'&&(
          <div className="max-w-lg space-y-4">
            <div className="p-4 rounded-xl bg-white/3 border border-white/5"><div className="text-sm font-bold text-white/80">Genre Morph</div><div className="text-xs text-white/40 mt-1">Blend two genres into one unique beat.</div></div>
            <div className="grid grid-cols-2 gap-3">
              <div className="space-y-1.5"><label className="text-[10px] text-white/30 uppercase tracking-wider">Genre 1</label>
                <Select value={morphGenre1} onValueChange={setMorphGenre1}><SelectTrigger className="bg-white/5 border-white/10 text-white/70 text-sm capitalize"><SelectValue/></SelectTrigger><SelectContent>{AI_GENRES.map(g=><SelectItem key={g} value={g} className="capitalize">{g.replace(/_/g,' ')}</SelectItem>)}</SelectContent></Select>
              </div>
              <div className="space-y-1.5"><label className="text-[10px] text-white/30 uppercase tracking-wider">Genre 2</label>
                <Select value={morphGenre2} onValueChange={setMorphGenre2}><SelectTrigger className="bg-white/5 border-white/10 text-white/70 text-sm capitalize"><SelectValue/></SelectTrigger><SelectContent>{AI_GENRES.map(g=><SelectItem key={g} value={g} className="capitalize">{g.replace(/_/g,' ')}</SelectItem>)}</SelectContent></Select>
              </div>
            </div>
            <div className="space-y-1.5">
              <div className="flex justify-between text-[10px]"><span style={{color:'#8b5cf6'}}>{morphGenre1.replace(/_/g,' ')} {morphBlend}%</span><span style={{color:'#06b6d4'}}>{100-morphBlend}% {morphGenre2.replace(/_/g,' ')}</span></div>
              <input type="range" min={0} max={100} value={morphBlend} onChange={e=>setMorphBlend(+e.target.value)} className="w-full accent-primary"/>
            </div>
            <button onClick={handleGenreMorph} disabled={loading} className="flex items-center gap-2 px-5 py-3 rounded-xl bg-primary/20 border border-primary/30 text-primary font-bold text-sm hover:bg-primary/30 transition-all disabled:opacity-50">
              {loading?<Loader2 className="w-4 h-4 animate-spin"/>:<Shuffle className="w-4 h-4"/>} Morph Genres
            </button>
          </div>
        )}
        {activeTab==='arranger'&&(
          <div className="max-w-lg space-y-4">
            <div className="p-4 rounded-xl bg-white/3 border border-white/5"><div className="text-sm font-bold text-white/80">AI Song Arranger</div><div className="text-xs text-white/40 mt-1 leading-relaxed">AI creates a complete song arrangement using patterns A/B/C/D with intro, verse, build, drop, bridge, outro.</div></div>
            <button onClick={handleAIArranger} disabled={loading} className="flex items-center gap-2 px-5 py-3 rounded-xl bg-primary/20 border border-primary/30 text-primary font-bold text-sm hover:bg-primary/30 transition-all disabled:opacity-50">
              {loading?<Loader2 className="w-4 h-4 animate-spin"/>:<Music className="w-4 h-4"/>} Arrange My Song
            </button>
          </div>
        )}
        {activeTab==='chat'&&(
          <div className="flex flex-col h-full" style={{minHeight:400}}>
            <div className="flex-1 space-y-3 mb-4 overflow-y-auto max-h-80">
              {chatMessages.map((msg,i)=>(
                <div key={i} className={cn("flex",msg.role==='user'?'justify-end':'justify-start')}>
                  <div className={cn("max-w-xs rounded-xl px-3 py-2 text-sm",msg.role==='user'?"bg-primary/20 text-primary/90 border border-primary/20":"bg-white/5 text-white/70 border border-white/8")}>{msg.content}</div>
                </div>
              ))}
              {loading&&<div className="flex justify-start"><div className="bg-white/5 rounded-xl px-3 py-2 border border-white/8"><Loader2 className="w-4 h-4 animate-spin text-white/40"/></div></div>}
              <div ref={chatEndRef}/>
            </div>
            <div className="flex gap-2">
              <input className="flex-1 bg-white/5 border border-white/10 rounded-xl px-3 py-2 text-sm text-white/70 placeholder-white/20 outline-none focus:border-primary/40" placeholder="Tell the AI what to do with your beat..." value={chatInput} onChange={e=>setChatInput(e.target.value)} onKeyDown={e=>e.key==='Enter'&&!e.shiftKey&&handleChat()}/>
              <button onClick={handleChat} disabled={loading||!chatInput.trim()} className="px-3 py-2 rounded-xl bg-primary/20 border border-primary/30 text-primary hover:bg-primary/30 transition-all disabled:opacity-50"><Send className="w-4 h-4"/></button>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}

const STORAGE_KEY='beatforge_projects_v2';
function getProjects(){ try{return JSON.parse(localStorage.getItem(STORAGE_KEY)||'{}');}catch{return {};} }

function ProjectManager({ open,onClose,onSave,onLoad }) {
  const [name,setName]=useState('Untitled Project');
  const [projects,setProjects]=useState({});
  useEffect(()=>{ if(open) setProjects(getProjects()); },[open]);
  const handleSave=()=>{
    const trimmed=name.trim(); if(!trimmed){toast.error("Enter a project name");return;}
    const data=onSave(), all=getProjects();
    all[trimmed]={...data,name:trimmed,savedAt:Date.now()};
    localStorage.setItem(STORAGE_KEY,JSON.stringify(all));
    setProjects({...all}); toast.success(`Saved "${trimmed}"`);
  };
  const handleLoad=(key)=>{ const all=getProjects(); if(all[key]){onLoad(all[key]);toast.success(`Loaded "${key}"`);onClose();} };
  const handleDelete=(key)=>{ const all=getProjects(); delete all[key]; localStorage.setItem(STORAGE_KEY,JSON.stringify(all)); setProjects({...all}); };
  const sorted=Object.entries(projects).sort((a,b)=>(b[1].savedAt||0)-(a[1].savedAt||0));
  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="bg-[#0e0e16] border-white/10 text-white max-w-md">
        <DialogHeader><DialogTitle className="text-sm font-black tracking-wider flex items-center gap-2"><FolderOpen className="w-4 h-4 text-primary"/> Project Manager</DialogTitle></DialogHeader>
        <div className="flex gap-2">
          <Input value={name} onChange={e=>setName(e.target.value)} placeholder="Project name..." className="bg-white/5 border-white/10 text-white text-sm h-8 flex-1" onKeyDown={e=>e.key==='Enter'&&handleSave()}/>
          <button onClick={handleSave} className="flex items-center gap-1.5 h-8 px-3 rounded-md bg-primary/20 border border-primary/40 text-primary hover:bg-primary/30 text-xs font-bold transition-all"><Save className="w-3 h-3"/> Save</button>
        </div>
        <div className="space-y-1 max-h-80 overflow-y-auto pr-1">
          {sorted.length===0&&<div className="text-center text-white/20 text-xs py-8">No saved projects yet.<br/>Save your first project above.</div>}
          {sorted.map(([key,data])=>(
            <div key={key} className="flex items-center gap-2 p-2.5 rounded-lg bg-white/4 hover:bg-white/7 border border-white/5 transition-colors group">
              <div className="flex-1 min-w-0">
                <div className="text-xs font-semibold text-white/80 truncate">{key}</div>
                <div className="text-[9px] text-white/30 flex items-center gap-1 mt-0.5"><Clock className="w-2.5 h-2.5"/>{data.bpm} BPM · {data.musicalKey} · {data.steps} steps · {new Date(data.savedAt).toLocaleDateString()}</div>
              </div>
              <button onClick={()=>handleLoad(key)} className="text-[9px] font-bold text-primary/70 hover:text-primary border border-primary/20 hover:border-primary/50 px-2.5 py-1 rounded transition-all">Load</button>
              <button onClick={()=>handleDelete(key)} className="text-white/20 hover:text-red-400 transition-colors p-1 opacity-0 group-hover:opacity-100"><Trash2 className="w-3 h-3"/></button>
            </div>
          ))}
        </div>
        <div className="text-[9px] text-white/20 text-center">Projects are saved locally in your browser</div>
      </DialogContent>
    </Dialog>
  );
}

// ─────────────────────────────────────────────────────────────
//  MAIN STUDIO PAGE
// ─────────────────────────────────────────────────────────────
const TABS=['sequencer','piano_roll','arranger','ai','synth'];
const TAB_LABELS={ sequencer:'Step Seq', piano_roll:'Piano Roll', arranger:'Arranger', ai:'AI Studio', synth:'Synth' };
const TAB_HINTS={
  sequencer:'Click=toggle · Drum right-click=velocity cycle · Melodic right-click=change note',
  piano_roll:'Click=add note · Right-click=remove · Drag right edge=resize',
  arranger:'Select pattern → click bar to place/remove block',
  ai:'', synth:''
};

export default function Studio() {
  const [bpm,setBpm]=useState(140);
  const [activePattern,setActivePattern]=useState('A');
  const [steps,setSteps]=useState(STEPS_DEFAULT);
  const [musicalKey,setMusicalKey]=useState('C Minor');
  const [genre,setGenre]=useState('trap');
  const [mood]=useState('energetic');
  const [swing,setSwing]=useState(0);
  const [patterns,setPatterns]=useState(()=>buildInitialPatterns(STEPS_DEFAULT));
  const [mixer,setMixer]=useState(buildDefaultMixer);
  const [pianoRollNotes,setPianoRollNotes]=useState(buildInitialPianoRoll);
  const [arrangement,setArrangement]=useState([]);
  const [arrangeMode,setArrangeMode]=useState(false);
  const [selectedChannel,setSelectedChannel]=useState(null);
  const [pianoRollInst,setPianoRollInst]=useState('lead');
  const [isGenerating,setIsGenerating]=useState(false);
  const [isExporting,setIsExporting]=useState(false);
  const [isStemsExporting,setIsStemsExporting]=useState(false);
  const [activeTab,setActiveTab]=useState('sequencer');
  const [showMixer,setShowMixer]=useState(true);
  const [showFX,setShowFX]=useState(false);
  const [showProjects,setShowProjects]=useState(false);
  const undoStack=useRef([]), redoStack=useRef([]);
  const { isPlaying,currentStep,play,stop,updateRefs }=useAudioEngine();

  useEffect(()=>{ updateRefs(patterns,mixer,bpm,activePattern,steps,musicalKey,mixer.__master?.volume,swing,pianoRollNotes,arrangement,arrangeMode); },[patterns,mixer,bpm,activePattern,steps,musicalKey,swing,pianoRollNotes,arrangement,arrangeMode,updateRefs]);

  const pushUndo=useCallback(()=>{ undoStack.current.push(JSON.parse(JSON.stringify(patterns))); if(undoStack.current.length>50)undoStack.current.shift(); redoStack.current=[]; },[patterns]);
  const handleUndo=useCallback(()=>{ if(!undoStack.current.length)return; redoStack.current.push(JSON.parse(JSON.stringify(patterns))); setPatterns(undoStack.current.pop()); },[patterns]);
  const handleRedo=useCallback(()=>{ if(!redoStack.current.length)return; undoStack.current.push(JSON.parse(JSON.stringify(patterns))); setPatterns(redoStack.current.pop()); },[patterns]);

  useEffect(()=>{
    const handler=e=>{
      if((e.ctrlKey||e.metaKey)&&e.key==='z'){e.preventDefault();handleUndo();}
      if((e.ctrlKey||e.metaKey)&&(e.key==='y'||(e.shiftKey&&e.key==='z'))){e.preventDefault();handleRedo();}
    };
    window.addEventListener('keydown',handler);
    return ()=>window.removeEventListener('keydown',handler);
  },[handleUndo,handleRedo]);

  const toggleStep=useCallback((instId,stepIdx)=>{
    pushUndo();
    setPatterns(prev=>{ const pat={...prev}; const track=[...(pat[activePattern][instId]||[])]; track[stepIdx]={...track[stepIdx],active:!track[stepIdx].active}; pat[activePattern]={...pat[activePattern],[instId]:track}; return pat; });
  },[activePattern,pushUndo]);
  const setNote=useCallback((instId,stepIdx,note)=>{ setPatterns(prev=>{ const pat={...prev}; const track=[...(pat[activePattern][instId]||[])]; track[stepIdx]={...track[stepIdx],note}; pat[activePattern]={...pat[activePattern],[instId]:track}; return pat; }); },[activePattern]);
  const setVelocity=useCallback((instId,stepIdx,velocity)=>{ setPatterns(prev=>{ const pat={...prev}; const track=[...(pat[activePattern][instId]||[])]; track[stepIdx]={...track[stepIdx],velocity}; pat[activePattern]={...pat[activePattern],[instId]:track}; return pat; }); },[activePattern]);
  const updateMixer=useCallback((chanId,key,value)=>setMixer(prev=>({...prev,[chanId]:{...(prev[chanId]||{}),[key]:value}})),[]);
  const handleStepsChange=useCallback((newSteps)=>{
    setSteps(newSteps);
    setPatterns(prev=>{ const updated={}; ALL_PATTERNS.forEach(p=>{ const newPat={}; INSTRUMENTS.forEach(inst=>{ const old=prev[p]?.[inst.id]||[]; newPat[inst.id]=newSteps>old.length?[...old,...defaultTrack(newSteps-old.length)]:old.slice(0,newSteps); }); updated[p]=newPat; }); return updated; });
  },[]);
  const handleAIGenerate=useCallback(async()=>{
    setIsGenerating(true); pushUndo();
    try { const {pattern,description}=await generateAIPattern({genre,mood,bpm,musicalKey,steps}); setPatterns(prev=>({...prev,[activePattern]:pattern})); toast.success(`AI pattern ready! ${description?.slice(0,70)||''}`); }
    catch(e){toast.error("AI generation failed.");}
    finally{setIsGenerating(false);}
  },[genre,mood,bpm,musicalKey,steps,activePattern,pushUndo]);
  const handleExport=useCallback(async()=>{
    setIsExporting(true);
    try{await exportWav({patterns,mixer,bpm,steps,musicalKey,activePattern,swing});toast.success("WAV exported!");}
    catch(e){toast.error("Export failed: "+e.message);}
    finally{setIsExporting(false);}
  },[patterns,mixer,bpm,steps,musicalKey,activePattern,swing]);
  const handleExportStems=useCallback(async()=>{
    setIsStemsExporting(true); toast("Exporting stems — this may take a moment...");
    try{await exportStems({patterns,mixer,bpm,steps,musicalKey,activePattern,swing});toast.success("All stems exported!");}
    catch(e){toast.error("Stems export failed: "+e.message);}
    finally{setIsStemsExporting(false);}
  },[patterns,mixer,bpm,steps,musicalKey,activePattern,swing]);
  const handlePianoRollChange=useCallback((notes)=>setPianoRollNotes(prev=>({...prev,[activePattern]:{...(prev[activePattern]||{}),[pianoRollInst]:notes}})),[activePattern,pianoRollInst]);
  const handleSaveProject=useCallback(()=>({bpm,musicalKey,genre,mood,swing,steps,activePattern,patterns,mixer,pianoRollNotes,arrangement}),[bpm,musicalKey,genre,mood,swing,steps,activePattern,patterns,mixer,pianoRollNotes,arrangement]);
  const handleLoadProject=useCallback((data)=>{
    if(data.bpm)setBpm(data.bpm); if(data.musicalKey)setMusicalKey(data.musicalKey); if(data.genre)setGenre(data.genre);
    if(data.swing!==undefined)setSwing(data.swing); if(data.steps)setSteps(data.steps);
    if(data.activePattern)setActivePattern(data.activePattern); if(data.patterns)setPatterns(data.patterns);
    if(data.mixer)setMixer(data.mixer); if(data.pianoRollNotes)setPianoRollNotes(data.pianoRollNotes);
    if(data.arrangement)setArrangement(data.arrangement);
    undoStack.current=[]; redoStack.current=[];
  },[]);

  const pianoRollNotesForCurrent=pianoRollNotes[activePattern]?.[pianoRollInst]||[];

  return (
    <div className="h-screen flex flex-col bg-[#080810] text-white overflow-hidden font-inter">
      <Transport
        isPlaying={isPlaying} bpm={bpm} onBpmChange={setBpm}
        activePattern={activePattern} onPatternChange={setActivePattern}
        steps={steps} onStepsChange={handleStepsChange}
        musicalKey={musicalKey} onKeyChange={setMusicalKey}
        genre={genre} onGenreChange={setGenre}
        masterVol={mixer.__master?.volume??0.75} onMasterVolChange={v=>updateMixer('__master','volume',v)}
        swing={swing} onSwingChange={setSwing}
        onPlay={play} onStop={stop}
        onAIGenerate={handleAIGenerate} isGenerating={isGenerating}
        onExport={handleExport} isExporting={isExporting}
        onExportStems={handleExportStems} isStemsExporting={isStemsExporting}
        onUndo={handleUndo} onRedo={handleRedo}
        undoAvailable={undoStack.current.length>0} redoAvailable={redoStack.current.length>0}
        arrangeMode={arrangeMode} onArrangeModeToggle={()=>setArrangeMode(v=>!v)}
        onOpenProjects={()=>setShowProjects(true)}
      />

      <div className="flex items-center gap-1 px-4 py-1.5 bg-[#0a0a10] border-b border-white/5">
        {TABS.map(tab=>(
          <button key={tab} onClick={()=>setActiveTab(tab)} className={cn("px-3 py-1 rounded-md text-[10px] font-bold uppercase tracking-wider transition-all border",activeTab===tab?"bg-primary/20 border-primary/30 text-primary":"border-transparent text-white/30 hover:text-white/60")}>
            {TAB_LABELS[tab]}
          </button>
        ))}
        <div className="ml-auto text-[9px] text-white/15 truncate max-w-xs">{TAB_HINTS[activeTab]}</div>
      </div>

      <div className="flex-1 flex flex-col overflow-hidden">
        {activeTab==='sequencer'&&<StepSequencer patterns={patterns} activePattern={activePattern} currentStep={isPlaying?currentStep:-1} steps={steps} onToggleStep={toggleStep} onSetNote={setNote} onSetVelocity={setVelocity} selectedChannel={selectedChannel} onSelectChannel={setSelectedChannel}/>}
        {activeTab==='piano_roll'&&<PianoRoll notes={pianoRollNotesForCurrent} onNotesChange={handlePianoRollChange} steps={steps} musicalKey={musicalKey} activePattern={activePattern} activeInstrument={pianoRollInst} onInstrumentChange={setPianoRollInst} bpm={bpm}/>}
        {activeTab==='arranger'&&<SongArranger arrangement={arrangement} onUpdate={setArrangement} activePattern={activePattern} isPlaying={isPlaying} currentBar={0}/>}
        {activeTab==='ai'&&<AIPanel patterns={patterns} mixer={mixer} bpm={bpm} musicalKey={musicalKey} genre={genre} mood={mood} steps={steps} activePattern={activePattern} onPatternsChange={setPatterns} onMixerChange={setMixer} onArrangementChange={setArrangement}/>}
        {activeTab==='synth'&&<SynthPanel mixer={mixer} onUpdate={updateMixer} selectedChannel={selectedChannel}/>}

        <AnimatePresence>
          {selectedChannel&&showFX&&activeTab==='sequencer'&&(
            <motion.div initial={{height:0,opacity:0}} animate={{height:'auto',opacity:1}} exit={{height:0,opacity:0}} transition={{duration:0.2}} className="overflow-hidden flex-shrink-0">
              <EffectsRack channelId={selectedChannel} effects={mixer[selectedChannel]||{}} onUpdate={(k,v)=>updateMixer(selectedChannel,k,v)} onClose={()=>setShowFX(false)}/>
            </motion.div>
          )}
        </AnimatePresence>
        {selectedChannel&&activeTab==='sequencer'&&!showFX&&(
          <div className="flex-shrink-0 border-t border-white/5 bg-[#0d0d12] px-4 py-1.5">
            <button onClick={()=>setShowFX(true)} className="text-[10px] text-white/30 hover:text-primary uppercase tracking-wider transition-colors">
              ▲ Open FX Rack — {INSTRUMENTS.find(i=>i.id===selectedChannel)?.name}
            </button>
          </div>
        )}
      </div>

      <AnimatePresence>
        {showMixer&&(
          <motion.div initial={{height:0}} animate={{height:'auto'}} exit={{height:0}} className="overflow-hidden flex-shrink-0">
            <MixerPanel mixer={mixer} onUpdate={updateMixer} selectedChannel={selectedChannel} onSelectChannel={setSelectedChannel}/>
          </motion.div>
        )}
      </AnimatePresence>
      <button onClick={()=>setShowMixer(v=>!v)} className="flex-shrink-0 bg-[#0a0a10] border-t border-white/5 text-[9px] text-white/20 hover:text-white/50 uppercase tracking-widest py-1.5 transition-colors">
        {showMixer?'▼ Hide Mixer':'▲ Show Mixer'}
      </button>

      <ProjectManager open={showProjects} onClose={()=>setShowProjects(false)} onSave={handleSaveProject} onLoad={handleLoadProject}/>
    </div>
  );
}
