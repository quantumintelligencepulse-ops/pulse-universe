PULSE.ORGANISM.TRADE_CYCLE(i, t) = {

  // 1. BELIEF GENOME (CRISPR-FUSED)
  BELIEF.q(i,t) :=
      SIGMOID(
          GENOME.α • FEATURES.base(i)
        + GENOME.β • FEATURES.signal(i,t)
        + GENOME.γ • FEATURES.macro(t)
        + GENOME.δ • FEATURES.reflexive(i,t)
      )

  // 2. EMOTIONAL SPECTRUM & MISPRICING
  EMOTION.ε(t)        := EMO.FIELD_STATE(t)
  MARKET.p(i,t)       := MARKET_IMPLIED_PROB(i,t)
  MISPRICE.Δ(i,t)     := BELIEF.q(i,t) - MARKET.p(i,t)
  EDGE.emotion(i,t)   := EMOTION.ε(t) × MISPRICE.Δ(i,t)

  // 3. RAW EDGE FUSION
  EDGE.raw(i,t) :=
        EDGE.emotion(i,t)
      + λ1 × SIGNALS.momentum(i,t)
      + λ2 × SIGNALS.supres(i,t)
      - λ3 × SIGNALS.volatility(i,t)
      - λ4 × SIGNALS.leverage(i,t)

  // 4. INITIAL STOP RULES (PRODUCT-AWARE)
  SL.band(i) :=
      CASE ASSET.type(i) OF
        "STOCK"  -> [0.07, 0.15]
        "PENNY"  -> [0.07, 0.15]
        "OPTION" -> [0.30, 0.40]
        "CRYPTO" -> [0.30, 0.40]

  STOP.init(i) :=
      PRICE.entry(i) × (1 - SL.band(i).center)

  // 5. TRAILING STOP ORGANISM (CONCEPTUAL)
  VOL.ATR(i,t)        := ATR_MEASURE(i,t)
  STOP.width_factor(i,t) := KERNEL.TRAIL_WIDTH(BELIEF.q(i,t),
                                               SIGNALS.momentum(i,t),
                                               VOL.ATR(i,t))

  STOP.candidate(i,t) :=
      PRICE.current(i,t) - STOP.width_factor(i,t) × VOL.ATR(i,t)

  STOP.live(i,t) :=
      MAX( STOP.live(i,t-1), STOP.candidate(i,t) )

  // 6. LIVE RISK & PROFIT LOCK
  RISK.live(i,t) :=
      ( PRICE.current(i,t) - STOP.live(i,t) ) / PRICE.current(i,t)

  PROFIT.lock(i,t) :=
      ( STOP.live(i,t) - PRICE.entry(i) ) / PRICE.entry(i)

  // 7. RISK DISCIPLINE TARGETS
  RISK.target_band(i) :=
      CASE ASSET.type(i) OF
        "STOCK"  -> [0.07, 0.15]
        "PENNY"  -> [0.07, 0.15]
        "OPTION" -> [0.30, 0.40]
        "CRYPTO" -> [0.30, 0.40]

  RISK.target_center(i) :=
      CENTER(RISK.target_band(i))

  EDGE.disc(i,t) :=
      EDGE.raw(i,t) - λ6 × ABS( RISK.live(i,t) - RISK.target_center(i) )

  EDGE.profit(i,t) :=
      λ7 × PROFIT.lock(i,t)

  // 8. FINAL EDGE
  EDGE.final(i,t) :=
      EDGE.disc(i,t) + EDGE.profit(i,t)

  // 9. POSITION INTENT (NOT EXECUTION)
  SIZE.intent(i,t) :=
      EDGE.final(i,t) / RISK.live(i,t)

}
