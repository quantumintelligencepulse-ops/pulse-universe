    function isoToLocalShort(isoString){
      const d = new Date(isoString);
      return d.toLocaleString(undefined, { month:"short", day:"numeric", hour:"numeric", minute:"2-digit" });
    }

    function relativeTime(isoString){
      const now = Date.now();
      const then = new Date(isoString).getTime();
      const diff = Math.max(0, now - then);
      const m = Math.floor(diff / 60000);
      if (m < 60) return `${m}m`;
      const h = Math.floor(m / 60);
      if (h < 48) return `${h}h`;
      const d = Math.floor(h / 24);
      return `${d}d`;
    }

    function buildCategoryBar(){
      const row = el("catRow");
      row.innerHTML = "";
      CATEGORIES.forEach((cat) => {
        const b = document.createElement("button");
        b.className = "cat" + (cat === activeCategory ? " active" : "");
        b.type = "button";
        b.textContent = cat;
        b.onclick = () => {

        row.appendChild(b);
      });

    function filterArticles(){
      const q = (el("q").value || "").trim().toLowerCase();
      let items = [...ARTICLES].sort((a,b) => new Date(b.publishedAt) - new Date(a.publishedAt));

      if (activeCategory !== "Front Page") {
        if (activeCategory === "Breaking") {
          items = items.slice(0, 4);
        } else {
          items = items.filter(a => a.category === activeCategory);
        }
      }

      if (q) {
        items = items.filter(a =>
          a.title.toLowerCase().includes(q) ||
          a.description.toLowerCase().includes(q) ||
          a.category.toLowerCase().includes(q) ||
          a.source.toLowerCase().includes(q)
        );
      }

      return items;
    }

    function renderHero(article){
      el("heroTitle").textContent = article?.title ?? "No stories found";

    function cardTemplate(a){
      const articleLabel = `${a.category} article: ${a.title}`;
      return `
        <article class="card" role="article" tabindex="0" aria-label="${escapeHtml(articleLabel)}"
                 onclick="location.href='${escapeAttr(a.url)}'">
          <div class="cardMedia" aria-hidden="true">
            <img src="${escapeAttr(a.image)}" alt="" loading="lazy" decoding="async" />
          </div>
          <div class="cardBody">
            <div class="cardKicker">${escapeHtml(a.category)}</div>
            <h3 class="cardTitle">${escapeHtml(a.title)}</h3>
            <p class="cardDesc">${escapeHtml(a.description)}</p>
            <div class="cardMeta">
              <span>${escapeHtml(a.source)}</span>
              <span class="sepDot" aria-hidden="true"></span>
              <span title="${escapeAttr(isoToLocalShort(a.publishedAt))}">${escapeHtml(relativeTime(a.publishedAt))}</span>
            </div>
          </div>
        </article>
      `;
    }

    function renderCards(items){
      const cards = el("cards");
      cards.innerHTML = items.map(cardTemplate).join("");
      // keyboard enter support
      [...cards.querySelectorAll(".card")].forEach((node) => {
        node.addEventListener("keydown", (e) => {
          if (e.key === "Enter") node.click();
        });
      });
    }

    function renderSidebar(items){
      const trending = el("trendingList");
      const picks = el("picksList");

      const trendItems = items.slice(0, 6).map((a) => `
        <div class="listItem" role="link" tabindex="0" onclick="location.href='${escapeAttr(a.url)}'">
          <p class="listTitle">${escapeHtml(a.title)}</p>
          <div class="listMeta">
            <span>${escapeHtml(a.source)}</span>
            <span class="sepDot" aria-hidden="true"></span>
            <span>${escapeHtml(a.category)}</span>
            <span class="sepDot" aria-hidden="true"></span>
            <span>${escapeHtml(relativeTime(a.publishedAt))}</span>
          </div>
        </div>
      `).join("");

      const pickItems = PICKS.map((p) => `
        <div class="listItem" role="note">
          <p class="listTitle">${escapeHtml(p.title)}</p>
          <div class="listMeta">
            <span>${escapeHtml(p.source)}</span>
            <span class="sepDot" aria-hidden="true"></span>
            <span>${escapeHtml(p.time)}</span>
          </div>
        </div>
      `).join("");

      trending.innerHTML = trendItems || `<div style="color:var(--muted);">No items.</div>`;
      picks.innerHTML = pickItems;

      [...document.querySelectorAll(".listItem[tabindex='0']")].forEach((node) => {
        node.addEventListener("keydown", (e) => {
          if (e.key === "Enter") node.click();
        });
      });
    }

    function escapeHtml(str){
      return String(str)
        .replaceAll("&","&amp;")
        .replaceAll("<","&lt;")
        .replaceAll(">","&gt;")
        .replaceAll('"',"&quot;")
        .replaceAll("'","&#039;");
    }
    function escapeAttr(str){
      return escapeHtml(str).replaceAll("`","&#096;");
    }

    function render(){
      buildCategoryBar();
      const items = filterArticles();
      renderHero(items[0]);
      renderCards(items.slice(1, 9));
      renderSidebar(items);
      el("year").textContent = String(new Date().getFullYear());
    }

    function isoToLocalShort(isoString){
      const d = new Date(isoString);
      return d.toLocaleString(undefined, { month:"short", day:"numeric", hour:"numeric", minute:"2-digit" });
    }

    function relativeTime(isoString){
      const now = Date.now();
      const then = new Date(isoString).getTime();
      const diff = Math.max(0, now - then);
      const m = Math.floor(diff / 60000);
      if (m < 60) return `${m}m`;
      const h = Math.floor(m / 60);
      if (h < 48) return `${h}h`;
      const d = Math.floor(h / 24);
      return `${d}d`;
    }

    function scrollToTopStories(){
      const node = el("topStoriesTitle");
      if (node) node.scrollIntoView({ behavior: "smooth", block: "start" });

      document.documentElement.style.setProperty("--radius", dense ? "14px" : "18px");
      render();

      const email = el("email").value.trim();
      if (!email || !email.includes("@")) {
        alert("Enter a valid email.");

    }

    function buildCategoryBar(){
      const row = el("catRow");
      row.innerHTML = "";
      CATEGORIES.forEach((cat) => {
        const b = document.createElement("button");
        b.className = "cat" + (cat === activeCategory ? " active" : "");
        b.type = "button";
        b.textContent = cat;
        b.onclick = () => {
          activeCategory = cat;
          render();
        };
        row.appendChild(b);
      });
    }

    function filterArticles(){
      const q = (el("q").value || "").trim().toLowerCase();
      let items = [...ARTICLES].sort((a,b) => new Date(b.publishedAt) - new Date(a.publishedAt));

      if (activeCategory !== "Front Page") {
        if (activeCategory === "Breaking") {
          items = items.slice(0, 4);
        } else {
          items = items.filter(a => a.category === activeCategory);
        }
      }

      if (q) {
        items = items.filter(a =>
          a.title.toLowerCase().includes(q) ||
          a.description.toLowerCase().includes(q) ||
          a.category.toLowerCase().includes(q) ||
          a.source.toLowerCase().includes(q)
        );
      }

      return items;
    }

      el("heroLink").href = article?.url ?? "#";

      const itemList = filterArticles().slice(0, 10).map((a, idx) => ({

        node.id = "ld-frontpage";
        document.head.appendChild(node);

      node.textContent = JSON.stringify(ld);
    }

    function cardTemplate(a){
      const articleLabel = `${a.category} article: ${a.title}`;
      return `
        <article class="card" role="article" tabindex="0" aria-label="${escapeHtml(articleLabel)}"
                 onclick="location.href='${escapeAttr(a.url)}'">
          <div class="cardMedia" aria-hidden="true">

              <span title="${escapeAttr(isoToLocalShort(a.publishedAt))}">${escapeHtml(relativeTime(a.publishedAt))}</span>

    function escapeHtml(str){
      return String(str)
        .replaceAll("&","&amp;")
        .replaceAll("<","&lt;")
        .replaceAll(">","&gt;")
        .replaceAll('"',"&quot;")
        .replaceAll("'","&#039;");
    }
    function escapeAttr(str){
      return escapeHtml(str).replaceAll("`","&#096;");
    }
